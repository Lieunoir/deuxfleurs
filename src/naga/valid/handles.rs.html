15impl super::Validator {
16    /// Validates that all handles within `module` are:
17    ///
18    /// * Valid, in the sense that they contain indices within each arena structure inside the
19    ///   [`crate::Module`] type.
20    /// * No arena contents contain any items that have forward dependencies; that is, the value
21    ///   associated with a handle only may contain references to handles in the same arena that
22    ///   were constructed before it.
23    ///
24    /// By validating the above conditions, we free up subsequent logic to assume that handle
25    /// accesses are infallible.
26    ///
27    /// # Errors
28    ///
29    /// Errors returned by this method are intentionally sparse, for simplicity of implementation.
30    /// It is expected that only buggy frontends or fuzzers should ever emit IR that fails this
31    /// validation pass.
32    pub(super) fn validate_module_handles(module: &crate::Module) -> Result<(), ValidationError> {
33        let &crate::Module {
34            ref constants,
35            ref overrides,
36            ref entry_points,
37            ref functions,
38            ref global_variables,
39            ref types,
40            ref special_types,
41            ref global_expressions,
42        } = module;
43
44        // NOTE: Types being first is important. All other forms of validation depend on this.
45        for (this_handle, ty) in types.iter() {
46            match ty.inner {
47                crate::TypeInner::Scalar { .. }
48                | crate::TypeInner::Vector { .. }
49                | crate::TypeInner::Matrix { .. }
50                | crate::TypeInner::ValuePointer { .. }
51                | crate::TypeInner::Atomic { .. }
52                | crate::TypeInner::Image { .. }
53                | crate::TypeInner::Sampler { .. }
54                | crate::TypeInner::AccelerationStructure
55                | crate::TypeInner::RayQuery => (),
56                crate::TypeInner::Pointer { base, space: _ } => {
57                    this_handle.check_dep(base)?;
58                }
59                crate::TypeInner::Array { base, .. }
60                | crate::TypeInner::BindingArray { base, .. } => {
61                    this_handle.check_dep(base)?;
62                }
63                crate::TypeInner::Struct {
64                    ref members,
65                    span: _,
66                } => {
67                    this_handle.check_dep_iter(members.iter().map(|m| m.ty))?;
68                }
69            }
70        }
71
72        for handle_and_expr in global_expressions.iter() {
73            Self::validate_const_expression_handles(handle_and_expr, constants, overrides, types)?;
74        }
75
76        let validate_type = |handle| Self::validate_type_handle(handle, types);
77        let validate_const_expr =
78            |handle| Self::validate_expression_handle(handle, global_expressions);
79
80        for (_handle, constant) in constants.iter() {
81            let &crate::Constant { name: _, ty, init } = constant;
82            validate_type(ty)?;
83            validate_const_expr(init)?;
84        }
85
86        for (_handle, override_) in overrides.iter() {
87            let &crate::Override {
88                name: _,
89                id: _,
90                ty,
91                init,
92            } = override_;
93            validate_type(ty)?;
94            if let Some(init_expr) = init {
95                validate_const_expr(init_expr)?;
96            }
97        }
98
99        for (_handle, global_variable) in global_variables.iter() {
100            let &crate::GlobalVariable {
101                name: _,
102                space: _,
103                binding: _,
104                ty,
105                init,
106            } = global_variable;
107            validate_type(ty)?;
108            if let Some(init_expr) = init {
109                validate_const_expr(init_expr)?;
110            }
111        }
112
113        let validate_function = |function_handle, function: &_| -> Result<_, InvalidHandleError> {
114            let &crate::Function {
115                name: _,
116                ref arguments,
117                ref result,
118                ref local_variables,
119                ref expressions,
120                ref named_expressions,
121                ref body,
122            } = function;
123
124            for arg in arguments.iter() {
125                let &crate::FunctionArgument {
126                    name: _,
127                    ty,
128                    binding: _,
129                } = arg;
130                validate_type(ty)?;
131            }
132
133            if let &Some(crate::FunctionResult { ty, binding: _ }) = result {
134                validate_type(ty)?;
135            }
136
137            for (_handle, local_variable) in local_variables.iter() {
138                let &crate::LocalVariable { name: _, ty, init } = local_variable;
139                validate_type(ty)?;
140                if let Some(init) = init {
141                    Self::validate_expression_handle(init, expressions)?;
142                }
143            }
144
145            for handle in named_expressions.keys().copied() {
146                Self::validate_expression_handle(handle, expressions)?;
147            }
148
149            for handle_and_expr in expressions.iter() {
150                Self::validate_expression_handles(
151                    handle_and_expr,
152                    constants,
153                    overrides,
154                    global_expressions,
155                    types,
156                    local_variables,
157                    global_variables,
158                    functions,
159                    function_handle,
160                )?;
161            }
162
163            Self::validate_block_handles(body, expressions, functions)?;
164
165            Ok(())
166        };
167
168        for entry_point in entry_points.iter() {
169            validate_function(None, &entry_point.function)?;
170        }
171
172        for (function_handle, function) in functions.iter() {
173            validate_function(Some(function_handle), function)?;
174        }
175
176        if let Some(ty) = special_types.ray_desc {
177            validate_type(ty)?;
178        }
179        if let Some(ty) = special_types.ray_intersection {
180            validate_type(ty)?;
181        }
182
183        Ok(())
184    }

221    fn validate_const_expression_handles(
222        (handle, expression): (Handle<crate::Expression>, &crate::Expression),
223        constants: &Arena<crate::Constant>,
224        overrides: &Arena<crate::Override>,
225        types: &UniqueArena<crate::Type>,
226    ) -> Result<(), InvalidHandleError> {
227        let validate_constant = |handle| Self::validate_constant_handle(handle, constants);
228        let validate_override = |handle| Self::validate_override_handle(handle, overrides);
229        let validate_type = |handle| Self::validate_type_handle(handle, types);
230
231        match *expression {
232            crate::Expression::Literal(_) => {}
233            crate::Expression::Constant(constant) => {
234                validate_constant(constant)?;
235                handle.check_dep(constants[constant].init)?;
236            }
237            crate::Expression::Override(override_) => {
238                validate_override(override_)?;
239                if let Some(init) = overrides[override_].init {
240                    handle.check_dep(init)?;
241                }
242            }
243            crate::Expression::ZeroValue(ty) => {
244                validate_type(ty)?;
245            }
246            crate::Expression::Compose { ty, ref components } => {
247                validate_type(ty)?;
248                handle.check_dep_iter(components.iter().copied())?;
249            }
250            _ => {}
251        }
252        Ok(())
253    }

255    #[allow(clippy::too_many_arguments)]
256    fn validate_expression_handles(
257        (handle, expression): (Handle<crate::Expression>, &crate::Expression),
258        constants: &Arena<crate::Constant>,
259        overrides: &Arena<crate::Override>,
260        global_expressions: &Arena<crate::Expression>,
261        types: &UniqueArena<crate::Type>,
262        local_variables: &Arena<crate::LocalVariable>,
263        global_variables: &Arena<crate::GlobalVariable>,
264        functions: &Arena<crate::Function>,
265        // The handle of the current function or `None` if it's an entry point
266        current_function: Option<Handle<crate::Function>>,
267    ) -> Result<(), InvalidHandleError> {
268        let validate_constant = |handle| Self::validate_constant_handle(handle, constants);
269        let validate_override = |handle| Self::validate_override_handle(handle, overrides);
270        let validate_const_expr =
271            |handle| Self::validate_expression_handle(handle, global_expressions);
272        let validate_type = |handle| Self::validate_type_handle(handle, types);
273
274        match *expression {
275            crate::Expression::Access { base, index } => {
276                handle.check_dep(base)?.check_dep(index)?;
277            }
278            crate::Expression::AccessIndex { base, .. } => {
279                handle.check_dep(base)?;
280            }
281            crate::Expression::Splat { value, .. } => {
282                handle.check_dep(value)?;
283            }
284            crate::Expression::Swizzle { vector, .. } => {
285                handle.check_dep(vector)?;
286            }
287            crate::Expression::Literal(_) => {}
288            crate::Expression::Constant(constant) => {
289                validate_constant(constant)?;
290            }
291            crate::Expression::Override(override_) => {
292                validate_override(override_)?;
293            }
294            crate::Expression::ZeroValue(ty) => {
295                validate_type(ty)?;
296            }
297            crate::Expression::Compose { ty, ref components } => {
298                validate_type(ty)?;
299                handle.check_dep_iter(components.iter().copied())?;
300            }
301            crate::Expression::FunctionArgument(_arg_idx) => (),
302            crate::Expression::GlobalVariable(global_variable) => {
303                global_variable.check_valid_for(global_variables)?;
304            }
305            crate::Expression::LocalVariable(local_variable) => {
306                local_variable.check_valid_for(local_variables)?;
307            }
308            crate::Expression::Load { pointer } => {
309                handle.check_dep(pointer)?;
310            }
311            crate::Expression::ImageSample {
312                image,
313                sampler,
314                gather: _,
315                coordinate,
316                array_index,
317                offset,
318                level,
319                depth_ref,
320            } => {
321                if let Some(offset) = offset {
322                    validate_const_expr(offset)?;
323                }
324
325                handle
326                    .check_dep(image)?
327                    .check_dep(sampler)?
328                    .check_dep(coordinate)?
329                    .check_dep_opt(array_index)?;
330
331                match level {
332                    crate::SampleLevel::Auto | crate::SampleLevel::Zero => (),
333                    crate::SampleLevel::Exact(expr) => {
334                        handle.check_dep(expr)?;
335                    }
336                    crate::SampleLevel::Bias(expr) => {
337                        handle.check_dep(expr)?;
338                    }
339                    crate::SampleLevel::Gradient { x, y } => {
340                        handle.check_dep(x)?.check_dep(y)?;
341                    }
342                };
343
344                handle.check_dep_opt(depth_ref)?;
345            }
346            crate::Expression::ImageLoad {
347                image,
348                coordinate,
349                array_index,
350                sample,
351                level,
352            } => {
353                handle
354                    .check_dep(image)?
355                    .check_dep(coordinate)?
356                    .check_dep_opt(array_index)?
357                    .check_dep_opt(sample)?
358                    .check_dep_opt(level)?;
359            }
360            crate::Expression::ImageQuery { image, query } => {
361                handle.check_dep(image)?;
362                match query {
363                    crate::ImageQuery::Size { level } => {
364                        handle.check_dep_opt(level)?;
365                    }
366                    crate::ImageQuery::NumLevels
367                    | crate::ImageQuery::NumLayers
368                    | crate::ImageQuery::NumSamples => (),
369                };
370            }
371            crate::Expression::Unary {
372                op: _,
373                expr: operand,
374            } => {
375                handle.check_dep(operand)?;
376            }
377            crate::Expression::Binary { op: _, left, right } => {
378                handle.check_dep(left)?.check_dep(right)?;
379            }
380            crate::Expression::Select {
381                condition,
382                accept,
383                reject,
384            } => {
385                handle
386                    .check_dep(condition)?
387                    .check_dep(accept)?
388                    .check_dep(reject)?;
389            }
390            crate::Expression::Derivative { expr: argument, .. } => {
391                handle.check_dep(argument)?;
392            }
393            crate::Expression::Relational { fun: _, argument } => {
394                handle.check_dep(argument)?;
395            }
396            crate::Expression::Math {
397                fun: _,
398                arg,
399                arg1,
400                arg2,
401                arg3,
402            } => {
403                handle
404                    .check_dep(arg)?
405                    .check_dep_opt(arg1)?
406                    .check_dep_opt(arg2)?
407                    .check_dep_opt(arg3)?;
408            }
409            crate::Expression::As {
410                expr: input,
411                kind: _,
412                convert: _,
413            } => {
414                handle.check_dep(input)?;
415            }
416            crate::Expression::CallResult(function) => {
417                Self::validate_function_handle(function, functions)?;
418                if let Some(handle) = current_function {
419                    handle.check_dep(function)?;
420                }
421            }
422            crate::Expression::AtomicResult { .. }
423            | crate::Expression::RayQueryProceedResult
424            | crate::Expression::SubgroupBallotResult
425            | crate::Expression::SubgroupOperationResult { .. }
426            | crate::Expression::WorkGroupUniformLoadResult { .. } => (),
427            crate::Expression::ArrayLength(array) => {
428                handle.check_dep(array)?;
429            }
430            crate::Expression::RayQueryGetIntersection {
431                query,
432                committed: _,
433            } => {
434                handle.check_dep(query)?;
435            }
436        }
437        Ok(())
438    }

440    fn validate_block_handles(
441        block: &crate::Block,
442        expressions: &Arena<crate::Expression>,
443        functions: &Arena<crate::Function>,
444    ) -> Result<(), InvalidHandleError> {
445        let validate_block = |block| Self::validate_block_handles(block, expressions, functions);
446        let validate_expr = |handle| Self::validate_expression_handle(handle, expressions);
447        let validate_expr_opt = |handle_opt| {
448            if let Some(handle) = handle_opt {
449                validate_expr(handle)?;
450            }
451            Ok(())
452        };
453
454        block.iter().try_for_each(|stmt| match *stmt {
455            crate::Statement::Emit(ref expr_range) => {
456                expr_range.check_valid_for(expressions)?;
457                Ok(())
458            }
459            crate::Statement::Block(ref block) => {
460                validate_block(block)?;
461                Ok(())
462            }
463            crate::Statement::If {
464                condition,
465                ref accept,
466                ref reject,
467            } => {
468                validate_expr(condition)?;
469                validate_block(accept)?;
470                validate_block(reject)?;
471                Ok(())
472            }
473            crate::Statement::Switch {
474                selector,
475                ref cases,
476            } => {
477                validate_expr(selector)?;
478                for &crate::SwitchCase {
479                    value: _,
480                    ref body,
481                    fall_through: _,
482                } in cases
483                {
484                    validate_block(body)?;
485                }
486                Ok(())
487            }
488            crate::Statement::Loop {
489                ref body,
490                ref continuing,
491                break_if,
492            } => {
493                validate_block(body)?;
494                validate_block(continuing)?;
495                validate_expr_opt(break_if)?;
496                Ok(())
497            }
498            crate::Statement::Return { value } => validate_expr_opt(value),
499            crate::Statement::Store { pointer, value } => {
500                validate_expr(pointer)?;
501                validate_expr(value)?;
502                Ok(())
503            }
504            crate::Statement::ImageStore {
505                image,
506                coordinate,
507                array_index,
508                value,
509            } => {
510                validate_expr(image)?;
511                validate_expr(coordinate)?;
512                validate_expr_opt(array_index)?;
513                validate_expr(value)?;
514                Ok(())
515            }
516            crate::Statement::Atomic {
517                pointer,
518                fun,
519                value,
520                result,
521            } => {
522                validate_expr(pointer)?;
523                match fun {
524                    crate::AtomicFunction::Add
525                    | crate::AtomicFunction::Subtract
526                    | crate::AtomicFunction::And
527                    | crate::AtomicFunction::ExclusiveOr
528                    | crate::AtomicFunction::InclusiveOr
529                    | crate::AtomicFunction::Min
530                    | crate::AtomicFunction::Max => (),
531                    crate::AtomicFunction::Exchange { compare } => validate_expr_opt(compare)?,
532                };
533                validate_expr(value)?;
534                if let Some(result) = result {
535                    validate_expr(result)?;
536                }
537                Ok(())
538            }
539            crate::Statement::WorkGroupUniformLoad { pointer, result } => {
540                validate_expr(pointer)?;
541                validate_expr(result)?;
542                Ok(())
543            }
544            crate::Statement::Call {
545                function,
546                ref arguments,
547                result,
548            } => {
549                Self::validate_function_handle(function, functions)?;
550                for arg in arguments.iter().copied() {
551                    validate_expr(arg)?;
552                }
553                validate_expr_opt(result)?;
554                Ok(())
555            }
556            crate::Statement::RayQuery { query, ref fun } => {
557                validate_expr(query)?;
558                match *fun {
559                    crate::RayQueryFunction::Initialize {
560                        acceleration_structure,
561                        descriptor,
562                    } => {
563                        validate_expr(acceleration_structure)?;
564                        validate_expr(descriptor)?;
565                    }
566                    crate::RayQueryFunction::Proceed { result } => {
567                        validate_expr(result)?;
568                    }
569                    crate::RayQueryFunction::Terminate => {}
570                }
571                Ok(())
572            }
573            crate::Statement::SubgroupBallot { result, predicate } => {
574                validate_expr_opt(predicate)?;
575                validate_expr(result)?;
576                Ok(())
577            }
578            crate::Statement::SubgroupCollectiveOperation {
579                op: _,
580                collective_op: _,
581                argument,
582                result,
583            } => {
584                validate_expr(argument)?;
585                validate_expr(result)?;
586                Ok(())
587            }
588            crate::Statement::SubgroupGather {
589                mode,
590                argument,
591                result,
592            } => {
593                validate_expr(argument)?;
594                match mode {
595                    crate::GatherMode::BroadcastFirst => {}
596                    crate::GatherMode::Broadcast(index)
597                    | crate::GatherMode::Shuffle(index)
598                    | crate::GatherMode::ShuffleDown(index)
599                    | crate::GatherMode::ShuffleUp(index)
600                    | crate::GatherMode::ShuffleXor(index) => validate_expr(index)?,
601                }
602                validate_expr(result)?;
603                Ok(())
604            }
605            crate::Statement::Break
606            | crate::Statement::Continue
607            | crate::Statement::Kill
608            | crate::Statement::Barrier(_) => Ok(()),
609        })
610    }

631#[derive(Clone, Debug, thiserror::Error)]
632#[cfg_attr(test, derive(PartialEq))]
633pub enum InvalidHandleError {
634    #[error(transparent)]
635    BadHandle(#[from] BadHandle),
636    #[error(transparent)]
637    ForwardDependency(#[from] FwdDepError),
638    #[error(transparent)]
639    BadRange(#[from] BadRangeError),
640}
641
642#[derive(Clone, Debug, thiserror::Error)]
643#[cfg_attr(test, derive(PartialEq))]
644#[error(
645    "{subject:?} of kind {subject_kind:?} depends on {depends_on:?} of kind {depends_on_kind}, \
646    which has not been processed yet"
647)]
648pub struct FwdDepError {
649    // This error is used for many `Handle` types, but there's no point in making this generic, so
650    // we just flatten them all to `Handle<()>` here.
651    subject: Handle<()>,
652    subject_kind: &'static str,
653    depends_on: Handle<()>,
654    depends_on_kind: &'static str,
655}
656
657impl<T> Handle<T> {
658    /// Check that `self` is valid within `arena` using [`Arena::check_contains_handle`].
659    pub(self) fn check_valid_for(self, arena: &Arena<T>) -> Result<(), InvalidHandleError> {
660        arena.check_contains_handle(self)?;
661        Ok(())
662    }
663
664    /// Check that `self` is valid within `arena` using [`UniqueArena::check_contains_handle`].
665    pub(self) fn check_valid_for_uniq(
666        self,
667        arena: &UniqueArena<T>,
668    ) -> Result<(), InvalidHandleError>
669    where
670        T: Eq + Hash,
671    {
672        arena.check_contains_handle(self)?;
673        Ok(())
674    }
675
676    /// Check that `depends_on` was constructed before `self` by comparing handle indices.
677    ///
678    /// If `self` is a valid handle (i.e., it has been validated using [`Self::check_valid_for`])
679    /// and this function returns [`Ok`], then it may be assumed that `depends_on` is also valid.
680    /// In [`naga`](crate)'s current arena-based implementation, this is useful for validating
681    /// recursive definitions of arena-based values in linear time.
682    ///
683    /// # Errors
684    ///
685    /// If `depends_on`'s handle is from the same [`Arena`] as `self'`s, but not constructed earlier
686    /// than `self`'s, this function returns an error.
687    pub(self) fn check_dep(self, depends_on: Self) -> Result<Self, FwdDepError> {
688        if depends_on < self {
689            Ok(self)
690        } else {
691            let erase_handle_type = |handle: Handle<_>| {
692                Handle::new(NonMaxU32::new((handle.index()).try_into().unwrap()).unwrap())
693            };
694            Err(FwdDepError {
695                subject: erase_handle_type(self),
696                subject_kind: std::any::type_name::<T>(),
697                depends_on: erase_handle_type(depends_on),
698                depends_on_kind: std::any::type_name::<T>(),
699            })
700        }
701    }
702
703    /// Like [`Self::check_dep`], except for [`Option`]al handle values.
704    pub(self) fn check_dep_opt(self, depends_on: Option<Self>) -> Result<Self, FwdDepError> {
705        self.check_dep_iter(depends_on.into_iter())
706    }
707
708    /// Like [`Self::check_dep`], except for [`Iterator`]s over handle values.
709    pub(self) fn check_dep_iter(
710        self,
711        depends_on: impl Iterator<Item = Self>,
712    ) -> Result<Self, FwdDepError> {
713        for handle in depends_on {
714            self.check_dep(handle)?;
715        }
716        Ok(self)
717    }
718}

720impl<T> crate::arena::Range<T> {
721    pub(self) fn check_valid_for(&self, arena: &Arena<T>) -> Result<(), BadRangeError> {
722        arena.check_contains_range(self)
723    }
724}
