12/// The results of performing a bounds check.
13///
14/// On success, `write_bounds_check` returns a value of this type.
15pub(super) enum BoundsCheckResult {
16    /// The index is statically known and in bounds, with the given value.
17    KnownInBounds(u32),
18
19    /// The given instruction computes the index to be used.
20    Computed(Word),
21
22    /// The given instruction computes a boolean condition which is true
23    /// if the index is in bounds.
24    Conditional(Word),
25}
26
27/// A value that we either know at translation time, or need to compute at runtime.
28pub(super) enum MaybeKnown<T> {
29    /// The value is known at shader translation time.
30    Known(T),
31
32    /// The value is computed by the instruction with the given id.
33    Computed(Word),
34}

37    /// Emit code to compute the length of a run-time array.
38    ///
39    /// Given `array`, an expression referring a runtime-sized array, return the
40    /// instruction id for the array's length.
41    pub(super) fn write_runtime_array_length(
42        &mut self,
43        array: Handle<crate::Expression>,
44        block: &mut Block,
45    ) -> Result<Word, Error> {
46        // Naga IR permits runtime-sized arrays as global variables, or as the
47        // final member of a struct that is a global variable, or one of these
48        // inside a buffer that is itself an element in a buffer bindings array.
49        // SPIR-V requires that runtime-sized arrays are wrapped in structs.
50        // See `helpers::global_needs_wrapper` and its uses.
51        let (opt_array_index_id, global_handle, opt_last_member_index) = match self
52            .ir_function
53            .expressions[array]
54        {
55            crate::Expression::AccessIndex { base, index } => {
56                match self.ir_function.expressions[base] {
57                    // The global variable is an array of buffer bindings of structs,
58                    // we are accessing one of them with a static index,
59                    // and the last member of it.
60                    crate::Expression::AccessIndex {
61                        base: base_outer,
62                        index: index_outer,
63                    } => match self.ir_function.expressions[base_outer] {
64                        crate::Expression::GlobalVariable(handle) => {
65                            let index_id = self.get_index_constant(index_outer);
66                            (Some(index_id), handle, Some(index))
67                        }
68                        _ => return Err(Error::Validation("array length expression case-1a")),
69                    },
70                    // The global variable is an array of buffer bindings of structs,
71                    // we are accessing one of them with a dynamic index,
72                    // and the last member of it.
73                    crate::Expression::Access {
74                        base: base_outer,
75                        index: index_outer,
76                    } => match self.ir_function.expressions[base_outer] {
77                        crate::Expression::GlobalVariable(handle) => {
78                            let index_id = self.cached[index_outer];
79                            (Some(index_id), handle, Some(index))
80                        }
81                        _ => return Err(Error::Validation("array length expression case-1b")),
82                    },
83                    // The global variable is a buffer, and we are accessing the last member.
84                    crate::Expression::GlobalVariable(handle) => {
85                        let global = &self.ir_module.global_variables[handle];
86                        match self.ir_module.types[global.ty].inner {
87                            // The global variable is an array of buffer bindings of run-time arrays.
88                            crate::TypeInner::BindingArray { .. } => (Some(index), handle, None),
89                            // The global variable is a struct, and we are accessing the last member
90                            _ => (None, handle, Some(index)),
91                        }
92                    }
93                    _ => return Err(Error::Validation("array length expression case-1c")),
94                }
95            }
96            // The global variable is an array of buffer bindings of arrays.
97            crate::Expression::Access { base, index } => match self.ir_function.expressions[base] {
98                crate::Expression::GlobalVariable(handle) => {
99                    let index_id = self.cached[index];
100                    let global = &self.ir_module.global_variables[handle];
101                    match self.ir_module.types[global.ty].inner {
102                        crate::TypeInner::BindingArray { .. } => (Some(index_id), handle, None),
103                        _ => return Err(Error::Validation("array length expression case-2a")),
104                    }
105                }
106                _ => return Err(Error::Validation("array length expression case-2b")),
107            },
108            // The global variable is a run-time array.
109            crate::Expression::GlobalVariable(handle) => {
110                let global = &self.ir_module.global_variables[handle];
111                if !global_needs_wrapper(self.ir_module, global) {
112                    return Err(Error::Validation("array length expression case-3"));
113                }
114                (None, handle, None)
115            }
116            _ => return Err(Error::Validation("array length expression case-4")),
117        };
118
119        let gvar = self.writer.global_variables[global_handle].clone();
120        let global = &self.ir_module.global_variables[global_handle];
121        let (last_member_index, gvar_id) = match opt_last_member_index {
122            Some(index) => (index, gvar.access_id),
123            None => {
124                if !global_needs_wrapper(self.ir_module, global) {
125                    return Err(Error::Validation(
126                        "pointer to a global that is not a wrapped array",
127                    ));
128                }
129                (0, gvar.var_id)
130            }
131        };
132        let structure_id = match opt_array_index_id {
133            // We are indexing inside a binding array, generate the access op.
134            Some(index_id) => {
135                let element_type_id = match self.ir_module.types[global.ty].inner {
136                    crate::TypeInner::BindingArray { base, size: _ } => {
137                        let class = map_storage_class(global.space);
138                        self.get_pointer_id(base, class)?
139                    }
140                    _ => return Err(Error::Validation("array length expression case-5")),
141                };
142                let structure_id = self.gen_id();
143                block.body.push(Instruction::access_chain(
144                    element_type_id,
145                    structure_id,
146                    gvar_id,
147                    &[index_id],
148                ));
149                structure_id
150            }
151            None => gvar_id,
152        };
153        let length_id = self.gen_id();
154        block.body.push(Instruction::array_length(
155            self.writer.get_uint_type_id(),
156            length_id,
157            structure_id,
158            last_member_index,
159        ));
160
161        Ok(length_id)
162    }

164    /// Compute the length of a subscriptable value.
165    ///
166    /// Given `sequence`, an expression referring to some indexable type, return
167    /// its length. The result may either be computed by SPIR-V instructions, or
168    /// known at shader translation time.
169    ///
170    /// `sequence` may be a `Vector`, `Matrix`, or `Array`, a `Pointer` to any
171    /// of those, or a `ValuePointer`. An array may be fixed-size, dynamically
172    /// sized, or use a specializable constant as its length.
173    fn write_sequence_length(
174        &mut self,
175        sequence: Handle<crate::Expression>,
176        block: &mut Block,
177    ) -> Result<MaybeKnown<u32>, Error> {
178        let sequence_ty = self.fun_info[sequence].ty.inner_with(&self.ir_module.types);
179        match sequence_ty.indexable_length(self.ir_module) {
180            Ok(crate::proc::IndexableLength::Known(known_length)) => {
181                Ok(MaybeKnown::Known(known_length))
182            }
183            Ok(crate::proc::IndexableLength::Dynamic) => {
184                let length_id = self.write_runtime_array_length(sequence, block)?;
185                Ok(MaybeKnown::Computed(length_id))
186            }
187            Err(err) => {
188                log::error!("Sequence length for {:?} failed: {}", sequence, err);
189                Err(Error::Validation("indexable length"))
190            }
191        }
192    }

194    /// Compute the maximum valid index of a subscriptable value.
195    ///
196    /// Given `sequence`, an expression referring to some indexable type, return
197    /// its maximum valid index - one less than its length. The result may
198    /// either be computed, or known at shader translation time.
199    ///
200    /// `sequence` may be a `Vector`, `Matrix`, or `Array`, a `Pointer` to any
201    /// of those, or a `ValuePointer`. An array may be fixed-size, dynamically
202    /// sized, or use a specializable constant as its length.
203    fn write_sequence_max_index(
204        &mut self,
205        sequence: Handle<crate::Expression>,
206        block: &mut Block,
207    ) -> Result<MaybeKnown<u32>, Error> {
208        match self.write_sequence_length(sequence, block)? {
209            MaybeKnown::Known(known_length) => {
210                // We should have thrown out all attempts to subscript zero-length
211                // sequences during validation, so the following subtraction should never
212                // underflow.
213                assert!(known_length > 0);
214                // Compute the max index from the length now.
215                Ok(MaybeKnown::Known(known_length - 1))
216            }
217            MaybeKnown::Computed(length_id) => {
218                // Emit code to compute the max index from the length.
219                let const_one_id = self.get_index_constant(1);
220                let max_index_id = self.gen_id();
221                block.body.push(Instruction::binary(
222                    spirv::Op::ISub,
223                    self.writer.get_uint_type_id(),
224                    max_index_id,
225                    length_id,
226                    const_one_id,
227                ));
228                Ok(MaybeKnown::Computed(max_index_id))
229            }
230        }
231    }

233    /// Restrict an index to be in range for a vector, matrix, or array.
234    ///
235    /// This is used to implement `BoundsCheckPolicy::Restrict`. An in-bounds
236    /// index is left unchanged. An out-of-bounds index is replaced with some
237    /// arbitrary in-bounds index. Note,this is not necessarily clamping; for
238    /// example, negative indices might be changed to refer to the last element
239    /// of the sequence, not the first, as clamping would do.
240    ///
241    /// Either return the restricted index value, if known, or add instructions
242    /// to `block` to compute it, and return the id of the result. See the
243    /// documentation for `BoundsCheckResult` for details.
244    ///
245    /// The `sequence` expression may be a `Vector`, `Matrix`, or `Array`, a
246    /// `Pointer` to any of those, or a `ValuePointer`. An array may be
247    /// fixed-size, dynamically sized, or use a specializable constant as its
248    /// length.
249    pub(super) fn write_restricted_index(
250        &mut self,
251        sequence: Handle<crate::Expression>,
252        index: Handle<crate::Expression>,
253        block: &mut Block,
254    ) -> Result<BoundsCheckResult, Error> {
255        let index_id = self.cached[index];
256
257        // Get the sequence's maximum valid index. Return early if we've already
258        // done the bounds check.
259        let max_index_id = match self.write_sequence_max_index(sequence, block)? {
260            MaybeKnown::Known(known_max_index) => {
261                if let Ok(known_index) = self
262                    .ir_module
263                    .to_ctx()
264                    .eval_expr_to_u32_from(index, &self.ir_function.expressions)
265                {
266                    // Both the index and length are known at compile time.
267                    //
268                    // In strict WGSL compliance mode, out-of-bounds indices cannot be
269                    // reported at shader translation time, and must be replaced with
270                    // in-bounds indices at run time. So we cannot assume that
271                    // validation ensured the index was in bounds. Restrict now.
272                    let restricted = std::cmp::min(known_index, known_max_index);
273                    return Ok(BoundsCheckResult::KnownInBounds(restricted));
274                }
275
276                self.get_index_constant(known_max_index)
277            }
278            MaybeKnown::Computed(max_index_id) => max_index_id,
279        };
280
281        // One or the other of the index or length is dynamic, so emit code for
282        // BoundsCheckPolicy::Restrict.
283        let restricted_index_id = self.gen_id();
284        block.body.push(Instruction::ext_inst(
285            self.writer.gl450_ext_inst_id,
286            spirv::GLOp::UMin,
287            self.writer.get_uint_type_id(),
288            restricted_index_id,
289            &[index_id, max_index_id],
290        ));
291        Ok(BoundsCheckResult::Computed(restricted_index_id))
292    }

294    /// Write an index bounds comparison to `block`, if needed.
295    ///
296    /// If we're able to determine statically that `index` is in bounds for
297    /// `sequence`, return `KnownInBounds(value)`, where `value` is the actual
298    /// value of the index. (In principle, one could know that the index is in
299    /// bounds without knowing its specific value, but in our simple-minded
300    /// situation, we always know it.)
301    ///
302    /// If instead we must generate code to perform the comparison at run time,
303    /// return `Conditional(comparison_id)`, where `comparison_id` is an
304    /// instruction producing a boolean value that is true if `index` is in
305    /// bounds for `sequence`.
306    ///
307    /// The `sequence` expression may be a `Vector`, `Matrix`, or `Array`, a
308    /// `Pointer` to any of those, or a `ValuePointer`. An array may be
309    /// fixed-size, dynamically sized, or use a specializable constant as its
310    /// length.
311    fn write_index_comparison(
312        &mut self,
313        sequence: Handle<crate::Expression>,
314        index: Handle<crate::Expression>,
315        block: &mut Block,
316    ) -> Result<BoundsCheckResult, Error> {
317        let index_id = self.cached[index];
318
319        // Get the sequence's length. Return early if we've already done the
320        // bounds check.
321        let length_id = match self.write_sequence_length(sequence, block)? {
322            MaybeKnown::Known(known_length) => {
323                if let Ok(known_index) = self
324                    .ir_module
325                    .to_ctx()
326                    .eval_expr_to_u32_from(index, &self.ir_function.expressions)
327                {
328                    // Both the index and length are known at compile time.
329                    //
330                    // It would be nice to assume that, since we are using the
331                    // `ReadZeroSkipWrite` policy, we are not in strict WGSL
332                    // compliance mode, and thus we can count on the validator to have
333                    // rejected any programs with known out-of-bounds indices, and
334                    // thus just return `KnownInBounds` here without actually
335                    // checking.
336                    //
337                    // But it's also reasonable to expect that bounds check policies
338                    // and error reporting policies should be able to vary
339                    // independently without introducing security holes. So, we should
340                    // support the case where bad indices do not cause validation
341                    // errors, and are handled via `ReadZeroSkipWrite`.
342                    //
343                    // In theory, when `known_index` is bad, we could return a new
344                    // `KnownOutOfBounds` variant here. But it's simpler just to fall
345                    // through and let the bounds check take place. The shader is
346                    // broken anyway, so it doesn't make sense to invest in emitting
347                    // the ideal code for it.
348                    if known_index < known_length {
349                        return Ok(BoundsCheckResult::KnownInBounds(known_index));
350                    }
351                }
352
353                self.get_index_constant(known_length)
354            }
355            MaybeKnown::Computed(length_id) => length_id,
356        };
357
358        // Compare the index against the length.
359        let condition_id = self.gen_id();
360        block.body.push(Instruction::binary(
361            spirv::Op::ULessThan,
362            self.writer.get_bool_type_id(),
363            condition_id,
364            index_id,
365            length_id,
366        ));
367
368        // Indicate that we did generate the check.
369        Ok(BoundsCheckResult::Conditional(condition_id))
370    }

372    /// Emit a conditional load for `BoundsCheckPolicy::ReadZeroSkipWrite`.
373    ///
374    /// Generate code to load a value of `result_type` if `condition` is true,
375    /// and generate a null value of that type if it is false. Call `emit_load`
376    /// to emit the instructions to perform the load. Return the id of the
377    /// merged value of the two branches.
378    pub(super) fn write_conditional_indexed_load<F>(
379        &mut self,
380        result_type: Word,
381        condition: Word,
382        block: &mut Block,
383        emit_load: F,
384    ) -> Word
385    where
386        F: FnOnce(&mut IdGenerator, &mut Block) -> Word,
387    {
388        // For the out-of-bounds case, we produce a zero value.
389        let null_id = self.writer.get_constant_null(result_type);
390
391        let mut selection = Selection::start(block, result_type);
392
393        // As it turns out, we don't actually need a full 'if-then-else'
394        // structure for this: SPIR-V constants are declared up front, so the
395        // 'else' block would have no instructions. Instead we emit something
396        // like this:
397        //
398        //     result = zero;
399        //     if in_bounds {
400        //         result = do the load;
401        //     }
402        //     use result;
403
404        // Continue only if the index was in bounds. Otherwise, branch to the
405        // merge block.
406        selection.if_true(self, condition, null_id);
407
408        // The in-bounds path. Perform the access and the load.
409        let loaded_value = emit_load(&mut self.writer.id_gen, selection.block());
410
411        selection.finish(self, loaded_value)
412    }

414    /// Emit code for bounds checks for an array, vector, or matrix access.
415    ///
416    /// This implements either `index_bounds_check_policy` or
417    /// `buffer_bounds_check_policy`, depending on the address space of the
418    /// pointer being accessed.
419    ///
420    /// Return a `BoundsCheckResult` indicating how the index should be
421    /// consumed. See that type's documentation for details.
422    pub(super) fn write_bounds_check(
423        &mut self,
424        base: Handle<crate::Expression>,
425        index: Handle<crate::Expression>,
426        block: &mut Block,
427    ) -> Result<BoundsCheckResult, Error> {
428        let policy = self.writer.bounds_check_policies.choose_policy(
429            base,
430            &self.ir_module.types,
431            self.fun_info,
432        );
433
434        Ok(match policy {
435            BoundsCheckPolicy::Restrict => self.write_restricted_index(base, index, block)?,
436            BoundsCheckPolicy::ReadZeroSkipWrite => {
437                self.write_index_comparison(base, index, block)?
438            }
439            BoundsCheckPolicy::Unchecked => BoundsCheckResult::Computed(self.cached[index]),
440        })
441    }

443    /// Emit code to subscript a vector by value with a computed index.
444    ///
445    /// Return the id of the element value.
446    pub(super) fn write_vector_access(
447        &mut self,
448        expr_handle: Handle<crate::Expression>,
449        base: Handle<crate::Expression>,
450        index: Handle<crate::Expression>,
451        block: &mut Block,
452    ) -> Result<Word, Error> {
453        let result_type_id = self.get_expression_type_id(&self.fun_info[expr_handle].ty);
454
455        let base_id = self.cached[base];
456        let index_id = self.cached[index];
457
458        let result_id = match self.write_bounds_check(base, index, block)? {
459            BoundsCheckResult::KnownInBounds(known_index) => {
460                let result_id = self.gen_id();
461                block.body.push(Instruction::composite_extract(
462                    result_type_id,
463                    result_id,
464                    base_id,
465                    &[known_index],
466                ));
467                result_id
468            }
469            BoundsCheckResult::Computed(computed_index_id) => {
470                let result_id = self.gen_id();
471                block.body.push(Instruction::vector_extract_dynamic(
472                    result_type_id,
473                    result_id,
474                    base_id,
475                    computed_index_id,
476                ));
477                result_id
478            }
479            BoundsCheckResult::Conditional(comparison_id) => {
480                // Run-time bounds checks were required. Emit
481                // conditional load.
482                self.write_conditional_indexed_load(
483                    result_type_id,
484                    comparison_id,
485                    block,
486                    |id_gen, block| {
487                        // The in-bounds path. Generate the access.
488                        let element_id = id_gen.next();
489                        block.body.push(Instruction::vector_extract_dynamic(
490                            result_type_id,
491                            element_id,
492                            base_id,
493                            index_id,
494                        ));
495                        element_id
496                    },
497                )
498            }
499        };
500
501        Ok(result_id)
502    }
