8impl PhysicalLayout {

23        sink.extend(iter::once(self.bound));

74impl Instruction {
75    pub(super) const fn new(op: Op) -> Self {
76        Instruction {
77            op,
78            wc: 1, // Always start at 1 for the first word (OP + WC),
79            type_id: None,
80            result_id: None,
81            operands: vec![],
82        }
83    }
84
85    #[allow(clippy::panic)]
86    pub(super) fn set_type(&mut self, id: Word) {
87        assert!(self.type_id.is_none(), "Type can only be set once");
88        self.type_id = Some(id);
89        self.wc += 1;
90    }
91
92    #[allow(clippy::panic)]
93    pub(super) fn set_result(&mut self, id: Word) {
94        assert!(self.result_id.is_none(), "Result can only be set once");
95        self.result_id = Some(id);
96        self.wc += 1;
97    }
98
99    pub(super) fn add_operand(&mut self, operand: Word) {
100        self.operands.push(operand);
101        self.wc += 1;
102    }
103
104    pub(super) fn add_operands(&mut self, operands: Vec<Word>) {
105        for operand in operands.into_iter() {
106            self.add_operand(operand)
107        }
108    }
109
110    pub(super) fn to_words(&self, sink: &mut impl Extend<Word>) {
111        sink.extend(Some(self.wc << 16 | self.op as u32));
112        sink.extend(self.type_id);
113        sink.extend(self.result_id);
114        sink.extend(self.operands.iter().cloned());
115    }
116}

118impl Instruction {
119    #[cfg(test)]
120    fn validate(&self, words: &[Word]) {
121        let mut inst_index = 0;
122        let (wc, op) = ((words[inst_index] >> 16) as u16, words[inst_index] as u16);
123        inst_index += 1;
124
125        assert_eq!(wc, words.len() as u16);
126        assert_eq!(op, self.op as u16);
127
128        if self.type_id.is_some() {
129            assert_eq!(words[inst_index], self.type_id.unwrap());
130            inst_index += 1;
131        }
132
133        if self.result_id.is_some() {
134            assert_eq!(words[inst_index], self.result_id.unwrap());
135            inst_index += 1;
136        }
137
138        for (op_index, i) in (inst_index..wc as usize).enumerate() {
139            assert_eq!(words[i], self.operands[op_index]);
140        }
141    }
142}

158#[test]
159fn test_logical_layout_in_words() {
160    let mut output = vec![];
161    let mut layout = LogicalLayout::default();
162    let layout_vectors = 11;
163    let mut instructions = Vec::with_capacity(layout_vectors);
164
165    let vector_names = &[
166        "Capabilities",
167        "Extensions",
168        "External Instruction Imports",
169        "Memory Model",
170        "Entry Points",
171        "Execution Modes",
172        "Debugs",
173        "Annotations",
174        "Declarations",
175        "Function Declarations",
176        "Function Definitions",
177    ];
178
179    for (i, _) in vector_names.iter().enumerate().take(layout_vectors) {
180        let mut dummy_instruction = Instruction::new(Op::Constant);
181        dummy_instruction.set_type((i + 1) as u32);
182        dummy_instruction.set_result((i + 2) as u32);
183        dummy_instruction.add_operand((i + 3) as u32);
184        dummy_instruction.add_operands(super::helpers::string_to_words(
185            format!("This is the vector: {}", vector_names[i]).as_str(),
186        ));
187        instructions.push(dummy_instruction);
188    }
189
190    instructions[0].to_words(&mut layout.capabilities);
191    instructions[1].to_words(&mut layout.extensions);
192    instructions[2].to_words(&mut layout.ext_inst_imports);
193    instructions[3].to_words(&mut layout.memory_model);
194    instructions[4].to_words(&mut layout.entry_points);
195    instructions[5].to_words(&mut layout.execution_modes);
196    instructions[6].to_words(&mut layout.debugs);
197    instructions[7].to_words(&mut layout.annotations);
198    instructions[8].to_words(&mut layout.declarations);
199    instructions[9].to_words(&mut layout.function_declarations);
200    instructions[10].to_words(&mut layout.function_definitions);
201
202    layout.in_words(&mut output);
203
204    let mut index: usize = 0;
205    for instruction in instructions {
206        let wc = instruction.wc as usize;
207        instruction.validate(&output[index..index + wc]);
208        index += wc;
209    }
