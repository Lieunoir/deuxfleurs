10pub enum Token<'a> {
11    Separator(char),
12    Paren(char),
13    Attribute,
14    Number(Result<Number, NumberError>),
15    Word(&'a str),
16    Operation(char),
17    LogicalOperation(char),
18    ShiftOperation(char),
19    AssignmentOperation(char),
20    IncrementOperation,
21    DecrementOperation,
22    Arrow,
23    Unknown(char),
24    Trivia,
25    End,
26}

28fn consume_any(input: &str, what: impl Fn(char) -> bool) -> (&str, &str) {
29    let pos = input.find(|c| !what(c)).unwrap_or(input.len());
30    input.split_at(pos)
31}

33/// Return the token at the start of `input`.
34///
35/// If `generic` is `false`, then the bit shift operators `>>` or `<<`
36/// are valid lookahead tokens for the current parser state (see [§3.1
37/// Parsing] in the WGSL specification). In other words:
38///
39/// -   If `generic` is `true`, then we are expecting an angle bracket
40///     around a generic type parameter, like the `<` and `>` in
41///     `vec3<f32>`, so interpret `<` and `>` as `Token::Paren` tokens,
42///     even if they're part of `<<` or `>>` sequences.
43///
44/// -   Otherwise, interpret `<<` and `>>` as shift operators:
45///     `Token::LogicalOperation` tokens.
46///
47/// [§3.1 Parsing]: https://gpuweb.github.io/gpuweb/wgsl/#parsing
48fn consume_token(input: &str, generic: bool) -> (Token<'_>, &str) {
49    let mut chars = input.chars();
50    let cur = match chars.next() {
51        Some(c) => c,
52        None => return (Token::End, ""),
53    };
54    match cur {
55        ':' | ';' | ',' => (Token::Separator(cur), chars.as_str()),
56        '.' => {
57            let og_chars = chars.as_str();
58            match chars.next() {
59                Some('0'..='9') => consume_number(input),
60                _ => (Token::Separator(cur), og_chars),
61            }
62        }
63        '@' => (Token::Attribute, chars.as_str()),
64        '(' | ')' | '{' | '}' | '[' | ']' => (Token::Paren(cur), chars.as_str()),
65        '<' | '>' => {
66            let og_chars = chars.as_str();
67            match chars.next() {
68                Some('=') if !generic => (Token::LogicalOperation(cur), chars.as_str()),
69                Some(c) if c == cur && !generic => {
70                    let og_chars = chars.as_str();
71                    match chars.next() {
72                        Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
73                        _ => (Token::ShiftOperation(cur), og_chars),
74                    }
75                }
76                _ => (Token::Paren(cur), og_chars),
77            }
78        }
79        '0'..='9' => consume_number(input),
80        '/' => {
81            let og_chars = chars.as_str();
82            match chars.next() {
83                Some('/') => {
84                    let _ = chars.position(is_comment_end);
85                    (Token::Trivia, chars.as_str())
86                }
87                Some('*') => {
88                    let mut depth = 1;
89                    let mut prev = None;
90
91                    for c in &mut chars {
92                        match (prev, c) {
93                            (Some('*'), '/') => {
94                                prev = None;
95                                depth -= 1;
96                                if depth == 0 {
97                                    return (Token::Trivia, chars.as_str());
98                                }
99                            }
100                            (Some('/'), '*') => {
101                                prev = None;
102                                depth += 1;
103                            }
104                            _ => {
105                                prev = Some(c);
106                            }
107                        }
108                    }
109
110                    (Token::End, "")
111                }
112                Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
113                _ => (Token::Operation(cur), og_chars),
114            }
115        }
116        '-' => {
117            let og_chars = chars.as_str();
118            match chars.next() {
119                Some('>') => (Token::Arrow, chars.as_str()),
120                Some('-') => (Token::DecrementOperation, chars.as_str()),
121                Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
122                _ => (Token::Operation(cur), og_chars),
123            }
124        }
125        '+' => {
126            let og_chars = chars.as_str();
127            match chars.next() {
128                Some('+') => (Token::IncrementOperation, chars.as_str()),
129                Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
130                _ => (Token::Operation(cur), og_chars),
131            }
132        }
133        '*' | '%' | '^' => {
134            let og_chars = chars.as_str();
135            match chars.next() {
136                Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
137                _ => (Token::Operation(cur), og_chars),
138            }
139        }
140        '~' => (Token::Operation(cur), chars.as_str()),
141        '=' | '!' => {
142            let og_chars = chars.as_str();
143            match chars.next() {
144                Some('=') => (Token::LogicalOperation(cur), chars.as_str()),
145                _ => (Token::Operation(cur), og_chars),
146            }
147        }
148        '&' | '|' => {
149            let og_chars = chars.as_str();
150            match chars.next() {
151                Some(c) if c == cur => (Token::LogicalOperation(cur), chars.as_str()),
152                Some('=') => (Token::AssignmentOperation(cur), chars.as_str()),
153                _ => (Token::Operation(cur), og_chars),
154            }
155        }
156        _ if is_blankspace(cur) => {
157            let (_, rest) = consume_any(input, is_blankspace);
158            (Token::Trivia, rest)
159        }
160        _ if is_word_start(cur) => {
161            let (word, rest) = consume_any(input, is_word_part);
162            (Token::Word(word), rest)
163        }
164        _ => (Token::Unknown(cur), chars.as_str()),
165    }
166}

168/// Returns whether or not a char is a comment end
169/// (Unicode Pattern_White_Space excluding U+0020, U+0009, U+200E and U+200F)
170const fn is_comment_end(c: char) -> bool {
171    match c {
172        '\u{000a}'..='\u{000d}' | '\u{0085}' | '\u{2028}' | '\u{2029}' => true,
173        _ => false,
174    }
175}
176
177/// Returns whether or not a char is a blankspace (Unicode Pattern_White_Space)
178const fn is_blankspace(c: char) -> bool {
179    match c {
180        '\u{0020}'
181        | '\u{0009}'..='\u{000d}'
182        | '\u{0085}'
183        | '\u{200e}'
184        | '\u{200f}'
185        | '\u{2028}'
186        | '\u{2029}' => true,
187        _ => false,
188    }
189}
190
191/// Returns whether or not a char is a word start (Unicode XID_Start + '_')
192fn is_word_start(c: char) -> bool {
193    c == '_' || unicode_xid::UnicodeXID::is_xid_start(c)
194}
195
196/// Returns whether or not a char is a word part (Unicode XID_Continue)
197fn is_word_part(c: char) -> bool {
198    unicode_xid::UnicodeXID::is_xid_continue(c)
199}

202pub(in crate::front::wgsl) struct Lexer<'a> {
203    input: &'a str,
204    pub(in crate::front::wgsl) source: &'a str,
205    // The byte offset of the end of the last non-trivia token.
206    last_end_offset: usize,
207}
208
209impl<'a> Lexer<'a> {
210    pub(in crate::front::wgsl) const fn new(input: &'a str) -> Self {
211        Lexer {
212            input,
213            source: input,
214            last_end_offset: 0,
215        }
216    }
217
218    /// Calls the function with a lexer and returns the result of the function as well as the span for everything the function parsed
219    ///
220    /// # Examples
221    /// ```ignore
222    /// let lexer = Lexer::new("5");
223    /// let (value, span) = lexer.capture_span(Lexer::next_uint_literal);
224    /// assert_eq!(value, 5);
225    /// ```
226    #[inline]
227    pub fn capture_span<T, E>(
228        &mut self,
229        inner: impl FnOnce(&mut Self) -> Result<T, E>,
230    ) -> Result<(T, Span), E> {
231        let start = self.current_byte_offset();
232        let res = inner(self)?;
233        let end = self.current_byte_offset();
234        Ok((res, Span::from(start..end)))
235    }
236
237    pub(in crate::front::wgsl) fn start_byte_offset(&mut self) -> usize {
238        loop {
239            // Eat all trivia because `next` doesn't eat trailing trivia.
240            let (token, rest) = consume_token(self.input, false);
241            if let Token::Trivia = token {
242                self.input = rest;
243            } else {
244                return self.current_byte_offset();
245            }
246        }
247    }
248
249    fn peek_token_and_rest(&mut self) -> (TokenSpan<'a>, &'a str) {
250        let mut cloned = self.clone();
251        let token = cloned.next();
252        let rest = cloned.input;
253        (token, rest)
254    }
255
256    const fn current_byte_offset(&self) -> usize {
257        self.source.len() - self.input.len()
258    }
259
260    pub(in crate::front::wgsl) fn span_from(&self, offset: usize) -> Span {
261        Span::from(offset..self.last_end_offset)
262    }
263
264    /// Return the next non-whitespace token from `self`.
265    ///
266    /// Assume we are a parse state where bit shift operators may
267    /// occur, but not angle brackets.
268    #[must_use]
269    pub(in crate::front::wgsl) fn next(&mut self) -> TokenSpan<'a> {
270        self.next_impl(false)
271    }
272
273    /// Return the next non-whitespace token from `self`.
274    ///
275    /// Assume we are in a parse state where angle brackets may occur,
276    /// but not bit shift operators.
277    #[must_use]
278    pub(in crate::front::wgsl) fn next_generic(&mut self) -> TokenSpan<'a> {
279        self.next_impl(true)
280    }
281
282    /// Return the next non-whitespace token from `self`, with a span.
283    ///
284    /// See [`consume_token`] for the meaning of `generic`.
285    fn next_impl(&mut self, generic: bool) -> TokenSpan<'a> {
286        let mut start_byte_offset = self.current_byte_offset();
287        loop {
288            let (token, rest) = consume_token(self.input, generic);
289            self.input = rest;
290            match token {
291                Token::Trivia => start_byte_offset = self.current_byte_offset(),
292                _ => {
293                    self.last_end_offset = self.current_byte_offset();
294                    return (token, self.span_from(start_byte_offset));
295                }
296            }
297        }
298    }
299
300    #[must_use]
301    pub(in crate::front::wgsl) fn peek(&mut self) -> TokenSpan<'a> {
302        let (token, _) = self.peek_token_and_rest();
303        token
304    }
305
306    pub(in crate::front::wgsl) fn expect_span(
307        &mut self,
308        expected: Token<'a>,
309    ) -> Result<Span, Error<'a>> {
310        let next = self.next();
311        if next.0 == expected {
312            Ok(next.1)
313        } else {
314            Err(Error::Unexpected(next.1, ExpectedToken::Token(expected)))
315        }
316    }
317
318    pub(in crate::front::wgsl) fn expect(&mut self, expected: Token<'a>) -> Result<(), Error<'a>> {
319        self.expect_span(expected)?;
320        Ok(())
321    }
322
323    pub(in crate::front::wgsl) fn expect_generic_paren(
324        &mut self,
325        expected: char,
326    ) -> Result<(), Error<'a>> {
327        let next = self.next_generic();
328        if next.0 == Token::Paren(expected) {
329            Ok(())
330        } else {
331            Err(Error::Unexpected(
332                next.1,
333                ExpectedToken::Token(Token::Paren(expected)),
334            ))
335        }
336    }
337
338    /// If the next token matches it is skipped and true is returned
339    pub(in crate::front::wgsl) fn skip(&mut self, what: Token<'_>) -> bool {
340        let (peeked_token, rest) = self.peek_token_and_rest();
341        if peeked_token.0 == what {
342            self.input = rest;
343            true
344        } else {
345            false
346        }
347    }

349    pub(in crate::front::wgsl) fn next_ident_with_span(
350        &mut self,
351    ) -> Result<(&'a str, Span), Error<'a>> {
352        match self.next() {
353            (Token::Word("_"), span) => Err(Error::InvalidIdentifierUnderscore(span)),
354            (Token::Word(word), span) if word.starts_with("__") => {
355                Err(Error::ReservedIdentifierPrefix(span))
356            }
357            (Token::Word(word), span) => Ok((word, span)),
358            other => Err(Error::Unexpected(other.1, ExpectedToken::Identifier)),
359        }
360    }
361
362    pub(in crate::front::wgsl) fn next_ident(
363        &mut self,
364    ) -> Result<super::ast::Ident<'a>, Error<'a>> {
365        let ident = self
366            .next_ident_with_span()
367            .map(|(name, span)| super::ast::Ident { name, span })?;
368
369        if crate::keywords::wgsl::RESERVED.contains(&ident.name) {
370            return Err(Error::ReservedKeyword(ident.span));
371        }
372
373        Ok(ident)
374    }
375
376    /// Parses a generic scalar type, for example `<f32>`.
377    pub(in crate::front::wgsl) fn next_scalar_generic(&mut self) -> Result<Scalar, Error<'a>> {
378        self.expect_generic_paren('<')?;
379        let pair = match self.next() {
380            (Token::Word(word), span) => {
381                conv::get_scalar_type(word).ok_or(Error::UnknownScalarType(span))
382            }
383            (_, span) => Err(Error::UnknownScalarType(span)),
384        }?;
385        self.expect_generic_paren('>')?;
386        Ok(pair)
387    }
388
389    /// Parses a generic scalar type, for example `<f32>`.
390    ///
391    /// Returns the span covering the inner type, excluding the brackets.
392    pub(in crate::front::wgsl) fn next_scalar_generic_with_span(
393        &mut self,
394    ) -> Result<(Scalar, Span), Error<'a>> {
395        self.expect_generic_paren('<')?;
396        let pair = match self.next() {
397            (Token::Word(word), span) => conv::get_scalar_type(word)
398                .map(|scalar| (scalar, span))
399                .ok_or(Error::UnknownScalarType(span)),
400            (_, span) => Err(Error::UnknownScalarType(span)),
401        }?;
402        self.expect_generic_paren('>')?;
403        Ok(pair)
404    }
405
406    pub(in crate::front::wgsl) fn next_storage_access(
407        &mut self,
408    ) -> Result<crate::StorageAccess, Error<'a>> {
409        let (ident, span) = self.next_ident_with_span()?;
410        match ident {
411            "read" => Ok(crate::StorageAccess::LOAD),
412            "write" => Ok(crate::StorageAccess::STORE),
413            "read_write" => Ok(crate::StorageAccess::LOAD | crate::StorageAccess::STORE),
414            _ => Err(Error::UnknownAccess(span)),
415        }
416    }
417
418    pub(in crate::front::wgsl) fn next_format_generic(
419        &mut self,
420    ) -> Result<(crate::StorageFormat, crate::StorageAccess), Error<'a>> {
421        self.expect(Token::Paren('<'))?;
422        let (ident, ident_span) = self.next_ident_with_span()?;
423        let format = conv::map_storage_format(ident, ident_span)?;
424        self.expect(Token::Separator(','))?;
425        let access = self.next_storage_access()?;
426        self.expect(Token::Paren('>'))?;
427        Ok((format, access))
428    }
429
430    pub(in crate::front::wgsl) fn open_arguments(&mut self) -> Result<(), Error<'a>> {
431        self.expect(Token::Paren('('))
432    }
433
434    pub(in crate::front::wgsl) fn close_arguments(&mut self) -> Result<(), Error<'a>> {
435        let _ = self.skip(Token::Separator(','));
436        self.expect(Token::Paren(')'))
437    }
438
439    pub(in crate::front::wgsl) fn next_argument(&mut self) -> Result<bool, Error<'a>> {
440        let paren = Token::Paren(')');
441        if self.skip(Token::Separator(',')) {
442            Ok(!self.skip(paren))
443        } else {
444            self.expect(paren).map(|()| false)
445        }
446    }
447}

451fn sub_test(source: &str, expected_tokens: &[Token]) {
452    let mut lex = Lexer::new(source);
453    for &token in expected_tokens {
454        assert_eq!(lex.next().0, token);
455    }
456    assert_eq!(lex.next().0, Token::End);
457}
