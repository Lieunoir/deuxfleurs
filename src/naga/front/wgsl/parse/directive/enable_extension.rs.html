6/// Tracks the status of every enable-extension known to Naga.
7#[derive(Clone, Debug, Eq, PartialEq)]
8pub struct EnableExtensions {}
9
10impl EnableExtensions {
11    pub(crate) const fn empty() -> Self {
12        Self {}
13    }
14
15    /// Add an enable-extension to the set requested by a module.
16    #[allow(unreachable_code)]
17    pub(crate) fn add(&mut self, ext: ImplementedEnableExtension) {
18        let _field: &mut bool = match ext {};
19        *_field = true;
20    }
21
22    /// Query whether an enable-extension tracked here has been requested.
23    #[allow(unused)]
24    pub(crate) const fn contains(&self, ext: ImplementedEnableExtension) -> bool {
25        match ext {}
26    }
27}
28
29impl Default for EnableExtensions {
30    fn default() -> Self {
31        Self::empty()
32    }
33}

35/// An enable-extension not guaranteed to be present in all environments.
36///
37/// WGSL spec.: <https://www.w3.org/TR/WGSL/#enable-extensions-sec>
38#[derive(Clone, Copy, Debug, Hash, Eq, PartialEq)]
39pub enum EnableExtension {
40    #[allow(unused)]
41    Implemented(ImplementedEnableExtension),
42    Unimplemented(UnimplementedEnableExtension),
43}
44
45impl EnableExtension {
46    const F16: &'static str = "f16";
47    const CLIP_DISTANCES: &'static str = "clip_distances";
48    const DUAL_SOURCE_BLENDING: &'static str = "dual_source_blending";
49
50    /// Convert from a sentinel word in WGSL into its associated [`EnableExtension`], if possible.
51    pub(crate) fn from_ident(word: &str, span: Span) -> Result<Self, Error<'_>> {
52        Ok(match word {
53            Self::F16 => Self::Unimplemented(UnimplementedEnableExtension::F16),
54            Self::CLIP_DISTANCES => {
55                Self::Unimplemented(UnimplementedEnableExtension::ClipDistances)
56            }
57            Self::DUAL_SOURCE_BLENDING => {
58                Self::Unimplemented(UnimplementedEnableExtension::DualSourceBlending)
59            }
60            _ => return Err(Error::UnknownEnableExtension(span, word)),
61        })
62    }
63
64    /// Maps this [`EnableExtension`] into the sentinel word associated with it in WGSL.
65    pub const fn to_ident(self) -> &'static str {
66        match self {
67            Self::Implemented(kind) => match kind {},
68            Self::Unimplemented(kind) => match kind {
69                UnimplementedEnableExtension::F16 => Self::F16,
70                UnimplementedEnableExtension::ClipDistances => Self::CLIP_DISTANCES,
71                UnimplementedEnableExtension::DualSourceBlending => Self::DUAL_SOURCE_BLENDING,
72            },
73        }
74    }
75}
76
77/// A variant of [`EnableExtension::Implemented`].
78#[derive(Clone, Copy, Debug, Hash, Eq, PartialEq)]
79pub enum ImplementedEnableExtension {}
80
81/// A variant of [`EnableExtension::Unimplemented`].
82#[derive(Clone, Copy, Debug, Hash, Eq, PartialEq)]
83pub enum UnimplementedEnableExtension {
84    /// Enables `f16`/`half` primitive support in all shader languages.
85    ///
86    /// In the WGSL standard, this corresponds to [`enable f16;`].
87    ///
88    /// [`enable f16;`]: https://www.w3.org/TR/WGSL/#extension-f16
89    F16,
90    /// Enables the `clip_distances` variable in WGSL.
91    ///
92    /// In the WGSL standard, this corresponds to [`enable clip_distances;`].
93    ///
94    /// [`enable clip_distances;`]: https://www.w3.org/TR/WGSL/#extension-f16
95    ClipDistances,
96    /// Enables the `blend_src` attribute in WGSL.
97    ///
98    /// In the WGSL standard, this corresponds to [`enable dual_source_blending;`].
99    ///
100    /// [`enable dual_source_blending;`]: https://www.w3.org/TR/WGSL/#extension-f16
101    DualSourceBlending,
102}
103
104impl UnimplementedEnableExtension {
105    pub(crate) const fn tracking_issue_num(self) -> u16 {
106        match self {
107            Self::F16 => 4384,
108            Self::ClipDistances => 6236,
109            Self::DualSourceBlending => 6402,
110        }
111    }
