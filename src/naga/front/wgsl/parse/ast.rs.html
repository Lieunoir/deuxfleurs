6#[derive(Debug, Default)]
7pub struct TranslationUnit<'a> {
8    pub decls: Arena<GlobalDecl<'a>>,
9    /// The common expressions arena for the entire translation unit.
10    ///
11    /// All functions, global initializers, array lengths, etc. store their
12    /// expressions here. We apportion these out to individual Naga
13    /// [`Function`]s' expression arenas at lowering time. Keeping them all in a
14    /// single arena simplifies handling of things like array lengths (which are
15    /// effectively global and thus don't clearly belong to any function) and
16    /// initializers (which can appear in both function-local and module-scope
17    /// contexts).
18    ///
19    /// [`Function`]: crate::Function
20    pub expressions: Arena<Expression<'a>>,
21
22    /// Non-user-defined types, like `vec4<f32>` or `array<i32, 10>`.
23    ///
24    /// These are referred to by `Handle<ast::Type<'a>>` values.
25    /// User-defined types are referred to by name until lowering.
26    pub types: Arena<Type<'a>>,
27}

41/// A reference to a module-scope definition or predeclared object.
42///
43/// Each [`GlobalDecl`] holds a set of these values, to be resolved to
44/// specific definitions later. To support de-duplication, `Eq` and
45/// `Hash` on a `Dependency` value consider only the name, not the
46/// source location at which the reference occurs.
47#[derive(Debug)]
48pub struct Dependency<'a> {
49    /// The name referred to.
50    pub ident: &'a str,
51
52    /// The location at which the reference to that name occurs.
53    pub usage: Span,
54}
55
56impl Hash for Dependency<'_> {
57    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
58        self.ident.hash(state);
59    }
60}
61
62impl PartialEq for Dependency<'_> {
63    fn eq(&self, other: &Self) -> bool {
64        self.ident == other.ident
65    }
66}

121    /// Local variable and function argument arena.
122    ///
123    /// Note that the `Local` here is actually a zero-sized type. The AST keeps
124    /// all the detailed information about locals - names, types, etc. - in
125    /// [`LocalDecl`] statements. For arguments, that information is kept in
126    /// [`arguments`]. This `Arena`'s only role is to assign a unique `Handle`
127    /// to each of them, and track their definitions' spans for use in
128    /// diagnostics.
129    ///
130    /// In the AST, when an [`Ident`] expression refers to a local variable or
131    /// argument, its [`IdentExpr`] holds the referent's `Handle<Local>` in this
132    /// arena.
133    ///
134    /// During lowering, [`LocalDecl`] statements add entries to a per-function
135    /// table that maps `Handle<Local>` values to their Naga representations,
136    /// accessed via [`StatementContext::local_table`] and
137    /// [`RuntimeExpressionContext::local_table`]. This table is then consulted when
138    /// lowering subsequent [`Ident`] expressions.
139    ///
140    /// [`LocalDecl`]: StatementKind::LocalDecl
141    /// [`arguments`]: Function::arguments
142    /// [`Ident`]: Expression::Ident
143    /// [`StatementContext::local_table`]: StatementContext::local_table
144    /// [`RuntimeExpressionContext::local_table`]: RuntimeExpressionContext::local_table
145    pub locals: Arena<Local>,
146
147    pub body: Block<'a>,
148}

327/// A type at the head of a [`Construct`] expression.
328///
329/// WGSL has two types of [`type constructor expressions`]:
330///
331/// - Those that fully specify the type being constructed, like
332///   `vec3<f32>(x,y,z)`, which obviously constructs a `vec3<f32>`.
333///
334/// - Those that leave the component type of the composite being constructed
335///   implicit, to be inferred from the argument types, like `vec3(x,y,z)`,
336///   which constructs a `vec3<T>` where `T` is the type of `x`, `y`, and `z`.
337///
338/// This enum represents the head type of both cases. The `PartialFoo` variants
339/// represent the second case, where the component type is implicit.
340///
341/// This does not cover structs or types referred to by type aliases. See the
342/// documentation for [`Construct`] and [`Call`] expressions for details.
343///
344/// [`Construct`]: Expression::Construct
345/// [`type constructor expressions`]: https://gpuweb.github.io/gpuweb/wgsl/#type-constructor-expr
346/// [`Call`]: Expression::Call
347#[derive(Debug)]
348pub enum ConstructorType<'a> {
349    /// A scalar type or conversion: `f32(1)`.
350    Scalar(Scalar),
351
352    /// A vector construction whose component type is inferred from the
353    /// argument: `vec3(1.0)`.
354    PartialVector { size: crate::VectorSize },
355
356    /// A vector construction whose component type is written out:
357    /// `vec3<f32>(1.0)`.
358    Vector {
359        size: crate::VectorSize,
360        scalar: Scalar,
361    },
362
363    /// A matrix construction whose component type is inferred from the
364    /// argument: `mat2x2(1,2,3,4)`.
365    PartialMatrix {
366        columns: crate::VectorSize,
367        rows: crate::VectorSize,
368    },
369
370    /// A matrix construction whose component type is written out:
371    /// `mat2x2<f32>(1,2,3,4)`.
372    Matrix {
373        columns: crate::VectorSize,
374        rows: crate::VectorSize,
375        width: crate::Bytes,
376    },
377
378    /// An array whose component type and size are inferred from the arguments:
379    /// `array(3,4,5)`.
380    PartialArray,
381
382    /// An array whose component type and size are written out:
383    /// `array<u32, 4>(3,4,5)`.
384    Array {
385        base: Handle<Type<'a>>,
386        size: ArraySize<'a>,
387    },
388
389    /// Constructing a value of a known Naga IR type.
390    ///
391    /// This variant is produced only during lowering, when we have Naga types
392    /// available, never during parsing.
393    Type(Handle<crate::Type>),
394}

405#[derive(Debug)]
406pub enum Expression<'a> {
407    Literal(Literal),
408    Ident(IdentExpr<'a>),
409
410    /// A type constructor expression.
411    ///
412    /// This is only used for expressions like `KEYWORD(EXPR...)` and
413    /// `KEYWORD<PARAM>(EXPR...)`, where `KEYWORD` is a [type-defining keyword] like
414    /// `vec3`. These keywords cannot be shadowed by user definitions, so we can
415    /// tell that such an expression is a construction immediately.
416    ///
417    /// For ordinary identifiers, we can't tell whether an expression like
418    /// `IDENTIFIER(EXPR, ...)` is a construction expression or a function call
419    /// until we know `IDENTIFIER`'s definition, so we represent those as
420    /// [`Call`] expressions.
421    ///
422    /// [type-defining keyword]: https://gpuweb.github.io/gpuweb/wgsl/#type-defining-keywords
423    /// [`Call`]: Expression::Call
424    Construct {
425        ty: ConstructorType<'a>,
426        ty_span: Span,
427        components: Vec<Handle<Expression<'a>>>,
428    },
429    Unary {
430        op: crate::UnaryOperator,
431        expr: Handle<Expression<'a>>,
432    },
433    AddrOf(Handle<Expression<'a>>),
434    Deref(Handle<Expression<'a>>),
435    Binary {
436        op: crate::BinaryOperator,
437        left: Handle<Expression<'a>>,
438        right: Handle<Expression<'a>>,
439    },
440
441    /// A function call or type constructor expression.
442    ///
443    /// We can't tell whether an expression like `IDENTIFIER(EXPR, ...)` is a
444    /// construction expression or a function call until we know `IDENTIFIER`'s
445    /// definition, so we represent everything of that form as one of these
446    /// expressions until lowering. At that point, [`Lowerer::call`] has
447    /// everything's definition in hand, and can decide whether to emit a Naga
448    /// [`Constant`], [`As`], [`Splat`], or [`Compose`] expression.
449    ///
450    /// [`Lowerer::call`]: Lowerer::call
451    /// [`Constant`]: crate::Expression::Constant
452    /// [`As`]: crate::Expression::As
453    /// [`Splat`]: crate::Expression::Splat
454    /// [`Compose`]: crate::Expression::Compose
455    Call {
456        function: Ident<'a>,
457        arguments: Vec<Handle<Expression<'a>>>,
458    },
459    Index {
460        base: Handle<Expression<'a>>,
461        index: Handle<Expression<'a>>,
462    },
463    Member {
464        base: Handle<Expression<'a>>,
465        field: Ident<'a>,
466    },
467    Bitcast {
468        expr: Handle<Expression<'a>>,
469        to: Handle<Type<'a>>,
470        ty_span: Span,
471    },
472}
