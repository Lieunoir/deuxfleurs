18    pub(super) unsafe fn set_object_name(&self, object: impl vk::Handle, name: &str) {
19        let Some(extension) = self.extension_fns.debug_utils.as_ref() else {
20            return;
21        };
22
23        // Keep variables outside the if-else block to ensure they do not
24        // go out of scope while we hold a pointer to them
25        let mut buffer: [u8; 64] = [0u8; 64];
26        let buffer_vec: Vec<u8>;
27
28        // Append a null terminator to the string
29        let name_bytes = if name.len() < buffer.len() {
30            // Common case, string is very small. Allocate a copy on the stack.
31            buffer[..name.len()].copy_from_slice(name.as_bytes());
32            // Add null terminator
33            buffer[name.len()] = 0;
34            &buffer[..name.len() + 1]
35        } else {
36            // Less common case, the string is large.
37            // This requires a heap allocation.
38            buffer_vec = name
39                .as_bytes()
40                .iter()
41                .cloned()
42                .chain(std::iter::once(0))
43                .collect();
44            &buffer_vec
45        };
46
47        let name = unsafe { CStr::from_bytes_with_nul_unchecked(name_bytes) };
48
49        let _result = unsafe {
50            extension.set_debug_utils_object_name(
51                &vk::DebugUtilsObjectNameInfoEXT::default()
52                    .object_handle(object)
53                    .object_name(name),
54            )
55        };
56    }

58    pub fn make_render_pass(
59        &self,
60        key: super::RenderPassKey,
61    ) -> Result<vk::RenderPass, crate::DeviceError> {
62        Ok(match self.render_passes.lock().entry(key) {
63            Entry::Occupied(e) => *e.get(),
64            Entry::Vacant(e) => {
65                let mut vk_attachments = Vec::new();
66                let mut color_refs = Vec::with_capacity(e.key().colors.len());
67                let mut resolve_refs = Vec::with_capacity(color_refs.capacity());
68                let mut ds_ref = None;
69                let samples = vk::SampleCountFlags::from_raw(e.key().sample_count);
70                let unused = vk::AttachmentReference {
71                    attachment: vk::ATTACHMENT_UNUSED,
72                    layout: vk::ImageLayout::UNDEFINED,
73                };
74                for cat in e.key().colors.iter() {
75                    let (color_ref, resolve_ref) = if let Some(cat) = cat.as_ref() {
76                        let color_ref = vk::AttachmentReference {
77                            attachment: vk_attachments.len() as u32,
78                            layout: cat.base.layout,
79                        };
80                        vk_attachments.push({
81                            let (load_op, store_op) = conv::map_attachment_ops(cat.base.ops);
82                            vk::AttachmentDescription::default()
83                                .format(cat.base.format)
84                                .samples(samples)
85                                .load_op(load_op)
86                                .store_op(store_op)
87                                .initial_layout(cat.base.layout)
88                                .final_layout(cat.base.layout)
89                        });
90                        let resolve_ref = if let Some(ref rat) = cat.resolve {
91                            let (load_op, store_op) = conv::map_attachment_ops(rat.ops);
92                            let vk_attachment = vk::AttachmentDescription::default()
93                                .format(rat.format)
94                                .samples(vk::SampleCountFlags::TYPE_1)
95                                .load_op(load_op)
96                                .store_op(store_op)
97                                .initial_layout(rat.layout)
98                                .final_layout(rat.layout);
99                            vk_attachments.push(vk_attachment);
100
101                            vk::AttachmentReference {
102                                attachment: vk_attachments.len() as u32 - 1,
103                                layout: rat.layout,
104                            }
105                        } else {
106                            unused
107                        };
108
109                        (color_ref, resolve_ref)
110                    } else {
111                        (unused, unused)
112                    };
113
114                    color_refs.push(color_ref);
115                    resolve_refs.push(resolve_ref);
116                }
117
118                if let Some(ref ds) = e.key().depth_stencil {
119                    ds_ref = Some(vk::AttachmentReference {
120                        attachment: vk_attachments.len() as u32,
121                        layout: ds.base.layout,
122                    });
123                    let (load_op, store_op) = conv::map_attachment_ops(ds.base.ops);
124                    let (stencil_load_op, stencil_store_op) =
125                        conv::map_attachment_ops(ds.stencil_ops);
126                    let vk_attachment = vk::AttachmentDescription::default()
127                        .format(ds.base.format)
128                        .samples(samples)
129                        .load_op(load_op)
130                        .store_op(store_op)
131                        .stencil_load_op(stencil_load_op)
132                        .stencil_store_op(stencil_store_op)
133                        .initial_layout(ds.base.layout)
134                        .final_layout(ds.base.layout);
135                    vk_attachments.push(vk_attachment);
136                }
137
138                let vk_subpasses = [{
139                    let mut vk_subpass = vk::SubpassDescription::default()
140                        .pipeline_bind_point(vk::PipelineBindPoint::GRAPHICS)
141                        .color_attachments(&color_refs)
142                        .resolve_attachments(&resolve_refs);
143
144                    if self
145                        .workarounds
146                        .contains(super::Workarounds::EMPTY_RESOLVE_ATTACHMENT_LISTS)
147                        && resolve_refs.is_empty()
148                    {
149                        vk_subpass.p_resolve_attachments = ptr::null();
150                    }
151
152                    if let Some(ref reference) = ds_ref {
153                        vk_subpass = vk_subpass.depth_stencil_attachment(reference)
154                    }
155                    vk_subpass
156                }];
157
158                let mut vk_info = vk::RenderPassCreateInfo::default()
159                    .attachments(&vk_attachments)
160                    .subpasses(&vk_subpasses);
161
162                let mut multiview_info;
163                let mask;
164                if let Some(multiview) = e.key().multiview {
165                    // Sanity checks, better to panic here than cause a driver crash
166                    assert!(multiview.get() <= 8);
167                    assert!(multiview.get() > 1);
168
169                    // Right now we enable all bits on the view masks and correlation masks.
170                    // This means we're rendering to all views in the subpass, and that all views
171                    // can be rendered concurrently.
172                    mask = [(1 << multiview.get()) - 1];
173
174                    // On Vulkan 1.1 or later, this is an alias for core functionality
175                    multiview_info = vk::RenderPassMultiviewCreateInfoKHR::default()
176                        .view_masks(&mask)
177                        .correlation_masks(&mask);
178                    vk_info = vk_info.push_next(&mut multiview_info);
179                }
180
181                let raw = unsafe { self.raw.create_render_pass(&vk_info, None)? };
182
183                *e.insert(raw)
184            }
185        })
186    }

188    pub fn make_framebuffer(
189        &self,
190        key: super::FramebufferKey,
191        raw_pass: vk::RenderPass,
192        pass_label: crate::Label,
193    ) -> Result<vk::Framebuffer, crate::DeviceError> {
194        Ok(match self.framebuffers.lock().entry(key) {
195            Entry::Occupied(e) => *e.get(),
196            Entry::Vacant(e) => {
197                let vk_views = e
198                    .key()
199                    .attachments
200                    .iter()
201                    .map(|at| at.raw)
202                    .collect::<ArrayVec<_, { super::MAX_TOTAL_ATTACHMENTS }>>();
203                let vk_view_formats = e
204                    .key()
205                    .attachments
206                    .iter()
207                    .map(|at| self.private_caps.map_texture_format(at.view_format))
208                    .collect::<ArrayVec<_, { super::MAX_TOTAL_ATTACHMENTS }>>();
209                let vk_view_formats_list = e
210                    .key()
211                    .attachments
212                    .iter()
213                    .map(|at| at.raw_view_formats.clone())
214                    .collect::<ArrayVec<_, { super::MAX_TOTAL_ATTACHMENTS }>>();
215
216                let vk_image_infos = e
217                    .key()
218                    .attachments
219                    .iter()
220                    .enumerate()
221                    .map(|(i, at)| {
222                        let mut info = vk::FramebufferAttachmentImageInfo::default()
223                            .usage(conv::map_texture_usage(at.view_usage))
224                            .flags(at.raw_image_flags)
225                            .width(e.key().extent.width)
226                            .height(e.key().extent.height)
227                            .layer_count(e.key().extent.depth_or_array_layers);
228                        // https://registry.khronos.org/vulkan/specs/1.3-extensions/man/html/VkRenderPassBeginInfo.html#VUID-VkRenderPassBeginInfo-framebuffer-03214
229                        if vk_view_formats_list[i].is_empty() {
230                            info = info.view_formats(&vk_view_formats[i..i + 1]);
231                        } else {
232                            info = info.view_formats(&vk_view_formats_list[i]);
233                        };
234                        info
235                    })
236                    .collect::<ArrayVec<_, { super::MAX_TOTAL_ATTACHMENTS }>>();
237
238                let mut vk_attachment_info = vk::FramebufferAttachmentsCreateInfo::default()
239                    .attachment_image_infos(&vk_image_infos);
240                let mut vk_info = vk::FramebufferCreateInfo::default()
241                    .render_pass(raw_pass)
242                    .width(e.key().extent.width)
243                    .height(e.key().extent.height)
244                    .layers(e.key().extent.depth_or_array_layers);
245
246                if self.private_caps.imageless_framebuffers {
247                    //TODO: https://github.com/MaikKlein/ash/issues/450
248                    vk_info = vk_info
249                        .flags(vk::FramebufferCreateFlags::IMAGELESS_KHR)
250                        .push_next(&mut vk_attachment_info);
251                    vk_info.attachment_count = e.key().attachments.len() as u32;
252                } else {
253                    vk_info = vk_info.attachments(&vk_views);
254                }
255
256                *e.insert(unsafe {
257                    let raw = self.raw.create_framebuffer(&vk_info, None).unwrap();
258                    if let Some(label) = pass_label {
259                        self.set_object_name(raw, label);
260                    }
261                    raw
262                })
263            }
264        })
265    }

267    fn make_memory_ranges<'a, I: 'a + Iterator<Item = crate::MemoryRange>>(
268        &self,
269        buffer: &'a super::Buffer,
270        ranges: I,
271    ) -> Option<impl 'a + Iterator<Item = vk::MappedMemoryRange>> {
272        let block = buffer.block.as_ref()?.lock();
273        let mask = self.private_caps.non_coherent_map_mask;
274        Some(ranges.map(move |range| {
275            vk::MappedMemoryRange::default()
276                .memory(*block.memory())
277                .offset((block.offset() + range.start) & !mask)
278                .size((range.end - range.start + mask) & !mask)
279        }))
280    }

282    unsafe fn free_resources(&self) {
283        for &raw in self.render_passes.lock().values() {
284            unsafe { self.raw.destroy_render_pass(raw, None) };
285        }
286        for &raw in self.framebuffers.lock().values() {
287            unsafe { self.raw.destroy_framebuffer(raw, None) };
288        }
289        if self.handle_is_owned {
290            unsafe { self.raw.destroy_device(None) };
291        }
292    }

297        &self,
298        size: u64,
299        memory_type: u32,
300        flags: gpu_alloc::AllocationFlags,
301    ) -> Result<vk::DeviceMemory, gpu_alloc::OutOfMemory> {
302        let mut info = vk::MemoryAllocateInfo::default()
303            .allocation_size(size)
304            .memory_type_index(memory_type);
305
306        let mut info_flags;
307
308        if flags.contains(gpu_alloc::AllocationFlags::DEVICE_ADDRESS) {
309            info_flags = vk::MemoryAllocateFlagsInfo::default()
310                .flags(vk::MemoryAllocateFlags::DEVICE_ADDRESS);
311            info = info.push_next(&mut info_flags);
312        }
313
314        match unsafe { self.raw.allocate_memory(&info, None) } {
315            Ok(memory) => {
316                self.memory_allocations_counter.add(1);
317                Ok(memory)
318            }
319            Err(vk::Result::ERROR_OUT_OF_DEVICE_MEMORY) => {
320                Err(gpu_alloc::OutOfMemory::OutOfDeviceMemory)
321            }
322            Err(vk::Result::ERROR_OUT_OF_HOST_MEMORY) => {
323                Err(gpu_alloc::OutOfMemory::OutOfHostMemory)
324            }
325            Err(vk::Result::ERROR_TOO_MANY_OBJECTS) => panic!("Too many objects"),
326            Err(err) => panic!("Unexpected Vulkan error: `{err}`"),
327        }
328    }

384    unsafe fn create_descriptor_pool(
385        &self,
386        descriptor_count: &gpu_descriptor::DescriptorTotalCount,
387        max_sets: u32,
388        flags: gpu_descriptor::DescriptorPoolCreateFlags,
389    ) -> Result<vk::DescriptorPool, gpu_descriptor::CreatePoolError> {
390        //Note: ignoring other types, since they can't appear here
391        let unfiltered_counts = [
392            (vk::DescriptorType::SAMPLER, descriptor_count.sampler),
393            (
394                vk::DescriptorType::SAMPLED_IMAGE,
395                descriptor_count.sampled_image,
396            ),
397            (
398                vk::DescriptorType::STORAGE_IMAGE,
399                descriptor_count.storage_image,
400            ),
401            (
402                vk::DescriptorType::UNIFORM_BUFFER,
403                descriptor_count.uniform_buffer,
404            ),
405            (
406                vk::DescriptorType::UNIFORM_BUFFER_DYNAMIC,
407                descriptor_count.uniform_buffer_dynamic,
408            ),
409            (
410                vk::DescriptorType::STORAGE_BUFFER,
411                descriptor_count.storage_buffer,
412            ),
413            (
414                vk::DescriptorType::STORAGE_BUFFER_DYNAMIC,
415                descriptor_count.storage_buffer_dynamic,
416            ),
417        ];
418
419        let filtered_counts = unfiltered_counts
420            .iter()
421            .cloned()
422            .filter(|&(_, count)| count != 0)
423            .map(|(ty, count)| vk::DescriptorPoolSize {
424                ty,
425                descriptor_count: count,
426            })
427            .collect::<ArrayVec<_, 8>>();
428
429        let mut vk_flags =
430            if flags.contains(gpu_descriptor::DescriptorPoolCreateFlags::UPDATE_AFTER_BIND) {
431                vk::DescriptorPoolCreateFlags::UPDATE_AFTER_BIND
432            } else {
433                vk::DescriptorPoolCreateFlags::empty()
434            };
435        if flags.contains(gpu_descriptor::DescriptorPoolCreateFlags::FREE_DESCRIPTOR_SET) {
436            vk_flags |= vk::DescriptorPoolCreateFlags::FREE_DESCRIPTOR_SET;
437        }
438        let vk_info = vk::DescriptorPoolCreateInfo::default()
439            .max_sets(max_sets)
440            .flags(vk_flags)
441            .pool_sizes(&filtered_counts);
442
443        match unsafe { self.raw.create_descriptor_pool(&vk_info, None) } {
444            Ok(pool) => Ok(pool),
445            Err(vk::Result::ERROR_OUT_OF_HOST_MEMORY) => {
446                Err(gpu_descriptor::CreatePoolError::OutOfHostMemory)
447            }
448            Err(vk::Result::ERROR_OUT_OF_DEVICE_MEMORY) => {
449                Err(gpu_descriptor::CreatePoolError::OutOfDeviceMemory)
450            }
451            Err(vk::Result::ERROR_FRAGMENTATION) => {
452                Err(gpu_descriptor::CreatePoolError::Fragmentation)
453            }
454            Err(other) => {
455                log::error!("create_descriptor_pool: {:?}", other);
456                Err(gpu_descriptor::CreatePoolError::OutOfHostMemory)
457            }
458        }
459    }

530    pub(super) unsafe fn create_swapchain(
531        &self,
532        surface: &super::Surface,
533        config: &crate::SurfaceConfiguration,
534        provided_old_swapchain: Option<super::Swapchain>,
535    ) -> Result<super::Swapchain, crate::SurfaceError> {
536        profiling::scope!("Device::create_swapchain");
537        let functor = khr::swapchain::Device::new(&surface.instance.raw, &self.shared.raw);
538
539        let old_swapchain = match provided_old_swapchain {
540            Some(osc) => osc.raw,
541            None => vk::SwapchainKHR::null(),
542        };
543
544        let color_space = if config.format == wgt::TextureFormat::Rgba16Float {
545            // Enable wide color gamut mode
546            // Vulkan swapchain for Android only supports DISPLAY_P3_NONLINEAR_EXT and EXTENDED_SRGB_LINEAR_EXT
547            vk::ColorSpaceKHR::EXTENDED_SRGB_LINEAR_EXT
548        } else {
549            vk::ColorSpaceKHR::SRGB_NONLINEAR
550        };
551
552        let original_format = self.shared.private_caps.map_texture_format(config.format);
553        let mut raw_flags = vk::SwapchainCreateFlagsKHR::empty();
554        let mut raw_view_formats: Vec<vk::Format> = vec![];
555        let mut wgt_view_formats = vec![];
556        if !config.view_formats.is_empty() {
557            raw_flags |= vk::SwapchainCreateFlagsKHR::MUTABLE_FORMAT;
558            raw_view_formats = config
559                .view_formats
560                .iter()
561                .map(|f| self.shared.private_caps.map_texture_format(*f))
562                .collect();
563            raw_view_formats.push(original_format);
564
565            wgt_view_formats.clone_from(&config.view_formats);
566            wgt_view_formats.push(config.format);
567        }
568
569        let mut info = vk::SwapchainCreateInfoKHR::default()
570            .flags(raw_flags)
571            .surface(surface.raw)
572            .min_image_count(config.maximum_frame_latency + 1) // TODO: https://github.com/gfx-rs/wgpu/issues/2869
573            .image_format(original_format)
574            .image_color_space(color_space)
575            .image_extent(vk::Extent2D {
576                width: config.extent.width,
577                height: config.extent.height,
578            })
579            .image_array_layers(config.extent.depth_or_array_layers)
580            .image_usage(conv::map_texture_usage(config.usage))
581            .image_sharing_mode(vk::SharingMode::EXCLUSIVE)
582            .pre_transform(vk::SurfaceTransformFlagsKHR::IDENTITY)
583            .composite_alpha(conv::map_composite_alpha_mode(config.composite_alpha_mode))
584            .present_mode(conv::map_present_mode(config.present_mode))
585            .clipped(true)
586            .old_swapchain(old_swapchain);
587
588        let mut format_list_info = vk::ImageFormatListCreateInfo::default();
589        if !raw_view_formats.is_empty() {
590            format_list_info = format_list_info.view_formats(&raw_view_formats);
591            info = info.push_next(&mut format_list_info);
592        }
593
594        let result = {
595            profiling::scope!("vkCreateSwapchainKHR");
596            unsafe { functor.create_swapchain(&info, None) }
597        };
598
599        // doing this before bailing out with error
600        if old_swapchain != vk::SwapchainKHR::null() {
601            unsafe { functor.destroy_swapchain(old_swapchain, None) }
602        }
603
604        let raw = match result {
605            Ok(swapchain) => swapchain,
606            Err(error) => {
607                return Err(match error {
608                    vk::Result::ERROR_SURFACE_LOST_KHR => crate::SurfaceError::Lost,
609                    vk::Result::ERROR_NATIVE_WINDOW_IN_USE_KHR => {
610                        crate::SurfaceError::Other("Native window is in use")
611                    }
612                    other => crate::DeviceError::from(other).into(),
613                })
614            }
615        };
616
617        let images =
618            unsafe { functor.get_swapchain_images(raw) }.map_err(crate::DeviceError::from)?;
619
620        // NOTE: It's important that we define at least images.len() wait
621        // semaphores, since we prospectively need to provide the call to
622        // acquire the next image with an unsignaled semaphore.
623        let surface_semaphores = (0..=images.len())
624            .map(|_| {
625                super::SwapchainImageSemaphores::new(&self.shared)
626                    .map(Mutex::new)
627                    .map(Arc::new)

644    /// # Safety
645    ///
646    /// - `vk_image` must be created respecting `desc`
647    /// - If `drop_guard` is `Some`, the application must manually destroy the image handle. This
648    ///   can be done inside the `Drop` impl of `drop_guard`.
649    /// - If the `ImageCreateFlags` does not contain `MUTABLE_FORMAT`, the `view_formats` of `desc` must be empty.
650    pub unsafe fn texture_from_raw(
651        vk_image: vk::Image,
652        desc: &crate::TextureDescriptor,
653        drop_guard: Option<crate::DropGuard>,
654    ) -> super::Texture {
655        let mut raw_flags = vk::ImageCreateFlags::empty();
656        let mut view_formats = vec![];
657        for tf in desc.view_formats.iter() {
658            if *tf == desc.format {
659                continue;
660            }
661            view_formats.push(*tf);
662        }
663        if !view_formats.is_empty() {
664            raw_flags |=
665                vk::ImageCreateFlags::MUTABLE_FORMAT | vk::ImageCreateFlags::EXTENDED_USAGE;
666            view_formats.push(desc.format)
667        }
668        if desc.format.is_multi_planar_format() {
669            raw_flags |= vk::ImageCreateFlags::MUTABLE_FORMAT;
670        }
671
672        super::Texture {
673            raw: vk_image,
674            drop_guard,
675            block: None,
676            usage: desc.usage,
677            format: desc.format,
678            raw_flags: vk::ImageCreateFlags::empty(),
679            copy_size: desc.copy_extent(),
680            view_formats,
681        }
682    }

710    fn compile_stage(
711        &self,
712        stage: &crate::ProgrammableStage<super::Api>,
713        naga_stage: naga::ShaderStage,
714        binding_map: &naga::back::spv::BindingMap,
715    ) -> Result<CompiledStage, crate::PipelineError> {
716        let stage_flags = crate::auxil::map_naga_stage(naga_stage);
717        let vk_module = match *stage.module {
718            super::ShaderModule::Raw(raw) => raw,
719            super::ShaderModule::Intermediate {
720                ref naga_shader,
721                runtime_checks,
722            } => {
723                let pipeline_options = naga::back::spv::PipelineOptions {
724                    entry_point: stage.entry_point.to_string(),
725                    shader_stage: naga_stage,
726                };
727                let needs_temp_options = !runtime_checks
728                    || !binding_map.is_empty()
729                    || naga_shader.debug_source.is_some()
730                    || !stage.zero_initialize_workgroup_memory;
731                let mut temp_options;
732                let options = if needs_temp_options {
733                    temp_options = self.naga_options.clone();
734                    if !runtime_checks {
735                        temp_options.bounds_check_policies = naga::proc::BoundsCheckPolicies {
736                            index: naga::proc::BoundsCheckPolicy::Unchecked,
737                            buffer: naga::proc::BoundsCheckPolicy::Unchecked,
738                            image_load: naga::proc::BoundsCheckPolicy::Unchecked,
739                            image_store: naga::proc::BoundsCheckPolicy::Unchecked,
740                            binding_array: naga::proc::BoundsCheckPolicy::Unchecked,
741                        };
742                    }
743                    if !binding_map.is_empty() {
744                        temp_options.binding_map = binding_map.clone();
745                    }
746
747                    if let Some(ref debug) = naga_shader.debug_source {
748                        temp_options.debug_info = Some(naga::back::spv::DebugInfo {
749                            source_code: &debug.source_code,
750                            file_name: debug.file_name.as_ref().as_ref(),
751                        })
752                    }
753                    if !stage.zero_initialize_workgroup_memory {
754                        temp_options.zero_initialize_workgroup_memory =
755                            naga::back::spv::ZeroInitializeWorkgroupMemoryMode::None;
756                    }
757
758                    &temp_options
759                } else {
760                    &self.naga_options
761                };
762
763                let (module, info) = naga::back::pipeline_constants::process_overrides(
764                    &naga_shader.module,
765                    &naga_shader.info,
766                    stage.constants,
767                )
768                .map_err(|e| crate::PipelineError::Linkage(stage_flags, format!("{e}")))?;
769
770                let spv = {
771                    profiling::scope!("naga::spv::write_vec");
772                    naga::back::spv::write_vec(&module, &info, options, Some(&pipeline_options))
773                }
774                .map_err(|e| crate::PipelineError::Linkage(stage_flags, format!("{e}")))?;
775                self.create_shader_module_impl(&spv)?
776            }
777        };
778
779        let mut flags = vk::PipelineShaderStageCreateFlags::empty();
780        if self.shared.features.contains(wgt::Features::SUBGROUP) {
781            flags |= vk::PipelineShaderStageCreateFlags::ALLOW_VARYING_SUBGROUP_SIZE
782        }
783
784        let entry_point = CString::new(stage.entry_point).unwrap();
785        let mut create_info = vk::PipelineShaderStageCreateInfo::default()
786            .flags(flags)
787            .stage(conv::map_shader_stage(stage_flags))
788            .module(vk_module);
789
790        // Circumvent struct lifetime check because of a self-reference inside CompiledStage
791        create_info.p_name = entry_point.as_ptr();
792
793        Ok(CompiledStage {
794            create_info,
795            _entry_point: entry_point,
796            temp_raw_module: match *stage.module {
797                super::ShaderModule::Raw(_) => None,
798                super::ShaderModule::Intermediate { .. } => Some(vk_module),
799            },
800        })
801    }

852    unsafe fn create_buffer(
853        &self,
854        desc: &crate::BufferDescriptor,
855    ) -> Result<super::Buffer, crate::DeviceError> {
856        let vk_info = vk::BufferCreateInfo::default()
857            .size(desc.size)
858            .usage(conv::map_buffer_usage(desc.usage))
859            .sharing_mode(vk::SharingMode::EXCLUSIVE);
860
861        let raw = unsafe { self.shared.raw.create_buffer(&vk_info, None)? };
862        let req = unsafe { self.shared.raw.get_buffer_memory_requirements(raw) };
863
864        let mut alloc_usage = if desc
865            .usage
866            .intersects(crate::BufferUses::MAP_READ | crate::BufferUses::MAP_WRITE)
867        {
868            let mut flags = gpu_alloc::UsageFlags::HOST_ACCESS;
869            //TODO: find a way to use `crate::MemoryFlags::PREFER_COHERENT`
870            flags.set(
871                gpu_alloc::UsageFlags::DOWNLOAD,
872                desc.usage.contains(crate::BufferUses::MAP_READ),
873            );
874            flags.set(
875                gpu_alloc::UsageFlags::UPLOAD,
876                desc.usage.contains(crate::BufferUses::MAP_WRITE),
877            );
878            flags
879        } else {
880            gpu_alloc::UsageFlags::FAST_DEVICE_ACCESS
881        };
882        alloc_usage.set(
883            gpu_alloc::UsageFlags::TRANSIENT,
884            desc.memory_flags.contains(crate::MemoryFlags::TRANSIENT),
885        );
886
887        let alignment_mask = if desc.usage.intersects(
888            crate::BufferUses::TOP_LEVEL_ACCELERATION_STRUCTURE_INPUT
889                | crate::BufferUses::BOTTOM_LEVEL_ACCELERATION_STRUCTURE_INPUT,
890        ) {
891            16
892        } else {
893            req.alignment
894        } - 1;
895
896        let block = unsafe {
897            self.mem_allocator.lock().alloc(
898                &*self.shared,
899                gpu_alloc::Request {
900                    size: req.size,
901                    align_mask: alignment_mask,
902                    usage: alloc_usage,
903                    memory_types: req.memory_type_bits & self.valid_ash_memory_types,
904                },
905            )?
906        };
907
908        unsafe {
909            self.shared
910                .raw
911                .bind_buffer_memory(raw, *block.memory(), block.offset())?
912        };
913
914        if let Some(label) = desc.label {
915            unsafe { self.shared.set_object_name(raw, label) };
916        }
917
918        self.counters.buffer_memory.add(block.size() as isize);
919        self.counters.buffers.add(1);
920
921        Ok(super::Buffer {
922            raw,
923            block: Some(Mutex::new(block)),
924        })
925    }

992    unsafe fn create_texture(
993        &self,
994        desc: &crate::TextureDescriptor,
995    ) -> Result<super::Texture, crate::DeviceError> {
996        let copy_size = desc.copy_extent();
997
998        let mut raw_flags = vk::ImageCreateFlags::empty();
999        if desc.is_cube_compatible() {
1000            raw_flags |= vk::ImageCreateFlags::CUBE_COMPATIBLE;
1001        }
1002
1003        let original_format = self.shared.private_caps.map_texture_format(desc.format);
1004        let mut vk_view_formats = vec![];
1005        let mut wgt_view_formats = vec![];
1006        if !desc.view_formats.is_empty() {
1007            raw_flags |= vk::ImageCreateFlags::MUTABLE_FORMAT;
1008            wgt_view_formats.clone_from(&desc.view_formats);
1009            wgt_view_formats.push(desc.format);
1010
1011            if self.shared.private_caps.image_format_list {
1012                vk_view_formats = desc
1013                    .view_formats
1014                    .iter()
1015                    .map(|f| self.shared.private_caps.map_texture_format(*f))
1016                    .collect();
1017                vk_view_formats.push(original_format)
1018            }
1019        }
1020        if desc.format.is_multi_planar_format() {
1021            raw_flags |= vk::ImageCreateFlags::MUTABLE_FORMAT;
1022        }
1023
1024        let mut vk_info = vk::ImageCreateInfo::default()
1025            .flags(raw_flags)
1026            .image_type(conv::map_texture_dimension(desc.dimension))
1027            .format(original_format)
1028            .extent(conv::map_copy_extent(&copy_size))
1029            .mip_levels(desc.mip_level_count)
1030            .array_layers(desc.array_layer_count())
1031            .samples(vk::SampleCountFlags::from_raw(desc.sample_count))
1032            .tiling(vk::ImageTiling::OPTIMAL)
1033            .usage(conv::map_texture_usage(desc.usage))
1034            .sharing_mode(vk::SharingMode::EXCLUSIVE)
1035            .initial_layout(vk::ImageLayout::UNDEFINED);
1036
1037        let mut format_list_info = vk::ImageFormatListCreateInfo::default();
1038        if !vk_view_formats.is_empty() {
1039            format_list_info = format_list_info.view_formats(&vk_view_formats);
1040            vk_info = vk_info.push_next(&mut format_list_info);
1041        }
1042
1043        let raw = unsafe { self.shared.raw.create_image(&vk_info, None)? };
1044        let req = unsafe { self.shared.raw.get_image_memory_requirements(raw) };
1045
1046        let block = unsafe {
1047            self.mem_allocator.lock().alloc(
1048                &*self.shared,
1049                gpu_alloc::Request {
1050                    size: req.size,
1051                    align_mask: req.alignment - 1,
1052                    usage: gpu_alloc::UsageFlags::FAST_DEVICE_ACCESS,
1053                    memory_types: req.memory_type_bits & self.valid_ash_memory_types,
1054                },
1055            )?
1056        };
1057
1058        self.counters.texture_memory.add(block.size() as isize);
1059
1060        unsafe {
1061            self.shared
1062                .raw
1063                .bind_image_memory(raw, *block.memory(), block.offset())?
1064        };
1065
1066        if let Some(label) = desc.label {
1067            unsafe { self.shared.set_object_name(raw, label) };
1068        }
1069
1070        self.counters.textures.add(1);
1071
1072        Ok(super::Texture {
1073            raw,
1074            drop_guard: None,
1075            block: Some(block),
1076            usage: desc.usage,
1077            format: desc.format,
1078            raw_flags,
1079            copy_size,
1080            view_formats: wgt_view_formats,
1081        })
1082    }

1254        let mut desc_count = gpu_descriptor::DescriptorTotalCount::default();
1255        let mut types = Vec::new();
1256        for entry in desc.entries {
1257            let count = entry.count.map_or(1, |c| c.get());
1258            if entry.binding as usize >= types.len() {
1259                types.resize(
1260                    entry.binding as usize + 1,
1261                    (vk::DescriptorType::INPUT_ATTACHMENT, 0),
1262                );
1263            }
1264            types[entry.binding as usize] = (
1265                conv::map_binding_type(entry.ty),
1266                entry.count.map_or(1, |c| c.get()),
1267            );
1268
1269            match entry.ty {
1270                wgt::BindingType::Buffer {
1271                    ty,
1272                    has_dynamic_offset,
1273                    ..
1274                } => match ty {
1275                    wgt::BufferBindingType::Uniform => {
1276                        if has_dynamic_offset {
1277                            desc_count.uniform_buffer_dynamic += count;
1278                        } else {
1279                            desc_count.uniform_buffer += count;
1280                        }
1281                    }
1282                    wgt::BufferBindingType::Storage { .. } => {
1283                        if has_dynamic_offset {
1284                            desc_count.storage_buffer_dynamic += count;
1285                        } else {
1286                            desc_count.storage_buffer += count;
1287                        }
1288                    }
1289                },
1290                wgt::BindingType::Sampler { .. } => {
1291                    desc_count.sampler += count;
1292                }
1293                wgt::BindingType::Texture { .. } => {
1294                    desc_count.sampled_image += count;
1295                }
1296                wgt::BindingType::StorageTexture { .. } => {
1297                    desc_count.storage_image += count;
1298                }
1299                wgt::BindingType::AccelerationStructure => {
1300                    desc_count.acceleration_structure += count;
1301                }
1302            }
1303        }

1306        let vk_bindings = desc
1307            .entries
1308            .iter()
1309            .map(|entry| vk::DescriptorSetLayoutBinding {
1310                binding: entry.binding,
1311                descriptor_type: types[entry.binding as usize].0,
1312                descriptor_count: types[entry.binding as usize].1,
1313                stage_flags: conv::map_shader_stage(entry.visibility),
1314                p_immutable_samplers: ptr::null(),
1315                _marker: Default::default(),
1316            })
1317            .collect::<Vec<_>>();
1318
1319        let vk_info = vk::DescriptorSetLayoutCreateInfo::default().bindings(&vk_bindings);
1320
1321        let binding_arrays = desc
1322            .entries
1323            .iter()
1324            .enumerate()
1325            .filter_map(|(idx, entry)| entry.count.map(|count| (idx as u32, count)))
1326            .collect();
1327
1328        let mut binding_flag_info;
1329        let binding_flag_vec;
1330
1331        let partially_bound = desc
1332            .flags
1333            .contains(crate::BindGroupLayoutFlags::PARTIALLY_BOUND);
1334
1335        let vk_info = if partially_bound {
1336            binding_flag_vec = desc
1337                .entries
1338                .iter()
1339                .map(|entry| {
1340                    let mut flags = vk::DescriptorBindingFlags::empty();
1341
1342                    if partially_bound && entry.count.is_some() {
1343                        flags |= vk::DescriptorBindingFlags::PARTIALLY_BOUND;
1344                    }
1345
1346                    flags
1347                })
1348                .collect::<Vec<_>>();
1349
1350            binding_flag_info = vk::DescriptorSetLayoutBindingFlagsCreateInfo::default()
1351                .binding_flags(&binding_flag_vec);
1352
1353            vk_info.push_next(&mut binding_flag_info)
1354        } else {
1355            vk_info
1356        };
1357
1358        let raw = unsafe {
1359            self.shared
1360                .raw
1361                .create_descriptor_set_layout(&vk_info, None)?
1362        };
1363
1364        if let Some(label) = desc.label {
1365            unsafe { self.shared.set_object_name(raw, label) };
1366        }
1367
1368        self.counters.bind_group_layouts.add(1);
1369
1370        Ok(super::BindGroupLayout {
1371            raw,
1372            desc_count,
1373            types: types.into_boxed_slice(),
1374            binding_arrays,
1375        })

1453    unsafe fn create_bind_group(
1454        &self,
1455        desc: &crate::BindGroupDescriptor<super::Api>,
1456    ) -> Result<super::BindGroup, crate::DeviceError> {
1457        let mut vk_sets = unsafe {
1458            self.desc_allocator.lock().allocate(
1459                &*self.shared,
1460                &desc.layout.raw,
1461                gpu_descriptor::DescriptorSetLayoutCreateFlags::empty(),
1462                &desc.layout.desc_count,
1463                1,
1464            )?
1465        };
1466
1467        let set = vk_sets.pop().unwrap();
1468        if let Some(label) = desc.label {
1469            unsafe { self.shared.set_object_name(*set.raw(), label) };
1470        }
1471
1472        /// Helper for splitting off and initializing a given number of elements on a pre-allocated
1473        /// stack, based on items returned from an [`ExactSizeIterator`].  Typically created from a
1474        /// [`MaybeUninit`] slice (see [`Vec::spare_capacity_mut()`]).
1475        /// The updated [`ExtensionStack`] of remaining uninitialized elements is returned, safely
1476        /// representing that the initialized and remaining elements are two independent mutable
1477        /// borrows.
1478        struct ExtendStack<'a, T> {
1479            remainder: &'a mut [MaybeUninit<T>],
1480        }
1481
1482        impl<'a, T> ExtendStack<'a, T> {
1483            fn from_vec_capacity(vec: &'a mut Vec<T>) -> Self {
1484                Self {
1485                    remainder: vec.spare_capacity_mut(),
1486                }
1487            }
1488
1489            fn extend_one(self, value: T) -> (Self, &'a mut T) {
1490                let (to_init, remainder) = self.remainder.split_first_mut().unwrap();
1491                let init = to_init.write(value);
1492                (Self { remainder }, init)
1493            }
1494
1495            fn extend(
1496                self,
1497                iter: impl IntoIterator<Item = T> + ExactSizeIterator,
1498            ) -> (Self, &'a mut [T]) {
1499                let (to_init, remainder) = self.remainder.split_at_mut(iter.len());
1500
1501                for (value, to_init) in iter.into_iter().zip(to_init.iter_mut()) {
1502                    to_init.write(value);
1503                }
1504
1505                // we can't use the safe (yet unstable) MaybeUninit::write_slice() here because of having an iterator to write
1506
1507                let init = {
1508                    #[allow(trivial_casts)]
1509                    // SAFETY: The loop above has initialized exactly as many items as to_init is
1510                    // long, so it is safe to cast away the MaybeUninit<T> wrapper into T.
1511
1512                    // Additional safety docs from unstable slice_assume_init_mut
1513                    // SAFETY: similar to safety notes for `slice_get_ref`, but we have a
1514                    // mutable reference which is also guaranteed to be valid for writes.
1515                    unsafe {
1516                        &mut *(to_init as *mut [MaybeUninit<T>] as *mut [T])
1517                    }
1518                };
1519                (Self { remainder }, init)
1520            }
1521        }
1522
1523        let mut writes = Vec::with_capacity(desc.entries.len());
1524        let mut buffer_infos = Vec::with_capacity(desc.buffers.len());
1525        let mut buffer_infos = ExtendStack::from_vec_capacity(&mut buffer_infos);
1526        let mut image_infos = Vec::with_capacity(desc.samplers.len() + desc.textures.len());
1527        let mut image_infos = ExtendStack::from_vec_capacity(&mut image_infos);
1528        // TODO: This length could be reduced to just the number of top-level acceleration
1529        // structure bindings, where multiple consecutive TLAS bindings that are set via
1530        // one `WriteDescriptorSet` count towards one "info" struct, not the total number of
1531        // acceleration structure bindings to write:
1532        let mut acceleration_structure_infos =
1533            Vec::with_capacity(desc.acceleration_structures.len());
1534        let mut acceleration_structure_infos =
1535            ExtendStack::from_vec_capacity(&mut acceleration_structure_infos);
1536        let mut raw_acceleration_structures =
1537            Vec::with_capacity(desc.acceleration_structures.len());
1538        let mut raw_acceleration_structures =
1539            ExtendStack::from_vec_capacity(&mut raw_acceleration_structures);
1540        for entry in desc.entries {
1541            let (ty, size) = desc.layout.types[entry.binding as usize];
1542            if size == 0 {
1543                continue; // empty slot
1544            }
1545            let mut write = vk::WriteDescriptorSet::default()
1546                .dst_set(*set.raw())
1547                .dst_binding(entry.binding)
1548                .descriptor_type(ty);
1549
1550            write = match ty {
1551                vk::DescriptorType::SAMPLER => {
1552                    let start = entry.resource_index;
1553                    let end = start + entry.count;
1554                    let local_image_infos;
1555                    (image_infos, local_image_infos) =
1556                        image_infos.extend(desc.samplers[start as usize..end as usize].iter().map(
1557                            |sampler| vk::DescriptorImageInfo::default().sampler(sampler.raw),
1558                        ));
1559                    write.image_info(local_image_infos)
1560                }
1561                vk::DescriptorType::SAMPLED_IMAGE | vk::DescriptorType::STORAGE_IMAGE => {
1562                    let start = entry.resource_index;
1563                    let end = start + entry.count;
1564                    let local_image_infos;
1565                    (image_infos, local_image_infos) =
1566                        image_infos.extend(desc.textures[start as usize..end as usize].iter().map(
1567                            |binding| {
1568                                let layout = conv::derive_image_layout(
1569                                    binding.usage,
1570                                    binding.view.attachment.view_format,
1571                                );
1572                                vk::DescriptorImageInfo::default()
1573                                    .image_view(binding.view.raw)
1574                                    .image_layout(layout)
1575                            },
1576                        ));
1577                    write.image_info(local_image_infos)
1578                }
1579                vk::DescriptorType::UNIFORM_BUFFER
1580                | vk::DescriptorType::UNIFORM_BUFFER_DYNAMIC
1581                | vk::DescriptorType::STORAGE_BUFFER
1582                | vk::DescriptorType::STORAGE_BUFFER_DYNAMIC => {
1583                    let start = entry.resource_index;
1584                    let end = start + entry.count;
1585                    let local_buffer_infos;
1586                    (buffer_infos, local_buffer_infos) =
1587                        buffer_infos.extend(desc.buffers[start as usize..end as usize].iter().map(
1588                            |binding| {
1589                                vk::DescriptorBufferInfo::default()
1590                                    .buffer(binding.buffer.raw)
1591                                    .offset(binding.offset)
1592                                    .range(
1593                                        binding.size.map_or(vk::WHOLE_SIZE, wgt::BufferSize::get),
1594                                    )
1595                            },
1596                        ));
1597                    write.buffer_info(local_buffer_infos)
1598                }
1599                vk::DescriptorType::ACCELERATION_STRUCTURE_KHR => {
1600                    let start = entry.resource_index;
1601                    let end = start + entry.count;
1602
1603                    let local_raw_acceleration_structures;
1604                    (
1605                        raw_acceleration_structures,
1606                        local_raw_acceleration_structures,
1607                    ) = raw_acceleration_structures.extend(
1608                        desc.acceleration_structures[start as usize..end as usize]
1609                            .iter()
1610                            .map(|acceleration_structure| acceleration_structure.raw),
1611                    );
1612
1613                    let local_acceleration_structure_infos;
1614                    (
1615                        acceleration_structure_infos,
1616                        local_acceleration_structure_infos,
1617                    ) = acceleration_structure_infos.extend_one(
1618                        vk::WriteDescriptorSetAccelerationStructureKHR::default()
1619                            .acceleration_structures(local_raw_acceleration_structures),
1620                    );
1621
1622                    write
1623                        .descriptor_count(entry.count)
1624                        .push_next(local_acceleration_structure_infos)
1625                }
1626                _ => unreachable!(),
1627            };
1628
1629            writes.push(write);
1630        }
1631
1632        unsafe { self.shared.raw.update_descriptor_sets(&writes, &[]) };
1633
1634        self.counters.bind_groups.add(1);
1635
1636        Ok(super::BindGroup { set })
1637    }

1786            .polygon_mode(conv::map_polygon_mode(desc.primitive.polygon_mode))

1832                vk_rasterization = vk_rasterization
1833                    .depth_bias_enable(true)
1834                    .depth_bias_constant_factor(ds.bias.constant as f32)
