1/*!
2# Vulkan API internals.
3
4## Stack memory
5
6Ash expects slices, which we don't generally have available.
7We cope with this requirement by the combination of the following ways:
8  - temporarily allocating `Vec` on heap, where overhead is permitted
9  - growing temporary local storage
10  - using `implace_it` on iterators
11
12## Framebuffers and Render passes
13
14Render passes are cached on the device and kept forever.
15
16Framebuffers are also cached on the device, but they are removed when
17any of the image views (they have) gets removed.
18If Vulkan supports image-less framebuffers,
19then the actual views are excluded from the framebuffer key.
20
21## Fences
22
23If timeline semaphores are available, they are used 1:1 with wgpu-hal fences.
24Otherwise, we manage a pool of `VkFence` objects behind each `hal::Fence`.
25
26!*/

149/// The semaphores needed to use one image in a swapchain.
150#[derive(Debug)]
151struct SwapchainImageSemaphores {
152    /// A semaphore that is signaled when this image is safe for us to modify.
153    ///
154    /// When [`vkAcquireNextImageKHR`] returns the index of the next swapchain
155    /// image that we should use, that image may actually still be in use by the
156    /// presentation engine, and is not yet safe to modify. However, that
157    /// function does accept a semaphore that it will signal when the image is
158    /// indeed safe to begin messing with.
159    ///
160    /// This semaphore is:
161    ///
162    /// - waited for by the first queue submission to operate on this image
163    ///   since it was acquired, and
164    ///
165    /// - signaled by [`vkAcquireNextImageKHR`] when the acquired image is ready
166    ///   for us to use.
167    ///
168    /// [`vkAcquireNextImageKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkAcquireNextImageKHR
169    acquire: vk::Semaphore,
170
171    /// True if the next command submission operating on this image should wait
172    /// for [`acquire`].
173    ///
174    /// We must wait for `acquire` before drawing to this swapchain image, but
175    /// because `wgpu-hal` queue submissions are always strongly ordered, only
176    /// the first submission that works with a swapchain image actually needs to
177    /// wait. We set this flag when this image is acquired, and clear it the
178    /// first time it's passed to [`Queue::submit`] as a surface texture.
179    ///
180    /// [`acquire`]: SwapchainImageSemaphores::acquire
181    /// [`Queue::submit`]: crate::Queue::submit
182    should_wait_for_acquire: bool,
183
184    /// A pool of semaphores for ordering presentation after drawing.
185    ///
186    /// The first [`present_index`] semaphores in this vector are:
187    ///
188    /// - all waited on by the call to [`vkQueuePresentKHR`] that presents this
189    ///   image, and
190    ///
191    /// - each signaled by some [`vkQueueSubmit`] queue submission that draws to
192    ///   this image, when the submission finishes execution.
193    ///
194    /// This vector accumulates one semaphore per submission that writes to this
195    /// image. This is awkward, but hard to avoid: [`vkQueuePresentKHR`]
196    /// requires a semaphore to order it with respect to drawing commands, and
197    /// we can't attach new completion semaphores to a command submission after
198    /// it's been submitted. This means that, at submission time, we must create
199    /// the semaphore we might need if the caller's next action is to enqueue a
200    /// presentation of this image.
201    ///
202    /// An alternative strategy would be for presentation to enqueue an empty
203    /// submit, ordered relative to other submits in the usual way, and
204    /// signaling a single presentation semaphore. But we suspect that submits
205    /// are usually expensive enough, and semaphores usually cheap enough, that
206    /// performance-sensitive users will avoid making many submits, so that the
207    /// cost of accumulated semaphores will usually be less than the cost of an
208    /// additional submit.
209    ///
210    /// Only the first [`present_index`] semaphores in the vector are actually
211    /// going to be signalled by submitted commands, and need to be waited for
212    /// by the next present call. Any semaphores beyond that index were created
213    /// for prior presents and are simply being retained for recycling.
214    ///
215    /// [`present_index`]: SwapchainImageSemaphores::present_index
216    /// [`vkQueuePresentKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueuePresentKHR
217    /// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
218    present: Vec<vk::Semaphore>,
219
220    /// The number of semaphores in [`present`] to be signalled for this submission.
221    ///
222    /// [`present`]: SwapchainImageSemaphores::present
223    present_index: usize,
224
225    /// The fence value of the last command submission that wrote to this image.
226    ///
227    /// The next time we try to acquire this image, we'll block until
228    /// this submission finishes, proving that [`acquire`] is ready to
229    /// pass to `vkAcquireNextImageKHR` again.
230    ///
231    /// [`acquire`]: SwapchainImageSemaphores::acquire
232    previously_used_submission_index: crate::FenceValue,
233}
234
235impl SwapchainImageSemaphores {
236    fn new(device: &DeviceShared) -> Result<Self, crate::DeviceError> {
237        Ok(Self {
238            acquire: device.new_binary_semaphore()?,
239            should_wait_for_acquire: true,
240            present: Vec::new(),
241            present_index: 0,
242            previously_used_submission_index: 0,
243        })
244    }
245
246    fn set_used_fence_value(&mut self, value: crate::FenceValue) {
247        self.previously_used_submission_index = value;
248    }
249
250    /// Return the semaphore that commands drawing to this image should wait for, if any.
251    ///
252    /// This only returns `Some` once per acquisition; see
253    /// [`SwapchainImageSemaphores::should_wait_for_acquire`] for details.
254    fn get_acquire_wait_semaphore(&mut self) -> Option<vk::Semaphore> {
255        if self.should_wait_for_acquire {
256            self.should_wait_for_acquire = false;
257            Some(self.acquire)
258        } else {
259            None
260        }
261    }
262
263    /// Return a semaphore that a submission that writes to this image should
264    /// signal when it's done.
265    ///
266    /// See [`SwapchainImageSemaphores::present`] for details.
267    fn get_submit_signal_semaphore(
268        &mut self,
269        device: &DeviceShared,
270    ) -> Result<vk::Semaphore, crate::DeviceError> {
271        // Try to recycle a semaphore we created for a previous presentation.
272        let sem = match self.present.get(self.present_index) {
273            Some(sem) => *sem,
274            None => {
275                let sem = device.new_binary_semaphore()?;
276                self.present.push(sem);
277                sem
278            }
279        };
280
281        self.present_index += 1;
282
283        Ok(sem)
284    }
285
286    /// Return the semaphores that a presentation of this image should wait on.
287    ///
288    /// Return a slice of semaphores that the call to [`vkQueueSubmit`] that
289    /// ends this image's acquisition should wait for. See
290    /// [`SwapchainImageSemaphores::present`] for details.
291    ///
292    /// Reset `self` to be ready for the next acquisition cycle.
293    ///
294    /// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
295    fn get_present_wait_semaphores(&mut self) -> &[vk::Semaphore] {
296        let old_index = self.present_index;
297
298        // Since this marks the end of this acquire/draw/present cycle, take the
299        // opportunity to reset `self` in preparation for the next acquisition.
300        self.present_index = 0;
301        self.should_wait_for_acquire = true;
302
303        &self.present[0..old_index]
304    }
305
306    unsafe fn destroy(&self, device: &ash::Device) {
307        unsafe {
308            device.destroy_semaphore(self.acquire, None);
309            for sem in &self.present {
310                device.destroy_semaphore(*sem, None);
311            }
312        }
313    }
314}

338        let semaphore_count = self.surface_semaphores.len();
339        self.next_semaphore_index = (self.next_semaphore_index + 1) % semaphore_count;
340    }
341
342    fn get_surface_semaphores(&self) -> Arc<Mutex<SwapchainImageSemaphores>> {
343        self.surface_semaphores[self.next_semaphore_index].clone()
344    }

546/// Semaphores for forcing queue submissions to run in order.
547///
548/// The [`wgpu_hal::Queue`] trait promises that if two calls to [`submit`] are
549/// ordered, then the first submission will finish on the GPU before the second
550/// submission begins. To get this behavior on Vulkan we need to pass semaphores
551/// to [`vkQueueSubmit`] for the commands to wait on before beginning execution,
552/// and to signal when their execution is done.
553///
554/// Normally this can be done with a single semaphore, waited on and then
555/// signalled for each submission. At any given time there's exactly one
556/// submission that would signal the semaphore, and exactly one waiting on it,
557/// as Vulkan requires.
558///
559/// However, as of Oct 2021, bug [#5508] in the Mesa ANV drivers caused them to
560/// hang if we use a single semaphore. The workaround is to alternate between
561/// two semaphores. The bug has been fixed in Mesa, but we should probably keep
562/// the workaround until, say, Oct 2026.
563///
564/// [`wgpu_hal::Queue`]: crate::Queue
565/// [`submit`]: crate::Queue::submit
566/// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
567/// [#5508]: https://gitlab.freedesktop.org/mesa/mesa/-/issues/5508
568#[derive(Clone)]
569struct RelaySemaphores {
570    /// The semaphore the next submission should wait on before beginning
571    /// execution on the GPU. This is `None` for the first submission, which
572    /// should not wait on anything at all.
573    wait: Option<vk::Semaphore>,
574
575    /// The semaphore the next submission should signal when it has finished
576    /// execution on the GPU.
577    signal: vk::Semaphore,
578}
579
580impl RelaySemaphores {
581    fn new(device: &DeviceShared) -> Result<Self, crate::DeviceError> {
582        Ok(Self {
583            wait: None,
584            signal: device.new_binary_semaphore()?,
585        })
586    }
587
588    /// Advances the semaphores, returning the semaphores that should be used for a submission.
589    fn advance(&mut self, device: &DeviceShared) -> Result<Self, crate::DeviceError> {
590        let old = self.clone();
591
592        // Build the state for the next submission.
593        match self.wait {
594            None => {
595                // The `old` values describe the first submission to this queue.
596                // The second submission should wait on `old.signal`, and then
597                // signal a new semaphore which we'll create now.
598                self.wait = Some(old.signal);
599                self.signal = device.new_binary_semaphore()?;
600            }
601            Some(ref mut wait) => {
602                // What this submission signals, the next should wait.
603                mem::swap(wait, &mut self.signal);
604            }
605        };
606
607        Ok(old)
608    }
609
610    /// Destroys the semaphores.
611    unsafe fn destroy(&self, device: &ash::Device) {
612        unsafe {
613            if let Some(wait) = self.wait {
614                device.destroy_semaphore(wait, None);
615            }
616            device.destroy_semaphore(self.signal, None);
617        }
618    }
619}

817/// The [`Api::Fence`] type for [`vulkan::Api`].
818///
819/// This is an `enum` because there are two possible implementations of
820/// `wgpu-hal` fences on Vulkan: Vulkan fences, which work on any version of
821/// Vulkan, and Vulkan timeline semaphores, which are easier and cheaper but
822/// require non-1.0 features.
823///
824/// [`Device::create_fence`] returns a [`TimelineSemaphore`] if
825/// [`VK_KHR_timeline_semaphore`] is available and enabled, and a [`FencePool`]
826/// otherwise.
827///
828/// [`Api::Fence`]: crate::Api::Fence
829/// [`vulkan::Api`]: Api
830/// [`Device::create_fence`]: crate::Device::create_fence
831/// [`TimelineSemaphore`]: Fence::TimelineSemaphore
832/// [`VK_KHR_timeline_semaphore`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#VK_KHR_timeline_semaphore
833/// [`FencePool`]: Fence::FencePool
834#[derive(Debug)]
835pub enum Fence {
836    /// A Vulkan [timeline semaphore].
837    ///
838    /// These are simpler to use than Vulkan fences, since timeline semaphores
839    /// work exactly the way [`wpgu_hal::Api::Fence`] is specified to work.
840    ///
841    /// [timeline semaphore]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-semaphores
842    /// [`wpgu_hal::Api::Fence`]: crate::Api::Fence
843    TimelineSemaphore(vk::Semaphore),
844
845    /// A collection of Vulkan [fence]s, each associated with a [`FenceValue`].
846    ///
847    /// The effective [`FenceValue`] of this variant is the greater of
848    /// `last_completed` and the maximum value associated with a signalled fence
849    /// in `active`.
850    ///
851    /// Fences are available in all versions of Vulkan, but since they only have
852    /// two states, "signaled" and "unsignaled", we need to use a separate fence
853    /// for each queue submission we might want to wait for, and remember which
854    /// [`FenceValue`] each one represents.
855    ///
856    /// [fence]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-fences
857    /// [`FenceValue`]: crate::FenceValue
858    FencePool {
859        last_completed: crate::FenceValue,
860        /// The pending fence values have to be ascending.
861        active: Vec<(crate::FenceValue, vk::Fence)>,
862        free: Vec<vk::Fence>,
863    },
864}
865
866impl Fence {
867    /// Return the highest [`FenceValue`] among the signalled fences in `active`.
868    ///
869    /// As an optimization, assume that we already know that the fence has
870    /// reached `last_completed`, and don't bother checking fences whose values
871    /// are less than that: those fences remain in the `active` array only
872    /// because we haven't called `maintain` yet to clean them up.
873    ///
874    /// [`FenceValue`]: crate::FenceValue
875    fn check_active(
876        device: &ash::Device,
877        mut last_completed: crate::FenceValue,
878        active: &[(crate::FenceValue, vk::Fence)],
879    ) -> Result<crate::FenceValue, crate::DeviceError> {
880        for &(value, raw) in active.iter() {
881            unsafe {
882                if value > last_completed && device.get_fence_status(raw)? {
883                    last_completed = value;
884                }
885            }
886        }
887        Ok(last_completed)
888    }
889
890    /// Return the highest signalled [`FenceValue`] for `self`.
891    ///
892    /// [`FenceValue`]: crate::FenceValue
893    fn get_latest(
894        &self,
895        device: &ash::Device,
896        extension: Option<&ExtensionFn<khr::timeline_semaphore::Device>>,
897    ) -> Result<crate::FenceValue, crate::DeviceError> {
898        match *self {
899            Self::TimelineSemaphore(raw) => unsafe {
900                Ok(match *extension.unwrap() {
901                    ExtensionFn::Extension(ref ext) => ext.get_semaphore_counter_value(raw)?,
902                    ExtensionFn::Promoted => device.get_semaphore_counter_value(raw)?,
903                })
904            },
905            Self::FencePool {
906                last_completed,
907                ref active,
908                free: _,
909            } => Self::check_active(device, last_completed, active),
910        }
911    }
912
913    /// Trim the internal state of this [`Fence`].
914    ///
915    /// This function has no externally visible effect, but you should call it
916    /// periodically to keep this fence's resource consumption under control.
917    ///
918    /// For fences using the [`FencePool`] implementation, this function
919    /// recycles fences that have been signaled. If you don't call this,
920    /// [`Queue::submit`] will just keep allocating a new Vulkan fence every
921    /// time it's called.
922    ///
923    /// [`FencePool`]: Fence::FencePool
924    /// [`Queue::submit`]: crate::Queue::submit
925    fn maintain(&mut self, device: &ash::Device) -> Result<(), crate::DeviceError> {
926        match *self {
927            Self::TimelineSemaphore(_) => {}
928            Self::FencePool {
929                ref mut last_completed,
930                ref mut active,
931                ref mut free,
932            } => {
933                let latest = Self::check_active(device, *last_completed, active)?;
934                let base_free = free.len();
935                for &(value, raw) in active.iter() {
936                    if value <= latest {
937                        free.push(raw);
938                    }
939                }
940                if free.len() != base_free {
941                    active.retain(|&(value, _)| value > latest);
942                    unsafe { device.reset_fences(&free[base_free..]) }?
943                }
944                *last_completed = latest;
945            }
946        }
947        Ok(())
948    }
949}

951impl crate::Queue for Queue {
952    type A = Api;
953
954    unsafe fn submit(
955        &self,
956        command_buffers: &[&CommandBuffer],
957        surface_textures: &[&SurfaceTexture],
958        (signal_fence, signal_value): (&mut Fence, crate::FenceValue),
959    ) -> Result<(), crate::DeviceError> {
960        let mut fence_raw = vk::Fence::null();
961
962        let mut wait_stage_masks = Vec::new();
963        let mut wait_semaphores = Vec::new();
964        let mut signal_semaphores = Vec::new();
965        let mut signal_values = Vec::new();
966
967        // Double check that the same swapchain image isn't being given to us multiple times,
968        // as that will deadlock when we try to lock them all.
969        debug_assert!(
970            {
971                let mut check = HashSet::with_capacity(surface_textures.len());
972                // We compare the Arcs by pointer, as Eq isn't well defined for SurfaceSemaphores.
973                for st in surface_textures {
974                    check.insert(Arc::as_ptr(&st.surface_semaphores));
975                }
976                check.len() == surface_textures.len()
977            },
978            "More than one surface texture is being used from the same swapchain. This will cause a deadlock in release."
979        );
980
981        let locked_swapchain_semaphores = surface_textures
982            .iter()
983            .map(|st| {
984                st.surface_semaphores
985                    .try_lock()
986                    .expect("Failed to lock surface semaphore.")
987            })
988            .collect::<Vec<_>>();
989
990        for mut swapchain_semaphore in locked_swapchain_semaphores {
991            swapchain_semaphore.set_used_fence_value(signal_value);
992
993            // If we're the first submission to operate on this image, wait on
994            // its acquire semaphore, to make sure the presentation engine is
995            // done with it.
996            if let Some(sem) = swapchain_semaphore.get_acquire_wait_semaphore() {
997                wait_stage_masks.push(vk::PipelineStageFlags::TOP_OF_PIPE);
998                wait_semaphores.push(sem);
999            }
1000
1001            // Get a semaphore to signal when we're done writing to this surface
1002            // image. Presentation of this image will wait for this.
1003            let signal_semaphore = swapchain_semaphore.get_submit_signal_semaphore(&self.device)?;
1004            signal_semaphores.push(signal_semaphore);
1005            signal_values.push(!0);
1006        }
1007
1008        // In order for submissions to be strictly ordered, we encode a dependency between each submission
1009        // using a pair of semaphores. This adds a wait if it is needed, and signals the next semaphore.
1010        let semaphore_state = self.relay_semaphores.lock().advance(&self.device)?;
1011
1012        if let Some(sem) = semaphore_state.wait {
1013            wait_stage_masks.push(vk::PipelineStageFlags::TOP_OF_PIPE);
1014            wait_semaphores.push(sem);
1015        }
1016
1017        signal_semaphores.push(semaphore_state.signal);
1018        signal_values.push(!0);
1019
1020        // We need to signal our wgpu::Fence if we have one, this adds it to the signal list.
1021        signal_fence.maintain(&self.device.raw)?;
1022        match *signal_fence {
1023            Fence::TimelineSemaphore(raw) => {
1024                signal_semaphores.push(raw);
1025                signal_values.push(signal_value);
1026            }
1027            Fence::FencePool {
1028                ref mut active,
1029                ref mut free,
1030                ..
1031            } => {
1032                fence_raw = match free.pop() {
1033                    Some(raw) => raw,
1034                    None => unsafe {
1035                        self.device
1036                            .raw
1037                            .create_fence(&vk::FenceCreateInfo::default(), None)?
1038                    },
1039                };
1040                active.push((signal_value, fence_raw));
1041            }
1042        }
1043
1044        let vk_cmd_buffers = command_buffers
1045            .iter()
1046            .map(|cmd| cmd.raw)
1047            .collect::<Vec<_>>();
1048
1049        let mut vk_info = vk::SubmitInfo::default().command_buffers(&vk_cmd_buffers);
1050
1051        vk_info = vk_info
1052            .wait_semaphores(&wait_semaphores)
1053            .wait_dst_stage_mask(&wait_stage_masks)
1054            .signal_semaphores(&signal_semaphores);
1055
1056        let mut vk_timeline_info;
1057
1058        if self.device.private_caps.timeline_semaphores {
1059            vk_timeline_info =
1060                vk::TimelineSemaphoreSubmitInfo::default().signal_semaphore_values(&signal_values);
1061            vk_info = vk_info.push_next(&mut vk_timeline_info);
1062        }
1063
1064        profiling::scope!("vkQueueSubmit");
1065        unsafe {
1066            self.device
1067                .raw
1068                .queue_submit(self.raw, &[vk_info], fence_raw)?
1069        };
1070        Ok(())
1071    }
1072
1073    unsafe fn present(
1074        &self,
1075        surface: &Surface,
1076        texture: SurfaceTexture,
1077    ) -> Result<(), crate::SurfaceError> {
1078        let mut swapchain = surface.swapchain.write();
1079        let ssc = swapchain.as_mut().unwrap();
1080        let mut swapchain_semaphores = texture.surface_semaphores.lock();
1081
1082        let swapchains = [ssc.raw];
1083        let image_indices = [texture.index];
1084        let vk_info = vk::PresentInfoKHR::default()
1085            .swapchains(&swapchains)
1086            .image_indices(&image_indices)
1087            .wait_semaphores(swapchain_semaphores.get_present_wait_semaphores());
1088
1089        let suboptimal = {
1090            profiling::scope!("vkQueuePresentKHR");
1091            unsafe { self.swapchain_fn.queue_present(self.raw, &vk_info) }.map_err(|error| {
1092                match error {
1093                    vk::Result::ERROR_OUT_OF_DATE_KHR => crate::SurfaceError::Outdated,
1094                    vk::Result::ERROR_SURFACE_LOST_KHR => crate::SurfaceError::Lost,
1095                    _ => crate::DeviceError::from(error).into(),
1096                }
1097            })?
1098        };
1099        if suboptimal {
1100            // We treat `VK_SUBOPTIMAL_KHR` as `VK_SUCCESS` on Android.
1101            // On Android 10+, libvulkan's `vkQueuePresentKHR` implementation returns `VK_SUBOPTIMAL_KHR` if not doing pre-rotation
1102            // (i.e `VkSwapchainCreateInfoKHR::preTransform` not being equal to the current device orientation).
1103            // This is always the case when the device orientation is anything other than the identity one, as we unconditionally use `VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR`.
1104            #[cfg(not(target_os = "android"))]
1105            log::warn!("Suboptimal present of frame {}", texture.index);
1106        }
1107        Ok(())
1108    }

1115impl From<vk::Result> for crate::DeviceError {
1116    fn from(result: vk::Result) -> Self {
1117        #![allow(unreachable_code)]
1118        match result {
1119            vk::Result::ERROR_OUT_OF_HOST_MEMORY | vk::Result::ERROR_OUT_OF_DEVICE_MEMORY => {
1120                #[cfg(feature = "oom_panic")]
1121                panic!("Out of memory ({result:?})");
1122
1123                Self::OutOfMemory
1124            }
1125            vk::Result::ERROR_DEVICE_LOST => {
1126                #[cfg(feature = "device_lost_panic")]
1127                panic!("Device lost");
1128
1129                Self::Lost
1130            }
1131            _ => {
1132                #[cfg(feature = "internal_error_panic")]
1133                panic!("Internal error: {result:?}");
1134
1135                log::warn!("Unrecognized device error {result:?}");
1136                Self::Lost
1137            }
1138        }
1139    }
