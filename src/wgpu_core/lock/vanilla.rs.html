1//! Plain, uninstrumented wrappers around [`parking_lot`] lock types.
2//!
3//! These definitions are used when no particular lock instrumentation
4//! Cargo feature is selected.
5
6/// A plain wrapper around [`parking_lot::Mutex`].
7///
8/// This is just like [`parking_lot::Mutex`], except that our [`new`]
9/// method takes a rank, indicating where the new mutex should sit in
10/// `wgpu-core`'s lock ordering. The rank is ignored.
11///
12/// See the [`lock`] module documentation for other wrappers.
13///
14/// [`new`]: Mutex::new
15/// [`lock`]: crate::lock
16pub struct Mutex<T>(parking_lot::Mutex<T>);
17
18/// A guard produced by locking [`Mutex`].
19///
20/// This is just a wrapper around a [`parking_lot::MutexGuard`].
21pub struct MutexGuard<'a, T>(parking_lot::MutexGuard<'a, T>);
22
23impl<T> Mutex<T> {
24    pub fn new(_rank: super::rank::LockRank, value: T) -> Mutex<T> {
25        Mutex(parking_lot::Mutex::new(value))
26    }
27
28    pub fn lock(&self) -> MutexGuard<T> {
29        MutexGuard(self.0.lock())
30    }
31}
32
33impl<'a, T> std::ops::Deref for MutexGuard<'a, T> {
34    type Target = T;
35
36    fn deref(&self) -> &Self::Target {
37        self.0.deref()
38    }
39}
40
41impl<'a, T> std::ops::DerefMut for MutexGuard<'a, T> {
42    fn deref_mut(&mut self) -> &mut Self::Target {
43        self.0.deref_mut()
44    }
45}
46
47impl<T: std::fmt::Debug> std::fmt::Debug for Mutex<T> {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        self.0.fmt(f)
50    }
51}
