112trait LabelHelpers<'a> {
113    fn to_hal(&'a self, flags: wgt::InstanceFlags) -> Option<&'a str>;
114    fn to_string(&self) -> String;
115}
116impl<'a> LabelHelpers<'a> for Label<'a> {
117    fn to_hal(&'a self, flags: wgt::InstanceFlags) -> Option<&'a str> {
118        if flags.contains(wgt::InstanceFlags::DISCARD_HAL_LABELS) {
119            return None;
120        }
121
122        self.as_ref().map(|cow| cow.as_ref())
123    }
124    fn to_string(&self) -> String {
125        self.as_ref().map(|cow| cow.to_string()).unwrap_or_default()
126    }
127}

131        return None;
132    }
133
134    opt

148// #[cfg] attributes in exported macros are interesting!
149//
150// The #[cfg] conditions in a macro's expansion are evaluated using the
151// configuration options (features, target architecture and os, etc.) in force
152// where the macro is *used*, not where it is *defined*. That is, if crate A
153// defines a macro like this:
154//
155//     #[macro_export]
156//     macro_rules! if_bleep {
157//         { } => {
158//             #[cfg(feature = "bleep")]
159//             bleep();
160//         }
161//     }
162//
163// and then crate B uses it like this:
164//
165//     fn f() {
166//         if_bleep! { }
167//     }
168//
169// then it is crate B's `"bleep"` feature, not crate A's, that determines
170// whether the macro expands to a function call or an empty statement. The
171// entire configuration predicate is evaluated in the use's context, not the
172// definition's.
173//
174// Since `wgpu-core` selects back ends using features, we need to make sure the
175// arms of the `gfx_select!` macro are pruned according to `wgpu-core`'s
176// features, not those of whatever crate happens to be using `gfx_select!`. This
177// means we can't use `#[cfg]` attributes in `gfx_select!`s definition itself.
178// Instead, for each backend, `gfx_select!` must use a macro whose definition is
179// selected by `#[cfg]` in `wgpu-core`. The configuration predicate is still
180// evaluated when the macro is used; we've just moved the `#[cfg]` into a macro
181// used by `wgpu-core` itself.
182
183/// Define an exported macro named `$public` that expands to an expression if
184/// the feature `$feature` is enabled, or to a panic otherwise.
185///
186/// This is used in the definition of `gfx_select!`, to dispatch the
187/// call to the appropriate backend, but panic if that backend was not
188/// compiled in.
189///
190/// For a call like this:
191///
192/// ```ignore
193/// define_backend_caller! { name, private, "feature" if cfg_condition }
194/// ```
195///
196/// define a macro `name`, used like this:
197///
198/// ```ignore
199/// name!(expr)
200/// ```
201///
202/// that expands to `expr` if `#[cfg(cfg_condition)]` is enabled, or a
203/// panic otherwise. The panic message complains that `"feature"` is
204/// not enabled.
205///
206/// Because of odd technical limitations on exporting macros expanded
207/// by other macros, you must supply both a public-facing name for the
208/// macro and a private name, `$private`, which is never used
209/// outside this macro. For details:
210/// <https://github.com/rust-lang/rust/pull/52234#issuecomment-976702997>
211macro_rules! define_backend_caller {
212    { $public:ident, $private:ident, $feature:literal if $cfg:meta } => {
213        #[cfg($cfg)]
214        #[macro_export]
215        macro_rules! $private {
216            ( $call:expr ) => ( $call )
217        }
218
219        #[cfg(not($cfg))]
220        #[macro_export]
221        macro_rules! $private {
222            ( $call:expr ) => (
223                panic!("Identifier refers to disabled backend feature {:?}", $feature)
224            )
225        }
226
227        // See note about rust-lang#52234 above.
228        #[doc(hidden)] pub use $private as $public;
229    }
230}

232// Define a macro for each `gfx_select!` match arm. For example,
233//
234//     gfx_if_vulkan!(expr)
235//
236// expands to `expr` if the `"vulkan"` feature is enabled, or to a panic
237// otherwise.
238define_backend_caller! { gfx_if_vulkan, gfx_if_vulkan_hidden, "vulkan" if all(feature = "vulkan", not(target_arch = "wasm32")) }
239define_backend_caller! { gfx_if_metal, gfx_if_metal_hidden, "metal" if all(feature = "metal", any(target_os = "macos", target_os = "ios")) }
240define_backend_caller! { gfx_if_dx12, gfx_if_dx12_hidden, "dx12" if all(feature = "dx12", windows) }
241define_backend_caller! { gfx_if_gles, gfx_if_gles_hidden, "gles" if feature = "gles" }
242define_backend_caller! { gfx_if_empty, gfx_if_empty_hidden, "empty" if all(
243    not(any(feature = "metal", feature = "vulkan", feature = "gles")),
244    any(target_os = "macos", target_os = "ios"),
245) }

247/// Dispatch on an [`Id`]'s backend to a backend-generic method.
248///
249/// Uses of this macro have the form:
250///
251/// ```ignore
252///
253///     gfx_select!(id => value.method(args...))
254///
255/// ```
256///
257/// This expands to an expression that calls `value.method::<A>(args...)` for
258/// the backend `A` selected by `id`. The expansion matches on `id.backend()`,
259/// with an arm for each backend type in [`wgpu_types::Backend`] which calls the
260/// specialization of `method` for the given backend. This allows resource
261/// identifiers to select backends dynamically, even though many `wgpu_core`
262/// methods are compiled and optimized for a specific back end.
263///
264/// This macro is typically used to call methods on [`wgpu_core::global::Global`],
265/// many of which take a single `hal::Api` type parameter. For example, to
266/// create a new buffer on the device indicated by `device_id`, one would say:
267///
268/// ```ignore
269/// gfx_select!(device_id => global.device_create_buffer(device_id, ...))
270/// ```
271///
272/// where the `device_create_buffer` method is defined like this:
273///
274/// ```ignore
275/// impl Global {
276///    pub fn device_create_buffer<A: HalApi>(&self, ...) -> ...
277///    { ... }
278/// }
279/// ```
280///
281/// That `gfx_select!` call uses `device_id`'s backend to select the right
282/// backend type `A` for a call to `Global::device_create_buffer<A>`.
283///
284/// However, there's nothing about this macro that is specific to `hub::Global`.
285/// For example, Firefox's embedding of `wgpu_core` defines its own types with
286/// methods that take `hal::Api` type parameters. Firefox uses `gfx_select!` to
287/// dynamically dispatch to the right specialization based on the resource's id.
288///
289/// [`wgpu_types::Backend`]: wgt::Backend
290/// [`wgpu_core::global::Global`]: crate::global::Global
291/// [`Id`]: id::Id
292#[macro_export]
293macro_rules! gfx_select {
294    // Simple two-component expression, like `self.0.method(..)`.
295    ($id:expr => $c0:ident.$c1:tt.$method:ident $params:tt) => {
296        $crate::gfx_select!($id => {$c0.$c1}, $method $params)
297    };
298
299    // Simple identifier-only expression, like `global.method(..)`.
300    ($id:expr => $c0:ident.$method:ident $params:tt) => {
301        $crate::gfx_select!($id => {$c0}, $method $params)
302    };
303
304    ($id:expr => {$($c:tt)*}, $method:ident $params:tt) => {
305        match $id.backend() {
306            wgt::Backend::Vulkan => $crate::gfx_if_vulkan!($($c)*.$method::<$crate::api::Vulkan> $params),
307            wgt::Backend::Metal => $crate::gfx_if_metal!($($c)*.$method::<$crate::api::Metal> $params),
308            wgt::Backend::Dx12 => $crate::gfx_if_dx12!($($c)*.$method::<$crate::api::Dx12> $params),
309            wgt::Backend::Gl => $crate::gfx_if_gles!($($c)*.$method::<$crate::api::Gles> $params),
310            wgt::Backend::Empty => $crate::gfx_if_empty!($($c)*.$method::<$crate::api::Empty> $params),
311            other => panic!("Unexpected backend {:?}", other),
312        }
313    };
314}

336#[inline]
337pub(crate) fn get_lowest_common_denom(a: u32, b: u32) -> u32 {
338    let gcd = if a >= b {
339        get_greatest_common_divisor(a, b)
340    } else {
341        get_greatest_common_divisor(b, a)
342    };
343    a * b / gcd
344}
345
346#[inline]
347pub(crate) fn get_greatest_common_divisor(mut a: u32, mut b: u32) -> u32 {
348    assert!(a >= b);
349    loop {
350        let c = a % b;
351        if c == 0 {
352            return b;
353        } else {
354            a = b;
355            b = c;
356        }
357    }
358}
