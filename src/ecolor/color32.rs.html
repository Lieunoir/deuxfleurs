3/// This format is used for space-efficient color representation (32 bits).
4///
5/// Instead of manipulating this directly it is often better
6/// to first convert it to either [`Rgba`] or [`crate::Hsva`].
7///
8/// Internally this uses 0-255 gamma space `sRGBA` color with premultiplied alpha.
9/// Alpha channel is in linear space.
10///
11/// The special value of alpha=0 means the color is to be treated as an additive color.
12#[repr(C)]
13#[derive(Clone, Copy, Default, Eq, Hash, PartialEq)]
14#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
15#[cfg_attr(feature = "bytemuck", derive(bytemuck::Pod, bytemuck::Zeroable))]
16pub struct Color32(pub(crate) [u8; 4]);
17
18impl std::fmt::Debug for Color32 {
19    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
20        let [r, g, b, a] = self.0;
21        write!(f, "#{r:02X}_{g:02X}_{b:02X}_{a:02X}")
22    }
23}
24

31    }
32}

101    /// From `sRGBA` WITHOUT premultiplied alpha.
102    #[inline]
103    pub fn from_rgba_unmultiplied(r: u8, g: u8, b: u8, a: u8) -> Self {
104        use std::sync::OnceLock;
105        match a {
106            // common-case optimization
107            0 => Self::TRANSPARENT,
108            // common-case optimization
109            255 => Self::from_rgb(r, g, b),
110            a => {
111                static LOOKUP_TABLE: OnceLock<[u8; 256 * 256]> = OnceLock::new();
112                let lut = LOOKUP_TABLE.get_or_init(|| {
113                    use crate::{gamma_u8_from_linear_f32, linear_f32_from_gamma_u8};
114                    core::array::from_fn(|i| {
115                        let [value, alpha] = (i as u16).to_ne_bytes();
116                        let value_lin = linear_f32_from_gamma_u8(value);
117                        let alpha_lin = linear_f32_from_linear_u8(alpha);
118                        gamma_u8_from_linear_f32(value_lin * alpha_lin)
119                    })
120                });
121
122                let [r, g, b] =
123                    [r, g, b].map(|value| lut[usize::from(u16::from_ne_bytes([value, a]))]);
124                Self::from_rgba_premultiplied(r, g, b, a)
125            }
126        }
127    }

210    /// Multiply with 0.5 to make color half as opaque, perceptually.
211    ///
212    /// Fast multiplication in gamma-space.
213    ///
214    /// This is perceptually even, and faster that [`Self::linear_multiply`].
215    #[inline]
216    pub fn gamma_multiply(self, factor: f32) -> Self {
217        debug_assert!(0.0 <= factor && factor.is_finite());
218        let Self([r, g, b, a]) = self;
219        Self([
220            (r as f32 * factor + 0.5) as u8,
221            (g as f32 * factor + 0.5) as u8,
222            (b as f32 * factor + 0.5) as u8,
223            (a as f32 * factor + 0.5) as u8,
224        ])
225    }
226
227    /// Multiply with 0.5 to make color half as opaque in linear space.
228    ///
229    /// This is using linear space, which is not perceptually even.
230    /// You likely want to use [`Self::gamma_multiply`] instead.
231    #[inline]
232    pub fn linear_multiply(self, factor: f32) -> Self {
233        debug_assert!(0.0 <= factor && factor.is_finite());
234        // As an unfortunate side-effect of using premultiplied alpha
235        // we need a somewhat expensive conversion to linear space and back.
236        Rgba::from(self).multiply(factor).into()
237    }

239    /// Converts to floating point values in the range 0-1 without any gamma space conversion.
240    ///
241    /// Use this with great care! In almost all cases, you want to convert to [`crate::Rgba`] instead
242    /// in order to obtain linear space color values.
243    #[inline]
244    pub fn to_normalized_gamma_f32(self) -> [f32; 4] {
245        let Self([r, g, b, a]) = self;
246        [
247            r as f32 / 255.0,
248            g as f32 / 255.0,
249            b as f32 / 255.0,
250            a as f32 / 255.0,
251        ]
252    }

254    /// Lerp this color towards `other` by `t` in gamma space.
255    pub fn lerp_to_gamma(&self, other: Self, t: f32) -> Self {
256        use emath::lerp;
257
258        Self::from_rgba_premultiplied(
259            fast_round(lerp((self[0] as f32)..=(other[0] as f32), t)),
260            fast_round(lerp((self[1] as f32)..=(other[1] as f32), t)),
261            fast_round(lerp((self[2] as f32)..=(other[2] as f32), t)),
262            fast_round(lerp((self[3] as f32)..=(other[3] as f32), t)),
263        )
264    }
