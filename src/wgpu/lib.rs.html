193/// The mapped portion of a buffer, if any, and its outstanding views.
194///
195/// This ensures that views fall within the mapped range and don't overlap, and
196/// also takes care of turning `Option<BufferSize>` sizes into actual buffer
197/// offsets.
198#[derive(Debug)]
199struct MapContext {
200    /// The overall size of the buffer.
201    ///
202    /// This is just a convenient copy of [`Buffer::size`].
203    total_size: BufferAddress,
204
205    /// The range of the buffer that is mapped.
206    ///
207    /// This is `0..0` if the buffer is not mapped. This becomes non-empty when
208    /// the buffer is mapped at creation time, and when you call `map_async` on
209    /// some [`BufferSlice`] (so technically, it indicates the portion that is
210    /// *or has been requested to be* mapped.)
211    ///
212    /// All [`BufferView`]s and [`BufferViewMut`]s must fall within this range.
213    initial_range: Range<BufferAddress>,
214
215    /// The ranges covered by all outstanding [`BufferView`]s and
216    /// [`BufferViewMut`]s. These are non-overlapping, and are all contained
217    /// within `initial_range`.
218    sub_ranges: Vec<Range<BufferAddress>>,
219}
220
221impl MapContext {
222    fn new(total_size: BufferAddress) -> Self {
223        Self {
224            total_size,
225            initial_range: 0..0,
226            sub_ranges: Vec::new(),
227        }
228    }
229
230    /// Record that the buffer is no longer mapped.
231    fn reset(&mut self) {
232        self.initial_range = 0..0;
233
234        assert!(
235            self.sub_ranges.is_empty(),
236            "You cannot unmap a buffer that still has accessible mapped views"
237        );
238    }
239
240    /// Record that the `size` bytes of the buffer at `offset` are now viewed.
241    ///
242    /// Return the byte offset within the buffer of the end of the viewed range.
243    ///
244    /// # Panics
245    ///
246    /// This panics if the given range overlaps with any existing range.
247    fn add(&mut self, offset: BufferAddress, size: Option<BufferSize>) -> BufferAddress {
248        let end = match size {
249            Some(s) => offset + s.get(),
250            None => self.initial_range.end,
251        };
252        assert!(self.initial_range.start <= offset && end <= self.initial_range.end);
253        // This check is essential for avoiding undefined behavior: it is the
254        // only thing that ensures that `&mut` references to the buffer's
255        // contents don't alias anything else.
256        for sub in self.sub_ranges.iter() {
257            assert!(
258                end <= sub.start || offset >= sub.end,
259                "Intersecting map range with {sub:?}"
260            );
261        }
262        self.sub_ranges.push(offset..end);
263        end
264    }
265
266    /// Record that the `size` bytes of the buffer at `offset` are no longer viewed.
267    ///
268    /// # Panics
269    ///
270    /// This panics if the given range does not exactly match one previously
271    /// passed to [`add`].
272    ///
273    /// [`add]`: MapContext::add
274    fn remove(&mut self, offset: BufferAddress, size: Option<BufferSize>) {
275        let end = match size {
276            Some(s) => offset + s.get(),
277            None => self.initial_range.end,
278        };
279
280        let index = self
281            .sub_ranges
282            .iter()
283            .position(|r| *r == (offset..end))
284            .expect("unable to remove range from map context");
285        self.sub_ranges.swap_remove(index);
286    }
287}

289/// Handle to a GPU-accessible buffer.
290///
291/// Created with [`Device::create_buffer`] or
292/// [`DeviceExt::create_buffer_init`](util::DeviceExt::create_buffer_init).
293///
294/// Corresponds to [WebGPU `GPUBuffer`](https://gpuweb.github.io/gpuweb/#buffer-interface).
295///
296/// A `Buffer`'s bytes have "interior mutability": functions like
297/// [`Queue::write_buffer`] or [mapping] a buffer for writing only require a
298/// `&Buffer`, not a `&mut Buffer`, even though they modify its contents. `wgpu`
299/// prevents simultaneous reads and writes of buffer contents using run-time
300/// checks.
301///
302/// [mapping]: Buffer#mapping-buffers
303///
304/// # Mapping buffers
305///
306/// If a `Buffer` is created with the appropriate [`usage`], it can be *mapped*:
307/// you can make its contents accessible to the CPU as an ordinary `&[u8]` or
308/// `&mut [u8]` slice of bytes. Buffers created with the
309/// [`mapped_at_creation`][mac] flag set are also mapped initially.
310///
311/// Depending on the hardware, the buffer could be memory shared between CPU and
312/// GPU, so that the CPU has direct access to the same bytes the GPU will
313/// consult; or it may be ordinary CPU memory, whose contents the system must
314/// copy to/from the GPU as needed. This crate's API is designed to work the
315/// same way in either case: at any given time, a buffer is either mapped and
316/// available to the CPU, or unmapped and ready for use by the GPU, but never
317/// both. This makes it impossible for either side to observe changes by the
318/// other immediately, and any necessary transfers can be carried out when the
319/// buffer transitions from one state to the other.
320///
321/// There are two ways to map a buffer:
322///
323/// - If [`BufferDescriptor::mapped_at_creation`] is `true`, then the entire
324///   buffer is mapped when it is created. This is the easiest way to initialize
325///   a new buffer. You can set `mapped_at_creation` on any kind of buffer,
326///   regardless of its [`usage`] flags.
327///
328/// - If the buffer's [`usage`] includes the [`MAP_READ`] or [`MAP_WRITE`]
329///   flags, then you can call `buffer.slice(range).map_async(mode, callback)`
330///   to map the portion of `buffer` given by `range`. This waits for the GPU to
331///   finish using the buffer, and invokes `callback` as soon as the buffer is
332///   safe for the CPU to access.
333///
334/// Once a buffer is mapped:
335///
336/// - You can call `buffer.slice(range).get_mapped_range()` to obtain a
337///   [`BufferView`], which dereferences to a `&[u8]` that you can use to read
338///   the buffer's contents.
339///
340/// - Or, you can call `buffer.slice(range).get_mapped_range_mut()` to obtain a
341///   [`BufferViewMut`], which dereferences to a `&mut [u8]` that you can use to
342///   read and write the buffer's contents.
343///
344/// The given `range` must fall within the mapped portion of the buffer. If you
345/// attempt to access overlapping ranges, even for shared access only, these
346/// methods panic.
347///
348/// While a buffer is mapped, you may not submit any commands to the GPU that
349/// access it. You may record command buffers that use the buffer, but if you
350/// submit them while the buffer is mapped, submission will panic.
351///
352/// When you are done using the buffer on the CPU, you must call
353/// [`Buffer::unmap`] to make it available for use by the GPU again. All
354/// [`BufferView`] and [`BufferViewMut`] views referring to the buffer must be
355/// dropped before you unmap it; otherwise, [`Buffer::unmap`] will panic.
356///
357/// # Example
358///
359/// If `buffer` was created with [`BufferUsages::MAP_WRITE`], we could fill it
360/// with `f32` values like this:
361///
362/// ```no_run
363/// # mod bytemuck {
364/// #     pub fn cast_slice_mut(bytes: &mut [u8]) -> &mut [f32] { todo!() }
365/// # }
366/// # let device: wgpu::Device = todo!();
367/// # let buffer: wgpu::Buffer = todo!();
368/// let buffer = std::sync::Arc::new(buffer);
369/// let capturable = buffer.clone();
370/// buffer.slice(..).map_async(wgpu::MapMode::Write, move |result| {
371///     if result.is_ok() {
372///         let mut view = capturable.slice(..).get_mapped_range_mut();
373///         let floats: &mut [f32] = bytemuck::cast_slice_mut(&mut view);
374///         floats.fill(42.0);
375///         drop(view);
376///         capturable.unmap();
377///     }
378/// });
379/// ```
380///
381/// This code takes the following steps:
382///
383/// - First, it moves `buffer` into an [`Arc`], and makes a clone for capture by
384///   the callback passed to [`map_async`]. Since a [`map_async`] callback may be
385///   invoked from another thread, interaction between the callback and the
386///   thread calling [`map_async`] generally requires some sort of shared heap
387///   data like this. In real code, the [`Arc`] would probably own some larger
388///   structure that itself owns `buffer`.
389///
390/// - Then, it calls [`Buffer::slice`] to make a [`BufferSlice`] referring to
391///   the buffer's entire contents.
392///
393/// - Next, it calls [`BufferSlice::map_async`] to request that the bytes to
394///   which the slice refers be made accessible to the CPU ("mapped"). This may
395///   entail waiting for previously enqueued operations on `buffer` to finish.
396///   Although [`map_async`] itself always returns immediately, it saves the
397///   callback function to be invoked later.
398///
399/// - When some later call to [`Device::poll`] or [`Instance::poll_all`] (not
400///   shown in this example) determines that the buffer is mapped and ready for
401///   the CPU to use, it invokes the callback function.
402///
403/// - The callback function calls [`Buffer::slice`] and then
404///   [`BufferSlice::get_mapped_range_mut`] to obtain a [`BufferViewMut`], which
405///   dereferences to a `&mut [u8]` slice referring to the buffer's bytes.
406///
407/// - It then uses the [`bytemuck`] crate to turn the `&mut [u8]` into a `&mut
408///   [f32]`, and calls the slice [`fill`] method to fill the buffer with a
409///   useful value.
410///
411/// - Finally, the callback drops the view and calls [`Buffer::unmap`] to unmap
412///   the buffer. In real code, the callback would also need to do some sort of
413///   synchronization to let the rest of the program know that it has completed
414///   its work.
415///
416/// If using [`map_async`] directly is awkward, you may find it more convenient to
417/// use [`Queue::write_buffer`] and [`util::DownloadBuffer::read_buffer`].
418/// However, those each have their own tradeoffs; the asynchronous nature of GPU
419/// execution makes it hard to avoid friction altogether.
420///
421/// [`Arc`]: std::sync::Arc
422/// [`map_async`]: BufferSlice::map_async
423/// [`bytemuck`]: https://crates.io/crates/bytemuck
424/// [`fill`]: slice::fill
425///
426/// ## Mapping buffers on the web
427///
428/// When compiled to WebAssembly and running in a browser content process,
429/// `wgpu` implements its API in terms of the browser's WebGPU implementation.
430/// In this context, `wgpu` is further isolated from the GPU:
431///
432/// - Depending on the browser's WebGPU implementation, mapping and unmapping
433///   buffers probably entails copies between WebAssembly linear memory and the
434///   graphics driver's buffers.
435///
436/// - All modern web browsers isolate web content in its own sandboxed process,
437///   which can only interact with the GPU via interprocess communication (IPC).
438///   Although most browsers' IPC systems use shared memory for large data
439///   transfers, there will still probably need to be copies into and out of the
440///   shared memory buffers.
441///
442/// All of these copies contribute to the cost of buffer mapping in this
443/// configuration.
444///
445/// [`usage`]: BufferDescriptor::usage
446/// [mac]: BufferDescriptor::mapped_at_creation
447/// [`MAP_READ`]: BufferUsages::MAP_READ
448/// [`MAP_WRITE`]: BufferUsages::MAP_WRITE
449#[derive(Debug)]
450pub struct Buffer {
451    context: Arc<C>,
452    id: ObjectId,
453    data: Box<Data>,
454    map_context: Mutex<MapContext>,
455    size: wgt::BufferAddress,
456    usage: BufferUsages,
457    // Todo: missing map_state https://www.w3.org/TR/webgpu/#dom-gpubuffer-mapstate
458}

460static_assertions::assert_impl_all!(Buffer: Send, Sync);

462/// A slice of a [`Buffer`], to be mapped, used for vertex or index data, or the like.
463///
464/// You can create a `BufferSlice` by calling [`Buffer::slice`]:
465///
466/// ```no_run
467/// # let buffer: wgpu::Buffer = todo!();
468/// let slice = buffer.slice(10..20);
469/// ```
470///
471/// This returns a slice referring to the second ten bytes of `buffer`. To get a
472/// slice of the entire `Buffer`:
473///
474/// ```no_run
475/// # let buffer: wgpu::Buffer = todo!();
476/// let whole_buffer_slice = buffer.slice(..);
477/// ```
478///
479/// You can pass buffer slices to methods like [`RenderPass::set_vertex_buffer`]
480/// and [`RenderPass::set_index_buffer`] to indicate which portion of the buffer
481/// a draw call should consult.
482///
483/// To access the slice's contents on the CPU, you must first [map] the buffer,
484/// and then call [`BufferSlice::get_mapped_range`] or
485/// [`BufferSlice::get_mapped_range_mut`] to obtain a view of the slice's
486/// contents. See the documentation on [mapping][map] for more details,
487/// including example code.
488///
489/// Unlike a Rust shared slice `&[T]`, whose existence guarantees that
490/// nobody else is modifying the `T` values to which it refers, a
491/// [`BufferSlice`] doesn't guarantee that the buffer's contents aren't
492/// changing. You can still record and submit commands operating on the
493/// buffer while holding a [`BufferSlice`]. A [`BufferSlice`] simply
494/// represents a certain range of the buffer's bytes.
495///
496/// The `BufferSlice` type is unique to the Rust API of `wgpu`. In the WebGPU
497/// specification, an offset and size are specified as arguments to each call
498/// working with the [`Buffer`], instead.
499///
500/// [map]: Buffer#mapping-buffers
501#[derive(Copy, Clone, Debug)]
502pub struct BufferSlice<'a> {
503    buffer: &'a Buffer,
504    offset: BufferAddress,
505    size: Option<BufferSize>,
506}
507#[cfg(send_sync)]
508static_assertions::assert_impl_all!(BufferSlice<'_>: Send, Sync);

1163/// Handle to a pipeline cache, which is used to accelerate
1164/// creating [`RenderPipeline`]s and [`ComputePipeline`]s
1165/// in subsequent executions
1166///
1167/// This reuse is only applicable for the same or similar devices.
1168/// See [`util::pipeline_cache_key`] for some details.
1169///
1170/// # Background
1171///
1172/// In most GPU drivers, shader code must be converted into a machine code
1173/// which can be executed on the GPU.
1174/// Generating this machine code can require a lot of computation.
1175/// Pipeline caches allow this computation to be reused between executions
1176/// of the program.
1177/// This can be very useful for reducing program startup time.
1178///
1179/// Note that most desktop GPU drivers will manage their own caches,
1180/// meaning that little advantage can be gained from this on those platforms.
1181/// However, on some platforms, especially Android, drivers leave this to the
1182/// application to implement.
1183///
1184/// Unfortunately, drivers do not expose whether they manage their own caches.
1185/// Some reasonable policies for applications to use are:
1186/// - Manage their own pipeline cache on all platforms
1187/// - Only manage pipeline caches on Android
1188///
1189/// # Usage
1190///
1191/// It is valid to use this resource when creating multiple pipelines, in
1192/// which case it will likely cache each of those pipelines.
1193/// It is also valid to create a new cache for each pipeline.
1194///
1195/// This resource is most useful when the data produced from it (using
1196/// [`PipelineCache::get_data`]) is persisted.
1197/// Care should be taken that pipeline caches are only used for the same device,
1198/// as pipeline caches from compatible devices are unlikely to provide any advantage.
1199/// `util::pipeline_cache_key` can be used as a file/directory name to help ensure that.
1200///
1201/// It is recommended to store pipeline caches atomically. If persisting to disk,
1202/// this can usually be achieved by creating a temporary file, then moving/[renaming]
1203/// the temporary file over the existing cache
1204///
1205/// # Storage Usage
1206///
1207/// There is not currently an API available to reduce the size of a cache.
1208/// This is due to limitations in the underlying graphics APIs used.
1209/// This is especially impactful if your application is being updated, so
1210/// previous caches are no longer being used.
1211///
1212/// One option to work around this is to regenerate the cache.
1213/// That is, creating the pipelines which your program runs using
1214/// with the stored cached data, then recreating the *same* pipelines
1215/// using a new cache, which your application then store.
1216///
1217/// # Implementations
1218///
1219/// This resource currently only works on the following backends:
1220///  - Vulkan
1221///
1222/// This type is unique to the Rust API of `wgpu`.
1223///
1224/// [renaming]: std::fs::rename
1225#[derive(Debug)]
1226pub struct PipelineCache {
1227    context: Arc<C>,
1228    id: ObjectId,
1229    data: Box<Data>,
1230}
1231

1233static_assertions::assert_impl_all!(PipelineCache: Send, Sync);
1234

1236    /// Get the data associated with this pipeline cache.
1237    /// The data format is an implementation detail of `wgpu`.
1238    /// The only defined operation on this data setting it as the `data` field
1239    /// on [`PipelineCacheDescriptor`], then to [`Device::create_pipeline_cache`].
1240    ///
1241    /// This function is unique to the Rust API of `wgpu`.
1242    pub fn get_data(&self) -> Option<Vec<u8>> {
1243        self.context
1244            .pipeline_cache_get_data(&self.id, self.data.as_ref())
1245    }
1246}

1564    Load,

2321    ///

2323    ///   If it is set and WebGPU support is detected, this instance will *only* be able to create

2325    ///   disable the `webgpu` compile-time feature or do add the [`Backends::BROWSER_WEBGPU`]
2326    ///   flag to the the `instance_desc`'s `backends` field.

2505        &self,
2506        target: impl Into<SurfaceTarget<'window>>,
2507    ) -> Result<Surface<'window>, CreateSurfaceError> {

2509        let handle_source;
2510
2511        let target = target.into();
2512        let mut surface = match target {
2513            SurfaceTarget::Window(window) => unsafe {

2516                        inner: CreateSurfaceErrorKind::RawHandle(e),
2517                    })?,
2518                );
2519                handle_source = Some(window);
2520
2521                surface
2522            }?,
2523

2533                // Note that we need to call this while we still have `value` around.
2534                // This is safe without storing canvas to `handle_origin` since the surface will create a copy internally.
2535                unsafe {
2536                    self.create_surface_unsafe(SurfaceTargetUnsafe::RawHandle {
2537                        raw_display_handle,
2538                        raw_window_handle,
2539                    })
2540                }?
2541            }
2542
2543            #[cfg(any(webgpu, webgl))]
2544            SurfaceTarget::OffscreenCanvas(canvas) => {
2545                handle_source = None;
2546
2547                let value: &wasm_bindgen::JsValue = &canvas;
2548                let obj = std::ptr::NonNull::from(value).cast();
2549                let raw_window_handle =
2550                    raw_window_handle::WebOffscreenCanvasWindowHandle::new(obj).into();
2551                let raw_display_handle = raw_window_handle::WebDisplayHandle::new().into();
2552
2553                // Note that we need to call this while we still have `value` around.
2554                // This is safe without storing canvas to `handle_origin` since the surface will create a copy internally.

2556                    self.create_surface_unsafe(SurfaceTargetUnsafe::RawHandle {
2557                        raw_display_handle,
2558                        raw_window_handle,
2559                    })

2562        };

4776impl<'encoder> RenderPass<'encoder> {
4777    /// Set push constant data for subsequent draw calls.
4778    ///
4779    /// Write the bytes in `data` at offset `offset` within push constant
4780    /// storage, all of which are accessible by all the pipeline stages in
4781    /// `stages`, and no others.  Both `offset` and the length of `data` must be
4782    /// multiples of [`PUSH_CONSTANT_ALIGNMENT`], which is always 4.
4783    ///
4784    /// For example, if `offset` is `4` and `data` is eight bytes long, this
4785    /// call will write `data` to bytes `4..12` of push constant storage.
4786    ///
4787    /// # Stage matching
4788    ///
4789    /// Every byte in the affected range of push constant storage must be
4790    /// accessible to exactly the same set of pipeline stages, which must match
4791    /// `stages`. If there are two bytes of storage that are accessible by
4792    /// different sets of pipeline stages - say, one is accessible by fragment
4793    /// shaders, and the other is accessible by both fragment shaders and vertex
4794    /// shaders - then no single `set_push_constants` call may affect both of
4795    /// them; to write both, you must make multiple calls, each with the
4796    /// appropriate `stages` value.
4797    ///
4798    /// Which pipeline stages may access a given byte is determined by the
4799    /// pipeline's [`PushConstant`] global variable and (if it is a struct) its
4800    /// members' offsets.
4801    ///
4802    /// For example, suppose you have twelve bytes of push constant storage,
4803    /// where bytes `0..8` are accessed by the vertex shader, and bytes `4..12`
4804    /// are accessed by the fragment shader. This means there are three byte
4805    /// ranges each accessed by a different set of stages:
4806    ///
4807    /// - Bytes `0..4` are accessed only by the fragment shader.
4808    ///
4809    /// - Bytes `4..8` are accessed by both the fragment shader and the vertex shader.
4810    ///
4811    /// - Bytes `8..12` are accessed only by the vertex shader.
4812    ///
4813    /// To write all twelve bytes requires three `set_push_constants` calls, one
4814    /// for each range, each passing the matching `stages` mask.
4815    ///
4816    /// [`PushConstant`]: https://docs.rs/naga/latest/naga/enum.StorageClass.html#variant.PushConstant
4817    pub fn set_push_constants(&mut self, stages: ShaderStages, offset: u32, data: &[u8]) {
4818        DynContext::render_pass_set_push_constants(
