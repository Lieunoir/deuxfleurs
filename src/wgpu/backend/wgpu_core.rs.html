270    fn handle_error(
271        &self,
272        sink_mutex: &Mutex<ErrorSinkRaw>,
273        source: impl Error + WasmNotSendSync + 'static,
274        label: Label<'_>,
275        fn_ident: &'static str,
276    ) {
277        let error = wgc::error::ContextError {
278            fn_ident,
279            source: Box::new(source),
280            label: label.unwrap_or_default().to_string(),
281        };
282        let mut sink = sink_mutex.lock();
283        let mut source_opt: Option<&(dyn Error + 'static)> = Some(&error);
284        while let Some(source) = source_opt {
285            if let Some(wgc::device::DeviceError::OutOfMemory) =
286                source.downcast_ref::<wgc::device::DeviceError>()
287            {
288                return sink.handle_error(crate::Error::OutOfMemory {
289                    source: Box::new(error),
290                });
291            }
292            source_opt = source.source();
293        }
294
295        // Otherwise, it is a validation error
296        sink.handle_error(crate::Error::Validation {
297            description: self.format_error(&error),
298            source: Box::new(error),
299        });
300    }

320    fn format_error(&self, err: &(impl Error + 'static)) -> String {
321        let mut output = String::new();
322        let mut level = 1;
323
324        fn print_tree(output: &mut String, level: &mut usize, e: &(dyn Error + 'static)) {
325            let mut print = |e: &(dyn Error + 'static)| {
326                use std::fmt::Write;
327                writeln!(output, "{}{}", " ".repeat(*level * 2), e).unwrap();
328
329                if let Some(e) = e.source() {
330                    *level += 1;
331                    print_tree(output, level, e);
332                    *level -= 1;
333                }
334            };
335            if let Some(multi) = e.downcast_ref::<wgc::error::MultiError>() {
336                for e in multi.errors() {
337                    print(e);
338                }
339            } else {
340                print(e);
341            }
342        }
343
344        print_tree(&mut output, &mut level, err);
345
346        format!("Validation Error\n\nCaused by:\n{}", output)
347    }

390fn map_pass_channel<V: Copy + Default>(
391    ops: Option<&Operations<V>>,
392) -> wgc::command::PassChannel<V> {
393    match ops {
394        Some(&Operations {
395            load: LoadOp::Clear(clear_value),
396            store,
397        }) => wgc::command::PassChannel {
398            load_op: wgc::command::LoadOp::Clear,
399            store_op: map_store_op(store),
400            clear_value,
401            read_only: false,
402        },
403        Some(&Operations {
404            load: LoadOp::Load,
405            store,
406        }) => wgc::command::PassChannel {
407            load_op: wgc::command::LoadOp::Load,
408            store_op: map_store_op(store),
409            clear_value: V::default(),
410            read_only: false,
411        },
412        None => wgc::command::PassChannel {
413            load_op: wgc::command::LoadOp::Load,
414            store_op: wgc::command::StoreOp::Store,
415            clear_value: V::default(),
416            read_only: true,
417        },
418    }

889    fn device_create_shader_module(
890        &self,
891        device: &Self::DeviceId,
892        device_data: &Self::DeviceData,
893        desc: ShaderModuleDescriptor<'_>,
894        shader_bound_checks: wgt::ShaderBoundChecks,
895    ) -> (Self::ShaderModuleId, Self::ShaderModuleData) {
896        let descriptor = wgc::pipeline::ShaderModuleDescriptor {
897            label: desc.label.map(Borrowed),
898            shader_bound_checks,
899        };
900        let source = match desc.source {
901            #[cfg(feature = "spirv")]
902            ShaderSource::SpirV(ref spv) => {
903                // Parse the given shader code and store its representation.
904                let options = naga::front::spv::Options {
905                    adjust_coordinate_space: false, // we require NDC_Y_UP feature
906                    strict_capabilities: true,
907                    block_ctx_dump_prefix: None,
908                };
909                wgc::pipeline::ShaderModuleSource::SpirV(Borrowed(spv), options)
910            }
911            #[cfg(feature = "glsl")]
912            ShaderSource::Glsl {
913                ref shader,
914                stage,
915                defines,
916            } => {
917                let options = naga::front::glsl::Options { stage, defines };
918                wgc::pipeline::ShaderModuleSource::Glsl(Borrowed(shader), options)
919            }
920            #[cfg(feature = "wgsl")]
921            ShaderSource::Wgsl(ref code) => wgc::pipeline::ShaderModuleSource::Wgsl(Borrowed(code)),
922            #[cfg(feature = "naga-ir")]
923            ShaderSource::Naga(module) => wgc::pipeline::ShaderModuleSource::Naga(module),
924            ShaderSource::Dummy(_) => panic!("found `ShaderSource::Dummy`"),
925        };
926        let (id, error) = wgc::gfx_select!(
927            device => self.0.device_create_shader_module(*device, &descriptor, source, None)
928        );
929        let compilation_info = match error {
930            Some(cause) => {
931                self.handle_error(
932                    &device_data.error_sink,
933                    cause.clone(),
934                    desc.label,
935                    "Device::create_shader_module",
936                );
937                CompilationInfo::from(cause)
938            }
939            None => CompilationInfo { messages: vec![] },
940        };
941
942        (id, ShaderModule { compilation_info })
943    }

998    fn device_create_bind_group(
999        &self,
1000        device: &Self::DeviceId,
1001        device_data: &Self::DeviceData,
1002        desc: &BindGroupDescriptor<'_>,
1003    ) -> (Self::BindGroupId, Self::BindGroupData) {
1004        use wgc::binding_model as bm;
1005
1006        let mut arrayed_texture_views = Vec::<ObjectId>::new();
1007        let mut arrayed_samplers = Vec::<ObjectId>::new();
1008        if device_data
1009            .features
1010            .contains(Features::TEXTURE_BINDING_ARRAY)
1011        {
1012            // gather all the array view IDs first
1013            for entry in desc.entries.iter() {
1014                if let BindingResource::TextureViewArray(array) = entry.resource {
1015                    arrayed_texture_views.extend(array.iter().map(|view| &view.id));
1016                }
1017                if let BindingResource::SamplerArray(array) = entry.resource {
1018                    arrayed_samplers.extend(array.iter().map(|sampler| &sampler.id));
1019                }
1020            }
1021        }
1022        let mut remaining_arrayed_texture_views = &arrayed_texture_views[..];
1023        let mut remaining_arrayed_samplers = &arrayed_samplers[..];
1024
1025        let mut arrayed_buffer_bindings = Vec::new();
1026        if device_data
1027            .features
1028            .contains(Features::BUFFER_BINDING_ARRAY)
1029        {
1030            // gather all the buffers first
1031            for entry in desc.entries.iter() {
1032                if let BindingResource::BufferArray(array) = entry.resource {
1033                    arrayed_buffer_bindings.extend(array.iter().map(|binding| bm::BufferBinding {
1034                        buffer_id: binding.buffer.id.into(),
1035                        offset: binding.offset,
1036                        size: binding.size,
1037                    }));
1038                }
1039            }
1040        }
1041        let mut remaining_arrayed_buffer_bindings = &arrayed_buffer_bindings[..];
1042
1043        let entries = desc
1044            .entries
1045            .iter()
1046            .map(|entry| bm::BindGroupEntry {
1047                binding: entry.binding,
1048                resource: match entry.resource {
1049                    BindingResource::Buffer(BufferBinding {
1050                        buffer,
1051                        offset,
1052                        size,
1053                    }) => bm::BindingResource::Buffer(bm::BufferBinding {
1054                        buffer_id: buffer.id.into(),
1055                        offset,
1056                        size,
1057                    }),
1058                    BindingResource::BufferArray(array) => {
1059                        let slice = &remaining_arrayed_buffer_bindings[..array.len()];
1060                        remaining_arrayed_buffer_bindings =
1061                            &remaining_arrayed_buffer_bindings[array.len()..];
1062                        bm::BindingResource::BufferArray(Borrowed(slice))
1063                    }
1064                    BindingResource::Sampler(sampler) => {
1065                        bm::BindingResource::Sampler(sampler.id.into())
1066                    }
1067                    BindingResource::SamplerArray(array) => {
1068                        let samplers = remaining_arrayed_samplers[..array.len()]
1069                            .iter()
1070                            .map(|id| <Self::SamplerId>::from(*id))
1071                            .collect::<Vec<_>>();
1072                        remaining_arrayed_samplers = &remaining_arrayed_samplers[array.len()..];
1073                        bm::BindingResource::SamplerArray(Owned(samplers))
1074                    }
1075                    BindingResource::TextureView(texture_view) => {
1076                        bm::BindingResource::TextureView(texture_view.id.into())
1077                    }
1078                    BindingResource::TextureViewArray(array) => {
1079                        let views = remaining_arrayed_texture_views[..array.len()]
1080                            .iter()
1081                            .map(|id| <Self::TextureViewId>::from(*id))
1082                            .collect::<Vec<_>>();
1083                        remaining_arrayed_texture_views =
1084                            &remaining_arrayed_texture_views[array.len()..];
1085                        bm::BindingResource::TextureViewArray(Owned(views))
1086                    }
1087                },
1088            })
1089            .collect::<Vec<_>>();
1090        let descriptor = bm::BindGroupDescriptor {
1091            label: desc.label.as_ref().map(|label| Borrowed(&label[..])),
1092            layout: desc.layout.id.into(),
1093            entries: Borrowed(&entries),
1094        };
1095
1096        let (id, error) = wgc::gfx_select!(device => self.0.device_create_bind_group(
1097            *device,
1098            &descriptor,
1099            None
1100        ));
1101        if let Some(cause) = error {
1102            self.handle_error(
1103                &device_data.error_sink,
1104                cause,
1105                desc.label,
1106                "Device::create_bind_group",
1107            );
1108        }
1109        (id, ())
1110    }

3379    fn handle_error(&mut self, err: crate::Error) {
3380        let filter = match err {
3381            crate::Error::OutOfMemory { .. } => crate::ErrorFilter::OutOfMemory,
3382            crate::Error::Validation { .. } => crate::ErrorFilter::Validation,
3383            crate::Error::Internal { .. } => crate::ErrorFilter::Internal,
3384        };
3385        match self
3386            .scopes
3387            .iter_mut()
3388            .rev()
3389            .find(|scope| scope.filter == filter)
3390        {
3391            Some(scope) => {
3392                if scope.error.is_none() {
3393                    scope.error = Some(err);
3394                }
3395            }
3396            None => {
3397                (self.uncaptured_handler)(err);
3398            }
3399        }
3400    }
