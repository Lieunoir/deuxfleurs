45fn get_bars(values: &[f32], n_bars: usize) -> Vec<Bar> {
46    let (min, max) = values
47        .iter()
48        .fold((f32::MAX, f32::MIN), |(min, max), value| {
49            (min.min(*value), max.max(*value))
50        });
51    let mut histogram = vec![0; n_bars];
52    for v in values {
53        let clamped = (v - min) / (max - min);
54        let clamped = (clamped * (n_bars as f32)) as usize;
55        histogram[clamped.min(n_bars - 1)] += 1;
56    }
57    histogram
58        .into_iter()
59        .enumerate()
60        .map(|(i, s)| {
61            Bar::new((min + (max - min) * i as f32) as f64, s as f64).width((max - min) as f64)
62        })
63        .collect()
64}

66impl ColorMap {
67    pub(crate) fn new(values: &[f32]) -> Self {
68        let mut res = Self {
69            bars: get_bars(values, 120),
70            colors: Colors::Viridis,
71            min: 0.,
72            max: 1.,
73        };
74        res.apply_bar_colors();
75        res
76    }
77
78    fn apply_bar_colors(&mut self) {
79        let color_map = self.colors;
80        let n_values = self.bars.len();
81        for (i, bar) in self.bars.iter_mut().enumerate() {
82            let value =
83                (-self.min as f32 + i as f32 / n_values as f32) / (self.max - self.min) as f32;
84            let opacify = if value < 0. || value > 1. { 0.6 } else { 1. };
85            let value = value.max(0.);
86            let value = value.min(1.);
87            let colors = color_map.compute_color(value);
88            let bar_c = bar.clone();
89            let color = Color32::from_rgb(
90                (colors[0] * opacify * 255.) as u8,
91                (colors[1] * opacify * 255.) as u8,
92                (colors[2] * opacify * 255.) as u8,
93            );
94            *bar = bar_c.fill(color);
95        }
96    }
97
98    pub(crate) fn get_value(&self) -> ColorMapValues {
99        ColorMapValues {
100            colors: self.colors.get_value(),
101            min: self.min as f32,
102            max: self.max as f32,
103            _pad: [0; 2],
104        }
105    }
106
107    pub(crate) fn recycle(&mut self, other: Self) {
108        self.min = other.min;
109        self.max = other.max;
110        self.colors = other.colors;
111        self.apply_bar_colors();
112    }
113}

375    fn compute_color(&self, x: f32) -> [f32; 3] {
376        //let x = (clamp(dist, data_uniform.min, data_uniform.max) - data_uniform.min) / (data_uniform.max - data_uniform.min);
377        let ColorsValues { red, green, blue } = self.get_value();
378        let polynomial = [
379            1.0,
380            x,
381            x * x,
382            x * x * x,
383            x * x * x * x,
384            x * x * x * x * x,
385            x * x * x * x * x * x,
386            x * x * x * x * x * x * x,
387        ];
388        //let v4: vec4<f32> = vec4<f32>(1.0, x, x*x, x*x*x);
389        //let v2: vec4<f32> = v4 * v4.w * x;
390        let red = polynomial
391            .iter()
392            .zip(red)
393            .fold(0., |acc, (c, v)| acc + (c * v));
394        let green = polynomial
395            .iter()
396            .zip(green)
397            .fold(0., |acc, (c, v)| acc + (c * v));
398        let blue = polynomial
399            .iter()
400            .zip(blue)
401            .fold(0., |acc, (c, v)| acc + (c * v));
402        [red, green, blue]
403    }

406pub fn windowing_ui(
407    ui: &mut egui::Ui,
408    width: &f64,
409    height: &f64,
410    min: f64,
411    max: f64,
412    lb: &mut f64,
413    ub: &mut f64,
414) -> egui::Response {
415    let desired_size = egui::vec2(*width as f32, *height as f32 / 10.0);
416    let (rect, mut response) = ui.allocate_exact_size(desired_size, egui::Sense::drag());
417    let range = max - min;
418    if ui.is_rect_visible(rect) {
419        let visuals = ui.style().interact(&response);
420        let bar_line = vec![rect.left_center(), rect.right_center()];
421        ui.painter().add(Path(PathShape::line(
422            bar_line,
423            Stroke::new(*height as f32 / 20.0, visuals.bg_fill),
424        )));
425        let mut bounds_i = [0.0; 2];
426        bounds_i[0] = *lb / range * width;
427        bounds_i[1] = *ub / range * width;
428        let mut lb_pos = Pos2 {
429            x: rect.left_center().x + bounds_i[0] as f32,
430            y: rect.left_center().y,
431        };
432        let mut ub_pos = Pos2 {
433            x: rect.left_center().x + bounds_i[1] as f32,
434            y: rect.right_center().y,
435        };
436
437        if response.dragged() {
438            let pos = response.interact_pointer_pos();
439
440            match pos {
441                None => {}
442                Some(p) => {
443                    let dist_1 = (lb_pos.x - p.x).abs();
444                    let dist_2 = (ub_pos.x - p.x).abs();
445                    if dist_1 < dist_2 {
446                        // dragging the lower one
447                        lb_pos.x = f32::min(f32::max(p.x, rect.left_center().x), ub_pos.x);
448                        response.mark_changed();
449                    } else {
450                        // dragging the upper one
451                        ub_pos.x = f32::max(f32::min(p.x, rect.right_center().x), lb_pos.x);
452                        response.mark_changed();
453                    }
454                }
455            }
456        }
457
458        bounds_i[0] = lb_pos.x as f64 - rect.left_center().x as f64;
459        *lb = bounds_i[0] / width * range;
460        bounds_i[1] = ub_pos.x as f64 - rect.left_center().x as f64;
461        *ub = bounds_i[1] / width * range;
462
463        let bar_line = vec![lb_pos, ub_pos];
464        ui.painter().add(Path(PathShape::line(
465            bar_line,
466            Stroke::new(*height as f32 / 20.0, visuals.fg_stroke.color),
467        )));
468        let radius = *width as f32 / 20.;
469        ui.painter().add(epaint::CircleShape {
470            center: lb_pos,
471            radius,
472            fill: visuals.bg_fill,
473            stroke: visuals.fg_stroke,
474        });
475        ui.painter().add(epaint::CircleShape {
476            center: ub_pos,
477            radius,
478            fill: visuals.bg_fill,
479            stroke: visuals.fg_stroke,
480        });
481    }
482    response
483}

485impl UiDataElement for ColorMap {
486    fn draw(&mut self, ui: &mut egui::Ui, _property_changed: &mut bool) -> bool {
487        let mut changed = false;
488        egui::ComboBox::from_label("ColorMap")
489            .selected_text(self.colors.get_name())
490            .show_ui(ui, |ui| {
491                changed |= ui
492                    .selectable_value(&mut self.colors, Colors::Turbo, Colors::Turbo.get_name())
493                    .changed();
494                changed |= ui
495                    .selectable_value(
496                        &mut self.colors,
497                        Colors::Viridis,
498                        Colors::Viridis.get_name(),
499                    )
500                    .changed();
501                changed |= ui
502                    .selectable_value(
503                        &mut self.colors,
504                        Colors::Inferno,
505                        Colors::Inferno.get_name(),
506                    )
507                    .changed();
508                changed |= ui
509                    .selectable_value(&mut self.colors, Colors::Magma, Colors::Magma.get_name())
510                    .changed();
511                changed |= ui
512                    .selectable_value(&mut self.colors, Colors::Plasma, Colors::Plasma.get_name())
513                    .changed();
514                changed |= ui
515                    .selectable_value(&mut self.colors, Colors::BrBG, Colors::BrBG.get_name())
516                    .changed();
517                changed |= ui
518                    .selectable_value(&mut self.colors, Colors::RdBU, Colors::RdBU.get_name())
519                    .changed();
520                changed |= ui
521                    .selectable_value(
522                        &mut self.colors,
523                        Colors::Coolwarm,
524                        Colors::Coolwarm.get_name(),
525                    )
526                    .changed();
527                changed |= ui
528                    .selectable_value(
529                        &mut self.colors,
530                        Colors::Twilight,
531                        Colors::Twilight.get_name(),
532                    )
533                    .changed();
534            });
535        if changed {
536            self.apply_bar_colors();
537        }
538
539        Plot::new("plot")
540            .label_formatter(|_, _| "".to_owned())
541            .width(200.)
542            .show_y(false)
543            .show_x(true)
544            .allow_zoom(false)
545            .allow_boxed_zoom(false)
546            .allow_scroll(false)
547            .allow_drag(false)
548            .clamp_grid(true)
549            .show_axes(false)
550            .show_grid(false)
551            .view_aspect(4.)
552            .coordinates_formatter(
553                egui_plot::Corner::LeftTop,
554                CoordinatesFormatter::new(|c, _| format!("{:.4}", c.x)),
555            )
556            .show(ui, |ui| {
557                ui.bar_chart(BarChart::new(self.bars.clone()).allow_hover(false));
558            });
559        ui.horizontal(|ui| {
560            if windowing_ui(ui, &100., &100., 0., 1., &mut self.min, &mut self.max).changed() {
561                changed = true;
562                self.apply_bar_colors();
563            }
564            ui.label("Range");
565        });
566        changed
567    }
