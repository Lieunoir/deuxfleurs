33impl<T> DataUniformBuilder for T
34where
35    T: bytemuck::Pod + Copy,
36{
37    fn build_uniform(&self, device: &wgpu::Device) -> Option<DataUniform> {
38        let buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
39            label: Some("Data buffer"),
40            contents: bytemuck::cast_slice(&[*self]),
41            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
42        });
43        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
44            entries: &[wgpu::BindGroupLayoutEntry {
45                binding: 0,
46                visibility: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
47                ty: wgpu::BindingType::Buffer {
48                    ty: wgpu::BufferBindingType::Uniform,
49                    has_dynamic_offset: false,
50                    min_binding_size: None,
51                },
52                count: None,
53            }],
54            label: Some("data_bind_group_layout"),
55        });
56
57        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
58            layout: &bind_group_layout,
59            entries: &[wgpu::BindGroupEntry {
60                binding: 0,
61                resource: buffer.as_entire_binding(),
62            }],
63            label: Some("data_bind_group"),
64        });
65        Some(DataUniform {
66            bind_group_layout,
67            bind_group,
68            buffer,
69        })
70    }
71
72    fn refresh_buffer(&self, queue: &mut wgpu::Queue, data_uniform: &DataUniform) {
73        queue.write_buffer(&data_uniform.buffer, 0, bytemuck::cast_slice(&[*self]));
74    }
75}
