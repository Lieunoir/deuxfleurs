15impl TransformSettings {
16    pub fn get_transform(&self) -> [[f32; 4]; 4] {
17        let model_m =
18            DMat4::from_scale_rotation_translation(self.scale, self.rotation, self.translation);
19        let mut model: [[f32; 4]; 4] = [[0.; 4]; 4];
20        for (row_m, row) in model_m.to_cols_array_2d().into_iter().zip(model.iter_mut()) {
21            for (v_m, v) in row_m.into_iter().zip(row.iter_mut()) {
22                *v = v_m as f32;
23            }
24        }
25        model
26    }
27
28    pub fn to_raw(&self) -> TransformRaw {
29        use cgmath::Matrix;
30        use cgmath::SquareMatrix;
31        //let model = self.transform;
32        let mut normal: [[f32; 3]; 3] = [[0.; 3]; 3];
33        let model = self.get_transform();
34        for (row, row_orig) in normal.iter_mut().zip(model) {
35            *row = row_orig[0..3].try_into().unwrap();
36        }
37        let mut normal: cgmath::Matrix3<f32> = normal.into();
38        normal = normal.invert().unwrap().transpose();
39        //Conversion tricks from mat3x3 to mat4x4
40        let normal: cgmath::Matrix4<f32> = normal.into();
41        TransformRaw {
42            model,
43            normal: normal.into(),
44        }
45    }
46    pub fn draw_transform(&mut self, ui: &mut egui::Ui, positions: &[[f32; 3]]) -> bool {
47        let mut changed = false;
48        ui.horizontal(|ui| {
49            ui.checkbox(&mut self.show_gizmo, "Show Gizmo");
50        });
51        ui.horizontal(|ui| {
52            if ui.add(egui::Button::new("Reset")).clicked() {
53                self.translation = DVec3::ZERO;
54                self.scale = DVec3::ONE;
55                self.rotation = DQuat::IDENTITY;
56                changed = true;
57            }
58            if ui.add(egui::Button::new("Center")).clicked() {
59                let mut min_x = std::f32::MAX;
60                let mut min_y = std::f32::MAX;
61                let mut min_z = std::f32::MAX;
62                let mut max_x = std::f32::MIN;
63                let mut max_y = std::f32::MIN;
64                let mut max_z = std::f32::MIN;
65
66                let model = self.get_transform();
67
68                let transfo_matrix: cgmath::Matrix4<f32> = model.into();
69                for position in positions {
70                    let threed_point: cgmath::Point3<f32> = (*position).into();
71
72                    let position = cgmath::Point3::<f32>::from_homogeneous(
73                        transfo_matrix * threed_point.to_homogeneous(),
74                    );
75                    if position[0] < min_x {
76                        min_x = position[0];
77                    }
78                    if position[1] < min_y {
79                        min_y = position[1];
80                    }
81                    if position[2] < min_z {
82                        min_z = position[2];
83                    }
84                    if position[0] > max_x {
85                        max_x = position[0];
86                    }
87                    if position[1] > max_y {
88                        max_y = position[1];
89                    }
90                    if position[2] > max_z {
91                        max_z = position[2];
92                    }
93                }
94                let x = (max_x + min_x) / 2.;
95                //let y = (max_y + min_y) / 2.;
96                let y = min_y;
97                let z = (max_z + min_z) / 2.;
98                self.translation += DVec3::from_array([-x as f64, -y as f64, -z as f64]);
99                changed = true;
100            }
101            if ui.add(egui::Button::new("Unit Scale")).clicked() {
102                let mut min_x = std::f32::MAX;
103                let mut min_y = std::f32::MAX;
104                let mut min_z = std::f32::MAX;
105                let mut max_x = std::f32::MIN;
106                let mut max_y = std::f32::MIN;
107                let mut max_z = std::f32::MIN;
108
109                let model = self.get_transform();
110
111                let transfo_matrix: cgmath::Matrix4<f32> = model.into();
112                for vertex in positions {
113                    let threed_point: cgmath::Point3<f32> = (*vertex).into();
114
115                    let position = cgmath::Point3::<f32>::from_homogeneous(
116                        transfo_matrix * threed_point.to_homogeneous(),
117                    );
118                    if position[0] < min_x {
119                        min_x = position[0];
120                    }
121                    if position[1] < min_y {
122                        min_y = position[1];
123                    }
124                    if position[2] < min_z {
125                        min_z = position[2];
126                    }
127                    if position[0] > max_x {
128                        max_x = position[0];
129                    }
130                    if position[1] > max_y {
131                        max_y = position[1];
132                    }
133                    if position[2] > max_z {
134                        max_z = position[2];
135                    }
136                }
137                let x = max_x - min_x;
138                let y = max_y - min_y;
139                let z = max_z - min_z;
140                let scale = 1. / (x * x + y * y + z * z).sqrt();
141                self.scale *= scale as f64;
142                changed = true;
143            }
144        });
145        changed
146    }

192    fn draw_gizmo(
193        &mut self,
194        ui: &mut egui::Ui,
195        _name: &str,
196        view: cgmath::Matrix4<f32>,
197        proj: cgmath::Matrix4<f32>,
198        gizmo_hovered: &mut bool,
199    ) -> bool {
200        if self.show_gizmo {
201            let viewport = ui.clip_rect();
202            let view: [[f32; 4]; 4] = view.into();
203            let proj: [[f32; 4]; 4] = proj.into();
204            let mut view_m = DMat4::ZERO;
205            let mut proj_m = DMat4::ZERO;
206            for (i, (row_m, row)) in view_m
207                .to_cols_array_2d()
208                .iter_mut()
209                .zip(view.into_iter())
210                .enumerate()
211            {
212                for (v_m, v) in row_m.iter_mut().zip(row.into_iter()) {
213                    *v_m = v as f64;
214                }
215                match i {
216                    0 => view_m.x_axis = (*row_m).into(),
217                    1 => view_m.y_axis = (*row_m).into(),
218                    2 => view_m.z_axis = (*row_m).into(),
219                    _ => view_m.w_axis = (*row_m).into(),
220                }
221            }
222            for (i, (row_m, row)) in proj_m
223                .to_cols_array_2d()
224                .iter_mut()
225                .zip(proj.into_iter())
226                .enumerate()
227            {
228                for (v_m, v) in row_m.iter_mut().zip(row.into_iter()) {
229                    *v_m = v as f64;
230                }
231                match i {
232                    0 => proj_m.x_axis = (*row_m).into(),
233                    1 => proj_m.y_axis = (*row_m).into(),
234                    2 => proj_m.z_axis = (*row_m).into(),
235                    _ => proj_m.w_axis = (*row_m).into(),
236                }
237            }
238            self.gizmo.update_config(GizmoConfig {
239                view_matrix: view_m.into(),
240                projection_matrix: proj_m.into(),
241                modes: GizmoMode::all()
242                    .difference(GizmoMode::all_scale())
243                    .difference(enum_set!(GizmoMode::TranslateView))
244                    .difference(enum_set!(GizmoMode::RotateView))
245                    .union(enum_set!(GizmoMode::ScaleUniform)),
246                orientation: GizmoOrientation::Local,
247                viewport,
248                ..Default::default()
249            });
250
251            let mut transform = Transform::from_scale_rotation_translation(
252                self.scale,
253                self.rotation,
254                self.translation,
255            );
256
257            let res = if let Some((_result, new_transforms)) = self.gizmo.interact(ui, &[transform])
258            {
259                for (new_transform, transform) in
260                    new_transforms.iter().zip(std::iter::once(&mut transform))
261                {
262                    // Apply the modified transforms
263                    *transform = *new_transform;
264                }
265                self.scale = transform.scale.into();
266                self.rotation = transform.rotation.into();
267                self.translation = transform.translation.into();
268
269                true
270            } else {
271                false
272            };
273            *gizmo_hovered |= self.gizmo.is_focused();
274            res
275        } else {
276            false
277        }
278    }
