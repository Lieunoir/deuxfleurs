11const PICKER_SHADER: &str = "
12// Vertex shader
13
14// Define any uniforms we expect from app
15struct CameraUniform {
16    view_pos: vec4<f32>,
17    view_proj: mat4x4<f32>,
18}
19
20struct Light {
21    position: vec3<f32>,
22    color: vec3<f32>,
23}
24
25struct TransformUniform {
26    model: mat4x4<f32>,
27    normal: mat4x4<f32>,
28}
29
30struct CounterUniform {
31    count: u32,
32    _padding_1: u32,
33    _padding_2: u32,
34    _padding_3: u32,
35}
36
37@group(0) @binding(0)
38var<uniform> camera: CameraUniform;
39@group(0) @binding(1)
40var<uniform> light: Light;
41
42@group(1) @binding(0)
43var<uniform> counter: CounterUniform;
44
45@group(2) @binding(0)
46var<uniform> transform: TransformUniform;
47
48// This is the input from the vertex buffer we created
49// We get the properties from our Vertex struct here
50// Note the index on location -- this relates to the properties placement in the buffer stride
51// e.g. 0 = 1st \"set\" of data, 1 = 2nd \"set\"
52struct VertexInput {
53    @location(0) position: vec3<f32>,
54    @builtin(vertex_index) face_index: u32,
55};
56
57// The output we send to our fragment shader
58struct VertexOutput {
59    @builtin(position) clip_position: vec4<f32>,
60    @location(0) face_index: u32,
61};
62
63@vertex
64fn vs_main(
65    model: VertexInput,
66) -> VertexOutput {
67    // We define the output we want to send over to frag shader
68    var out: VertexOutput;
69    let model_matrix = transform.model;
70
71    out.face_index = counter.count + model.face_index / 3;
72
73    // We set the \"position\" by using the `clip_position` property
74    // We multiply it by the camera position matrix and the instance position matrix
75    out.clip_position = camera.view_proj * model_matrix * vec4<f32>(model.position, 1.0);
76    return out;
77}
78
79// Fragment shader
80@fragment
81fn fs_main(in: VertexOutput) -> @location(0) vec4<f32> {
82//u32 {
83    //return bitcast<vec4<f32>>(res);
84    // webgl dosen't support rendering to u32, so we have to resort to this
85    let res = in.face_index;
86    let f1 = f32((res >> u32(24))) / 255.;
87    let f2 = f32(((res << u32(8)) >> u32(24))) / 255.;
88    let f3 = f32(((res << u32(16)) >> u32(24))) / 255.;
89    let f4 = f32(((res << u32(24)) >> u32(24))) / 255.;
90    return vec4<f32>(f4, f3, f2, f1);
91    //return unpack4x8unorm(in.face_index);
92}
93";

125fn build_render_pipeline(
126    device: &wgpu::Device,
127    camera_light_bind_group_layout: &wgpu::BindGroupLayout,
128    counter_bind_group_layout: &wgpu::BindGroupLayout,
129    transform_bind_group_layout: &wgpu::BindGroupLayout,
130) -> wgpu::RenderPipeline {
131    let render_pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
132        label: Some("SurfaceGeometry Picker Pipeline Layout"),
133        bind_group_layouts: &[
134            camera_light_bind_group_layout,
135            counter_bind_group_layout,
136            transform_bind_group_layout,
137        ],
138        push_constant_ranges: &[],
139    });
140    let shader = wgpu::ShaderModuleDescriptor {
141        label: Some("SurfaceGeometry Picker Shader"),
142        source: wgpu::ShaderSource::Wgsl(PICKER_SHADER.into()),
143    };
144    create_picker_pipeline(
145        device,
146        &render_pipeline_layout,
147        texture::Texture::PICKER_FORMAT,
148        Some(texture::Texture::DEPTH_FORMAT),
149        &[VertexData::desc()],
150        shader,
151        Some("surface picker"),
152    )
153}

155fn build_vertex_buffer(device: &wgpu::Device, surface: &SurfaceGeometry) -> wgpu::Buffer {
156    let mut gpu_vertices = Vec::with_capacity(3 * surface.internal_indices.len());
157    for indices in surface.indices.into_iter() {
158        for j in 1..indices.len() - 1 {
159            let index0 = indices[0];
160            let index1 = indices[j];
161            let index2 = indices[j + 1];
162            let v_indices = [index0, index1, index2];
163            for k in 0..3 {
164                gpu_vertices.push(VertexData {
165                    position: surface.vertices[v_indices[k] as usize],
166                });
167            }
168        }
169    }
170    device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
171        label: Some("Picker Vertex Buffer"),
172        contents: bytemuck::cast_slice(&gpu_vertices),
173        usage: wgpu::BufferUsages::VERTEX,
174    })
175}

177fn build_transform_buffer(device: &wgpu::Device, transform: &TransformSettings) -> wgpu::Buffer {
178    let transform_raw = transform.to_raw();
179    device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
180        label: Some("Picker Transform Buffer"),
181        contents: bytemuck::cast_slice(&[transform_raw]),
182        usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
183    })
184}

189    fn new(
190        surface: &Self::Geometry,
191        _settings: &Self::Settings,
192        transform: &TransformSettings,
193        device: &wgpu::Device,
194        camera_light_bind_group_layout: &wgpu::BindGroupLayout,
195        counter_bind_group_layout: &wgpu::BindGroupLayout,
196    ) -> Self {
197        let vertex_buffer = build_vertex_buffer(device, surface);
198        let transform_buffer = build_transform_buffer(device, transform);
199        let transform_bind_group_layout =
200            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
201                entries: &[wgpu::BindGroupLayoutEntry {
202                    binding: 0,
203                    visibility: wgpu::ShaderStages::VERTEX,
204                    ty: wgpu::BindingType::Buffer {
205                        ty: wgpu::BufferBindingType::Uniform,
206                        has_dynamic_offset: false,
207                        min_binding_size: None,
208                    },
209                    count: None,
210                }],
211                label: Some("transform_bind_group_layout"),
212            });
213
214        let transform_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
215            layout: &transform_bind_group_layout,
216            entries: &[wgpu::BindGroupEntry {
217                binding: 0,
218                resource: transform_buffer.as_entire_binding(),
219            }],
220            label: Some("transform_bind_group"),
221        });
222
223        let num_elements = (surface.internal_indices.len() * 3) as u32;
224        let render_pipeline = build_render_pipeline(
225            device,
226            camera_light_bind_group_layout,
227            counter_bind_group_layout,
228            &transform_bind_group_layout,
229        );
230        //let tot_elements = (surface.vertices.len() + surface.indices.size()) as u32;
231        let tot_elements = surface.indices.tot_triangles() as u32;
232        Self {
233            vertex_buffer,
234            num_elements,
235            transform_bind_group,
236            transform_buffer,
237            render_pipeline,
238            tot_elements,
239        }

257    fn get_element(
258        &self,
259        surface: &Self::Geometry,
260        transform: &TransformSettings,
261        camera: &Camera,
262        item: u32,
263        pos_x: f32,
264        pos_y: f32,
265    ) -> u32 {
266        let indices = &surface.indices;
267        let vertices = &surface.vertices;
268        let (face_index, face_indices) = match indices {
269            SurfaceIndices::Triangles(t) => (item, t[item as usize]),
270            SurfaceIndices::Quads(t) => (
271                item / 2,
272                if item % 2 == 0 {
273                    [
274                        t[item as usize / 2][0],
275                        t[item as usize / 2][1],
276                        t[item as usize / 2][2],
277                    ]
278                } else {
279                    [
280                        t[item as usize / 2][0],
281                        t[item as usize / 2][2],
282                        t[item as usize / 2][3],
283                    ]
284                },
285            ),
286            SurfaceIndices::Polygons(indices, s) => {
287                let mut elapsed = 0;
288                let mut index = 0;
289                let mut face = [0, 0, 0];
290                for (i, size) in s.iter().enumerate() {
291                    if elapsed + size - 2 > item {
292                        for j in 0..(size - 2) {
293                            if elapsed + j == item {
294                                index = i as u32;
295                                face = [
296                                    indices[elapsed as usize + i * 2 + 0],
297                                    indices[elapsed as usize + i * 2 + j as usize + 1],
298                                    indices[elapsed as usize + i * 2 + j as usize + 2],
299                                ];
300                                break;
301                            }
302                        }
303                        break;
304                    } else {
305                        elapsed += size - 2;
306                    }
307                }
308                (index, face)
309            }
310        };
311        let v1: cgmath::Point3<f32> = vertices[face_indices[0] as usize].into();
312        let v2: cgmath::Point3<f32> = vertices[face_indices[1] as usize].into();
313        let v3: cgmath::Point3<f32> = vertices[face_indices[2] as usize].into();
314        let v1 = v1.to_homogeneous();
315        let v2 = v2.to_homogeneous();
316        let v3 = v3.to_homogeneous();
317        let model: cgmath::Matrix4<f32> = transform.to_raw().get_model().into();
318        let camera = camera.build_view_projection_matrix();
319        let v1 = camera * model * v1;
320        let v2 = camera * model * v2;
321        let v3 = camera * model * v3;
322        let w1 = v1.w;
323        let w2 = v2.w;
324        let w3 = v3.w;
325        let v1 = v1 / v1.w;
326        let v2 = v2 / v2.w;
327        let v3 = v3 / v3.w;
328        let p = cgmath::vec3(pos_x, pos_y, 0.);
329        let v1 = cgmath::vec3(v1.x, v1.y, 0.);
330        let v2 = cgmath::vec3(v2.x, v2.y, 0.);
331        let v3 = cgmath::vec3(v3.x, v3.y, 0.);
332
333        let c3 = (v1 - p).cross(v2 - p).magnitude();
334        let c1 = (v2 - p).cross(v3 - p).magnitude();
335        let c2 = (v3 - p).cross(v1 - p).magnitude();
336        let tot = c3 + c1 + c2;
337        let c1 = c1 / tot;
338        let c2 = c2 / tot;
339        let c3 = c3 / tot;
340        let c1 = c1 / w1 / (c1 / w1 + c2 / w2 + c3 / w3);
341        let c2 = c2 / w2 / (c1 / w1 + c2 / w2 + c3 / w3);
342        let c3 = c3 / w3 / (c1 / w1 + c2 / w2 + c3 / w3);
343        if c1 > 0.7 {
344            face_indices[0]
345        } else if c2 > 0.7 {
346            face_indices[1]
347        } else if c3 > 0.7 {
348            face_indices[2]
349        } else {
350            vertices.len() as u32 + face_index
351        }
352    }
353}

355impl Render for Picker {
356    fn render<'a, 'b>(&'a self, render_pass: &mut wgpu::RenderPass<'b>)
357    where
358        'a: 'b,
359    {
360        render_pass.set_bind_group(2, &self.transform_bind_group, &[]);
361        render_pass.set_pipeline(&self.render_pipeline);
362        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
363        render_pass.draw(0..self.num_elements, 0..1);
