116// PBR functions taken from https://learnopengl.com/PBR/Theory
117fn DistributionGGX(N: vec3<f32>, H: vec3<f32>, a: f32) -> f32 {{
118    let a2     = a*a;
119    let NdotH  = max(dot(N, H), 0.0);
120    let NdotH2 = NdotH*NdotH;
121
122    let nom    = a2;
123    var denom  = (NdotH2 * (a2 - 1.0) + 1.0);
124    denom        = PI * denom * denom;
125
126    return nom / denom;
127}}
128
129fn GeometrySchlickGGX(NdotV: f32, k: f32) -> f32
130{{
131    let nom   = NdotV;
132    let denom = NdotV * (1.0 - k) + k;
133
134    return nom / denom;
135}}
136
137fn GeometrySmith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, k: f32) -> f32
138{{
139    let NdotV = max(dot(N, V), 0.0);
140    let NdotL = max(dot(N, L), 0.0);
141    let ggx1 = GeometrySchlickGGX(NdotV, k);
142    let ggx2 = GeometrySchlickGGX(NdotL, k);
143
144    return ggx1 * ggx2;
145}}

147fn fresnelSchlick(cosTheta: f32, F0: vec3<f32>) -> vec3<f32>
148{{
149    return F0 + (1.0 - F0) * pow(1.0 - cosTheta, 5.0);
150}}

353pub fn get_shader(data_format: Option<&SurfaceData>, smooth: bool, show_edge: bool) -> String {
354    let uniform = if let Some(mesh_data) = data_format {
355        match mesh_data {
356            SurfaceData::UVMap(_, _) | SurfaceData::UVCornerMap(_, _) => CHECKERBOARD_UNIFORM,
357            SurfaceData::VertexScalar(_, _) => COLORMAP_ISOLINES_UNIFORM,
358            SurfaceData::FaceScalar(_, _) => COLORMAP_UNIFORM,
359            _ => "",
360        }
361    } else {
362        COLOR_UNIFORM
363    };
364
365    let (data_decl, data_out) = match data_format {
366        Some(data) => match data {
367            SurfaceData::UVMap(..) | SurfaceData::UVCornerMap(..) => (
368                "
369struct DataInput {
370    @location(4) data: vec2<f32>,
371};",
372                "@location(3) data: vec2<f32>,",
373            ),
374            SurfaceData::VertexScalar(..) | SurfaceData::FaceScalar(..) => (
375                "
376struct DataInput {
377    @location(4) data: f32,
378};",
379                "@location(3) data: f32,",
380            ),
381            SurfaceData::Color(..) => (
382                "
383struct DataInput {
384    @location(4) data: vec3<f32>,
385};",
386                "@location(3) data: vec3<f32>,",
387            ),
388        },
389        None => ("", ""),
390    };
391
392    let (data_input, data_assign) = match data_format {
393        Some(_data) => (
394            "
395    data: DataInput,",
396            "out.data = data.data;",
397        ),
398        None => ("", ""),
399    };
400
401    let normal_interpolation = if smooth {
402        SMOOTH_NORMAL_INTERPOLATION
403    } else {
404        FLAT_NORMAL_INTERPOLATION
405    };
406    let render_modif = match data_format {
407        Some(SurfaceData::UVMap(_, _)) | Some(SurfaceData::UVCornerMap(_, _)) => CHECKERBOARD,
408        Some(SurfaceData::VertexScalar(_, _)) => COLORMAP_ISOLINES,
409        Some(SurfaceData::FaceScalar(_, _)) => COLORMAP,
410        _ => "",
411    };
412    let edge_shader = if show_edge { WITH_EDGE_SHADER } else { "" };
413    format!(
414        SHADER!(),
415        uniform,
416        data_decl,
417        data_out,
418        data_input,
419        normal_interpolation,
420        data_assign,
421        render_modif,
422        edge_shader
423    )
424}
