92// cylinder defined by extremes a and b, and radious ra
93fn cylIntersect( ro: vec3<f32>, rd: vec3<f32>, a: vec3<f32>, b: vec3<f32>, ra: f32 ) -> vec4<f32>
94{{
95    let ba = b  - a;
96    let oc = ro - a;
97    let baba = dot(ba,ba);
98    let bard = dot(ba,rd);
99    let baoc = dot(ba,oc);
100    let k2 = baba            - bard*bard;
101    let k1 = baba*dot(oc,rd) - baoc*bard;
102    let k0 = baba*dot(oc,oc) - baoc*baoc - ra*ra*baba;
103    var h = k1*k1 - k2*k0;
104    if( h<0.0 ) {{ return vec4(-1.0); }}//no intersection
105    h = sqrt(h);
106    let t = (-k1-h)/k2;
107    // body
108    let y = baoc + t*bard;
109    if( y>0.0 && y<baba ) {{ return vec4( t, (oc+t*rd - ba*y/baba)/ra ); }}
110    return vec4(-1.0);//no intersection
111}}
112
113// normal at sphere p of cylinder (a,b,ra), see above
114fn cylNormal( p: vec3<f32>, a: vec3<f32>, b: vec3<f32>, ra: f32 ) -> vec3<f32>
115{{
116    let pa = p - a;
117    let ba = b - a;
118    let baba = dot(ba,ba);
119    let paba = dot(pa,ba);
120    let h = dot(pa,ba)/baba;
121    return (pa - ba*h)/ra;
122}}

155pub fn get_shader(data_format: Option<&SegmentData>) -> String {
156    let data_struct = match data_format {
157        Some(SegmentData::Scalar(..)) => {
158            "
159struct DataInput {
160    @location(3) val_1: f32,
161    @location(4) val_2: f32,
162}"
163        }
164        Some(SegmentData::Color(_)) => {
165            "
166struct DataInput {
167    @location(3) val_1: vec3<f32>,
168    @location(4) val_2: vec3<f32>,
169}"
170        }
171        None => "",
172    };
173
174    let uniform = match data_format {
175        Some(SegmentData::Scalar(..)) => shader::COLORMAP_UNIFORM,
176        _ => "",
177    };
178
179    let output_val = match data_format {
180        Some(SegmentData::Scalar(..)) => "@location(3) val: f32,",
181        Some(SegmentData::Color(_)) => "@location(3) val: vec3<f32>",
182        None => "",
183    };
184
185    let input_val = match data_format {
186        Some(_) => "data: DataInput,",
187        None => "",
188    };
189
190    let set_output = match data_format {
191        Some(_) => "out.val = data.val_1 * (1. - t) + t * data.val_2;",
192        None => "",
193    };
194
195    let color_output = match data_format {
196        Some(SegmentData::Scalar(..)) => "let lambertian = colormap(in.val);",
197        Some(SegmentData::Color(_)) => "let lambertian = in.val;",
198        None => "let lambertian = settings.color;",
199    };
200
201    format!(
202        SHADER!(),
203        data_struct, uniform, output_val, input_val, set_output, color_output,
204    )
205}
