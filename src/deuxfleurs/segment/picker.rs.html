86impl ElementPicker for Picker {
87    type Geometry = SegmentGeometry;
88    type Settings = PCSettings;
89    fn new(
90        geometry: &Self::Geometry,
91        settings: &Self::Settings,
92        transform: &TransformSettings,
93        device: &wgpu::Device,
94        camera_light_bind_group_layout: &wgpu::BindGroupLayout,
95        counter_bind_group_layout: &wgpu::BindGroupLayout,
96    ) -> Self {
97        let s2 = 1.;
98        let positions = [
99            [-s2, -s2, 0.],
100            [s2, -s2, 0.],
101            [-s2, s2, 0.],
102            [s2, -s2, 0.],
103            [s2, s2, 0.],
104            [-s2, s2, 0.],
105        ];
106        let vertices = positions.map(|position| SphereVertex { position });
107        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
108            label: Some("PC Vertex Buffer"),
109            contents: bytemuck::cast_slice(&vertices),
110            usage: wgpu::BufferUsages::VERTEX,
111        });
112        let mut gpu_vertices = Vec::with_capacity(geometry.positions.len());
113        for (i, position) in geometry.positions.iter().enumerate() {
114            let vertex = VertexData {
115                center: *position,
116                index: i as u32,
117            };
118            gpu_vertices.push(vertex);
119        }
120        let center_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
121            label: Some("Curve Sphere Center Buffer"),
122            contents: bytemuck::cast_slice(&gpu_vertices),
123            usage: wgpu::BufferUsages::VERTEX,
124        });
125        let mut gpu_vertices2 = Vec::with_capacity(geometry.connections.len());
126        for (i, connection) in geometry.connections.iter().enumerate() {
127            let vertex = CylinderData {
128                position_1: geometry.positions[connection[0] as usize],
129                position_2: geometry.positions[connection[1] as usize],
130                index: (i + geometry.positions.len()) as u32,
131            };
132            gpu_vertices2.push(vertex);
133        }
134        let cylinder_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
135            label: Some("Curve Cylinder Buffer"),
136            contents: bytemuck::cast_slice(&gpu_vertices2),
137            usage: wgpu::BufferUsages::VERTEX,
138        });
139        let transform_raw = transform.to_raw();
140        let transform_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
141            label: Some("Picker Transform Buffer"),
142            contents: bytemuck::cast_slice(&[transform_raw]),
143            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
144        });
145        let settings_uniform = settings.build_uniform(device).unwrap();
146        let transform_bind_group_layout =
147            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
148                entries: &[wgpu::BindGroupLayoutEntry {
149                    binding: 0,
150                    visibility: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
151                    ty: wgpu::BindingType::Buffer {
152                        ty: wgpu::BufferBindingType::Uniform,
153                        has_dynamic_offset: false,
154                        min_binding_size: None,
155                    },
156                    count: None,
157                }],
158                label: Some("transform_bind_group_layout"),
159            });
160
161        let transform_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
162            layout: &transform_bind_group_layout,
163            entries: &[wgpu::BindGroupEntry {
164                binding: 0,
165                resource: transform_buffer.as_entire_binding(),
166            }],
167            label: Some("transform_bind_group"),
168        });
169
170        let render_pipeline_layout =
171            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
172                label: Some("Curve Picker Pipeline Layout"),
173                bind_group_layouts: &[
174                    camera_light_bind_group_layout,
175                    counter_bind_group_layout,
176                    &transform_bind_group_layout,
177                    &settings_uniform.bind_group_layout,
178                ],
179                push_constant_ranges: &[],
180            });
181        let shader = wgpu::ShaderModuleDescriptor {
182            label: Some("Curve Sphere Picker Shader"),
183            source: wgpu::ShaderSource::Wgsl(SPHERE_PICKER_SHADER.into()),
184        };
185        let sphere_render_pipeline = create_picker_pipeline(
186            device,
187            &render_pipeline_layout,
188            texture::Texture::PICKER_FORMAT,
189            Some(texture::Texture::DEPTH_FORMAT),
190            &[SphereVertex::desc(), VertexData::desc()],
191            shader,
192            Some("Curve Sphere picker"),
193        );
194        let cyl_shader = wgpu::ShaderModuleDescriptor {
195            label: Some("Curve Sphere Picker Shader"),
196            source: wgpu::ShaderSource::Wgsl(CYLINDER_PICKER_SHADER.into()),
197        };
198        let cyl_render_pipeline = create_picker_pipeline(
199            device,
200            &render_pipeline_layout,
201            texture::Texture::PICKER_FORMAT,
202            Some(texture::Texture::DEPTH_FORMAT),
203            &[SphereVertex::desc(), CylinderData::desc()],
204            cyl_shader,
205            Some("Curve Sphere picker"),
206        );
207        let num_elements = geometry.positions.len() as u32;
208        let cyl_elements = geometry.connections.len() as u32;
209        Self {
210            cyl_elements,
211            num_elements,
212            vertex_buffer,
213            center_buffer,
214            cylinder_buffer,
215            transform_bind_group,
216            transform_buffer,
217            sphere_render_pipeline,
218            cyl_render_pipeline,
219            settings_uniform,
220        }
221    }
222
223    fn update_transform(&self, queue: &mut wgpu::Queue, transform: &TransformSettings) {
224        let transform_raw = transform.to_raw();
225        queue.write_buffer(
226            &self.transform_buffer,
227            0,
228            bytemuck::cast_slice(&[transform_raw]),
229        );
230    }
231
232    fn update_settings(&self, queue: &mut wgpu::Queue, settings: &Self::Settings) {

234    }

237        self.num_elements + self.cyl_elements
238    }

241impl Render for Picker {
242    fn render<'a, 'b>(&'a self, render_pass: &mut wgpu::RenderPass<'b>)
243    where
244        'a: 'b,
245    {
246        render_pass.set_bind_group(2, &self.transform_bind_group, &[]);
247        render_pass.set_bind_group(3, &self.settings_uniform.bind_group, &[]);
248        render_pass.set_pipeline(&self.sphere_render_pipeline);
249        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
250        render_pass.set_vertex_buffer(1, self.center_buffer.slice(..));
251        render_pass.draw(0..6, 0..(self.num_elements));
252
253        render_pass.set_pipeline(&self.cyl_render_pipeline);
254        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
255        render_pass.set_vertex_buffer(1, self.cylinder_buffer.slice(..));
256        render_pass.draw(0..6, 0..(self.cyl_elements));
257    }

340// function from :
341// https://iquilezles.org/articles/intersectors/
342fn dot2(v: vec3<f32>) -> f32 { return dot(v, v); }

345fn sphIntersect( ro: vec3<f32>, rd: vec3<f32>, ce: vec3<f32>, ra: f32 ) -> vec2<f32>
346{
347    let oc = ro - ce;
348    let b = dot( oc, rd );
349    let c = dot( oc, oc ) - ra*ra;
350    var h = b*b - c;
351    if( h<0.0 ) { return vec2<f32>(-1.0); } // no intersection
352    h = sqrt( h );
353    return vec2<f32>( -b-h, -b+h );
354}

356struct FragOutput {
357    @builtin(frag_depth) depth: f32,
358    @location(0) color: vec4<f32>,
359}
360
361@fragment
362fn fs_main(in: VertexOutput) -> FragOutput {
363    let ro = camera.view_pos.xyz;
364	let rd = normalize(in.world_pos - camera.view_pos.xyz);
365    let ce = in.center;
366    let det = determinant(transform.normal);
367    let r = settings.radius / pow(det, 1. / 3.);
368
369    var out: FragOutput;
370
371    let t = sphIntersect( ro, rd, ce, r);
372    if(t.x < 0.0) {
373        discard;
374    }
375	let pos = ro + t.x * rd;
376
377	let clip_space_pos = camera.view_proj * vec4<f32>(pos, 1.);
378	out.depth = clip_space_pos.z / clip_space_pos.w;
379    let res =   counter.count + in.index;
380    // webgl dosen't support rendering to u32, so we have to resort to this
381    let f1 = f32((res >> u32(24))) / 255.;
382    let f2 = f32(((res << u32(8)) >> u32(24))) / 255.;
383    let f3 = f32(((res << u32(16)) >> u32(24))) / 255.;
384    let f4 = f32(((res << u32(24)) >> u32(24))) / 255.;
385    out.color = vec4<f32>(f4, f3, f2, f1);
386    //return bitcast<vec4<f32>>(res);
387	return out;
388}

474// cylinder defined by extremes a and b, and radious ra
475fn cylIntersect( ro: vec3<f32>, rd: vec3<f32>, a: vec3<f32>, b: vec3<f32>, ra: f32 ) -> vec4<f32>
476{
477    let ba = b  - a;
478    let oc = ro - a;
479    let baba = dot(ba,ba);
480    let bard = dot(ba,rd);
481    let baoc = dot(ba,oc);
482    let k2 = baba            - bard*bard;
483    let k1 = baba*dot(oc,rd) - baoc*bard;
484    let k0 = baba*dot(oc,oc) - baoc*baoc - ra*ra*baba;
485    var h = k1*k1 - k2*k0;
486    if( h<0.0 ) {{ return vec4(-1.0); }}//no intersection
487    h = sqrt(h);
488    let t = (-k1-h)/k2;
489    // body
490    let y = baoc + t*bard;
491    if( y>0.0 && y<baba ) {{ return vec4( t, (oc+t*rd - ba*y/baba)/ra ); }}
492    return vec4(-1.0);//no intersection
493}
494
495// normal at sphere p of cylinder (a,b,ra), see above
496fn cylNormal( p: vec3<f32>, a: vec3<f32>, b: vec3<f32>, ra: f32 ) -> vec3<f32>
497{
498    let pa = p - a;
499    let ba = b - a;
500    let baba = dot(ba,ba);
501    let paba = dot(pa,ba);
502    let h = dot(pa,ba)/baba;
503    return (pa - ba*h)/ra;
504}
