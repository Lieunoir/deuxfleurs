6// Camera informations for easier updating
7#[derive(Serialize, Deserialize)]
8pub struct Camera {
9    eye: cgmath::Point3<f32>,
10    target: cgmath::Point3<f32>,
11    up: cgmath::Vector3<f32>,
12    #[serde(skip)]
13    aspect: f32,
14    #[serde(skip)]
15    fovy: f32,
16    #[serde(skip)]
17    znear: f32,
18    #[serde(skip)]
19    zfar: f32,
20}
21
22impl Camera {
23    pub fn new(aspect: f32) -> Self {
24        Self {
25            eye: (0.0, 0.0, -2.5).into(),
26            target: (0.0, 0.0, 0.0).into(),
27            up: cgmath::Vector3::unit_y(),
28            aspect,
29            fovy: 45.0,
30            znear: 0.01,
31            zfar: 10.0,
32        }
33    }
34
35    pub fn build_view_projection_matrix(&self) -> cgmath::Matrix4<f32> {
36        let view = cgmath::Matrix4::look_at_rh(self.eye, self.target, self.up);
37        let proj = cgmath::perspective(cgmath::Deg(self.fovy), self.aspect, self.znear, self.zfar);
38        proj * view
39    }
40
41    pub fn build_view(&self) -> cgmath::Matrix4<f32> {
42        cgmath::Matrix4::look_at_rh(self.eye, self.target, self.up)
43    }
44
45    pub fn build_proj(&self) -> cgmath::Matrix4<f32> {
46        cgmath::perspective(cgmath::Deg(self.fovy), self.aspect, self.znear, self.zfar)
47    }
48
49    pub fn set_scene_size(&mut self, size: f32, center: cgmath::Point3<f32>) {
50        if size > 0. {
51            let dir = cgmath::Vector3::<f32>::new(0., 0., -3.);
52            self.eye = center + dir * size;
53            self.target = center;
54            self.znear = size * 0.01;
55            self.zfar = size * 10.;
56            self.up = cgmath::Vector3::unit_y();
57        }
58    }
59
60    pub fn resize(&mut self, width: u32, height: u32) {
61        self.aspect = width as f32 / height as f32;
62    }
63
64    pub fn get_position(&self) -> [f32; 3] {
65        self.eye.into()
66    }
67
68    pub fn copy(&self) -> Result<String, ()> {
69        serde_json::to_string(self).map_err(|_e| ())
70    }
71
72    pub fn set(&mut self, new_camera: String) {
73        if let Ok(new_c) = serde_json::from_str::<Camera>(&new_camera) {
74            self.eye = new_c.eye;
75            self.target = new_c.target;
76            self.up = new_c.up;
77        }
78    }
79}

84pub struct CameraUniform {
85    view_position: [f32; 4],
86    view_proj: [[f32; 4]; 4],
87    view_inv: [[f32; 4]; 4],
88    floor_bb: [f32; 4],
89    floor_proj: [[f32; 4]; 4],
90}

92impl CameraUniform {
93    pub fn new() -> Self {
94        Self {
95            view_position: [0.0; 4],
96            view_proj: cgmath::Matrix4::identity().into(),
97            view_inv: cgmath::Matrix4::identity().into(),
98            floor_bb: [0.0; 4],
99            floor_proj: cgmath::Matrix4::identity().into(),
100        }
101    }
102
103    pub fn update_view_proj(&mut self, camera: &Camera, sbv: &SBV, level: f32) {
104        // We're using Vector4 because ofthe camera_uniform 16 byte spacing requirement
105        self.view_position = camera.eye.to_homogeneous().into();
106        let view_proj = camera.build_view_projection_matrix();
107        self.view_proj = view_proj.into();
108        let view_inv = view_proj.inverse_transform().unwrap();
109        self.view_inv = view_inv.into();
110        //let orig : cgmath::Vector4<f32> = self.view_position.into();
111        let mut min_x = f32::MAX;
112        let mut min_z = f32::MAX;
113        let mut max_x = f32::MIN;
114        let mut max_z = f32::MIN;
115        let couples = [(-1., -1.), (-1., 1.), (1., -1.), (1., 1.)];
116        for (x, y) in couples {
117            let mut target = view_inv * cgmath::Vector4::new(x, y, 1., 1.);
118            target = target / target.w;
119            let mut origin = view_inv * cgmath::Vector4::new(x, y, 0., 1.);
120            origin = origin / origin.w;
121            let ray = target - origin;
122            if ray.y.abs() > 10e-9 {
123                let t = (level - origin.y) / ray.y;
124                if t <= 1. && t >= 0. {
125                    let pos = origin + t * ray;
126                    if pos.x < min_x {
127                        min_x = pos.x;
128                    }
129                    if pos.x > max_x {
130                        max_x = pos.x;
131                    }
132                    if pos.z < min_z {
133                        min_z = pos.z;
134                    }
135                    if pos.z > max_z {
136                        max_z = pos.z;
137                    }
138                }
139            }
140        }
141        let couples = [(-1., 0.), (-1., 1.), (1., 0.), (1., 1.)];
142        for (x, z) in couples {
143            let mut target = view_inv * cgmath::Vector4::new(x, -1., z, 1.);
144            target = target / target.w;
145            let mut origin = view_inv * cgmath::Vector4::new(x, 1., z, 1.);
146            origin = origin / origin.w;
147            let ray = target - origin;
148            if ray.y.abs() > 10e-9 {
149                let t = (level - origin.y) / ray.y;
150                if t <= 1. && t >= 0. {
151                    let pos = origin + t * ray;
152                    if pos.x < min_x {
153                        min_x = pos.x;
154                    }
155                    if pos.x > max_x {
156                        max_x = pos.x;
157                    }
158                    if pos.z < min_z {
159                        min_z = pos.z;
160                    }
161                    if pos.z > max_z {
162                        max_z = pos.z;
163                    }
164                }
165            }
166        }
167        let couples = [(-1., 0.), (-1., 1.), (1., 0.), (1., 1.)];
168        for (y, z) in couples {
169            let mut target = view_inv * cgmath::Vector4::new(-1., y, z, 1.);
170            target = target / target.w;
171            let mut origin = view_inv * cgmath::Vector4::new(1., y, z, 1.);
172            origin = origin / origin.w;
173            let ray = target - origin;
174            if ray.y.abs() > 10e-9 {
175                let t = (level - origin.y) / ray.y;
176                if t <= 1. && t >= 0. {
177                    let pos = origin + t * ray;
178                    if pos.x < min_x {
179                        min_x = pos.x;
180                    }
181                    if pos.x > max_x {
182                        max_x = pos.x;
183                    }
184                    if pos.z < min_z {
185                        min_z = pos.z;
186                    }
187                    if pos.z > max_z {
188                        max_z = pos.z;
189                    }
190                }
191            }
192        }
193        let sbv_bb = sbv.get_bb();
194        min_x = min_x.max(sbv_bb[0]);
195        max_x = max_x.min(sbv_bb[1]);
196        min_z = min_z.max(sbv_bb[2]);
197        max_z = max_z.min(sbv_bb[3]);
198        let c_x = 0.5 * (min_x + max_x);
199        let c_z = 0.5 * (min_z + max_z);
200        let d_x = 0.5 * (max_x - min_x);
201        let d_z = 0.5 * (max_z - min_z);
202        let eye = cgmath::Point3::<f32>::new(c_x, camera.zfar, c_z);
203        let target = cgmath::Point3::<f32>::new(c_x, 0., c_z);
204        let up = cgmath::Vector3::<f32>::new(0., 0., 1.);
205        let view = cgmath::Matrix4::look_at_rh(eye, target, up);
206        let proj = cgmath::ortho(d_x, -d_x, -d_z, d_z, -camera.zfar, 2. * camera.zfar);
207        self.floor_bb = [min_x, min_z, max_x - min_x, max_z - min_z];
208        self.floor_proj = (proj * view).into();
209    }
210}

223        Self {

232    pub fn process_events(&mut self, event: &WindowEvent, ui_hovered: bool) -> bool {
233        match event {
234            WindowEvent::CursorMoved { position, .. } if self.prev_mouse.is_none() => {
235                self.prev_mouse = Some(*position);
236                true
237            }
238            WindowEvent::CursorMoved { position, .. } => {
239                let prev = self.prev_mouse.unwrap();
240                self.pan_delta = Some(((position.x - prev.x) as f32, (position.y - prev.y) as f32));
241                self.prev_mouse = Some(*position);
242                if self.is_mouse_left_pressed || self.is_mouse_right_pressed {
243                    true
244                } else {
245                    false
246                }
247            }
248            WindowEvent::MouseInput { state, button, .. } => {
249                if !ui_hovered || *state != ElementState::Pressed {
250                    if *button == MouseButton::Left {
251                        self.is_mouse_left_pressed = *state == ElementState::Pressed;
252                    } else if *button == MouseButton::Right {
253                        self.is_mouse_right_pressed = *state == ElementState::Pressed;
254                    }
255                }
256                false
257            }
258            WindowEvent::MouseWheel { delta, .. } => {
259                if !ui_hovered {
260                    self.wheel_delta = Some(match delta {
261                        MouseScrollDelta::LineDelta(_, y) => *y,
262                        MouseScrollDelta::PixelDelta(p) => 0.02 * p.y as f32,
263                    });
264                    true
265                } else {
266                    false
267                }
268            }
269            _ => false,
270        }
271    }

273    pub fn update_camera(&mut self, camera: &mut Camera, settings: &Settings) {
274        let forward = camera.target - camera.eye;
275        let forward_norm = forward.normalize();
276
277        if let Some(delta) = self.wheel_delta {
278            camera.eye += forward_norm * delta * camera.zfar * 0.2 * settings.zoom_sensitivity;
279            camera.target += forward_norm * delta * camera.zfar * 0.2 * settings.zoom_sensitivity;
280            self.wheel_delta = None;
281        }
282
283        let right = forward_norm.cross(camera.up);
284        if let Some((dx, dy)) = self.pan_delta {
285            if self.is_mouse_left_pressed {
286                let origin = cgmath::Point3::<f32> {
287                    x: 0.,
288                    y: 0.,
289                    z: 0.,
290                };
291                let eye_norm = (camera.eye - cgmath::point3::<f32>(0., 0., 0.)).normalize();
292                let eye_mag = (camera.eye - cgmath::point3::<f32>(0., 0., 0.)).magnitude();
293                let center_right = eye_norm.cross(camera.up);
294                let old_eye = camera.eye;
295                camera.eye = cgmath::point3::<f32>(0., 0., 0.)
296                    + (eye_norm
297                        + center_right * dx * 0.2 * settings.mouse_sensitivity
298                        + camera.up * dy * 0.2 * settings.mouse_sensitivity)
299                        .normalize()
300                        * eye_mag;
301                let rotation =
302                    cgmath::Quaternion::between_vectors(old_eye - origin, camera.eye - origin);
303                camera.target = origin + rotation.rotate_vector(camera.target - origin);
304                camera.up =
305                    rotation.rotate_vector(old_eye + camera.up - origin) - (camera.eye - origin);
306                //camera.target = camera.eye + forward;
307            } else if self.is_mouse_right_pressed {
308                camera.eye += camera.zfar
309                    * 0.006
310                    * settings.mouse_sensitivity
311                    * (-dx * right + dy * camera.up);
312                camera.target += camera.zfar
313                    * 0.006
314                    * settings.mouse_sensitivity
315                    * (-dx * right + dy * camera.up);
316            }
317            self.pan_delta = None;
318        }
319
320        // Redo radius calc in case the up/ down is pressed.
321        /*

327            // that it doesn't change. The eye therefore still
328            // lies on the circle made by the target and eye.
329            camera.eye = camera.target - (forward + right * self.speed).normalize() * forward_mag;
330        }
331        if self.is_left_pressed {
