156    pub fn input(&mut self, event: &WindowEvent) -> bool {
157        match event {
158            WindowEvent::CursorMoved { position, .. } => {
159                self.cur_pos = (position.x as f32, position.y as f32);
160                let dx = self.cur_pos.0 - self.orig_pos.0;
161                let dy = self.cur_pos.1 - self.orig_pos.1;
162                if dx * dx + dy * dy > 5. {
163                    self.dragging = true;
164                }
165                false
166            }
167            WindowEvent::MouseInput { state, button, .. } => {
168                if *button == MouseButton::Left {
169                    if *state == ElementState::Pressed {
170                        self.dragging = false;
171                        self.orig_pos = self.cur_pos;
172                        false
173                    } else if !self.dragging {
174                        self.item_to_pick =
175                            Some((self.cur_pos.0 as usize, self.cur_pos.1 as usize));
176                        true
177                    } else {
178                        false
179                    }
180                } else {
181                    false
182                }
183            }
184            _ => false,
185        }
186    }

188    pub fn render(
189        &mut self,
190        device: &wgpu::Device,
191        encoder: &mut wgpu::CommandEncoder,
192        depth_texture_view: &wgpu::TextureView,
193        camera_light_bind_group: &wgpu::BindGroup,
194        surfaces: &IndexMap<String, Surface>,
195        clouds: &IndexMap<String, PointCloud>,
196        curves: &IndexMap<String, Segment>,
197    ) -> bool {
198        if !self.pick_locked && self.item_to_pick.is_some() {
199            {
200                let tex_view = &self.texture_view;
201                let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
202                    label: Some("Picker Render Pass"),
203                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
204                        view: tex_view,
205                        resolve_target: None,
206                        ops: wgpu::Operations {
207                            // Set the clear color during redraw
208                            // This is basically a background color applied if an object isn't taking up space
209
210                            // A standard clear color
211                            load: wgpu::LoadOp::Clear(wgpu::Color {
212                                r: 0.0,
213                                g: 0.0,
214                                b: 0.0,
215                                a: 0.0,
216                            }),
217                            store: wgpu::StoreOp::Store,
218                        },
219                    })],
220                    // Create a depth stencil buffer using the depth texture
221                    depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
222                        view: depth_texture_view,
223                        depth_ops: Some(wgpu::Operations {
224                            load: wgpu::LoadOp::Clear(1.0),
225                            store: wgpu::StoreOp::Store,
226                        }),
227                        stencil_ops: None,
228                    }),
229                    occlusion_query_set: None,
230                    timestamp_writes: None,
231                });
232                render_pass.set_bind_group(0, camera_light_bind_group, &[]);
233
234                if self.counters_dirty {
235                    let mut counter = 1;
236                    self.bind_groups = surfaces
237                        .values()
238                        .map(|surface| {
239                            let counter_uniform = CounterUniform {
240                                count: counter,
241                                _padding_1: 0,
242                                _padding_2: 0,
243                                _padding_3: 0,
244                            };
245                            counter += surface.inner.get_total_elements();
246
247                            //TODO use one dynamic buffer instead
248                            let counter_buffer =
249                                device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
250                                    label: Some("counter buffer"),
251                                    contents: bytemuck::cast_slice(&[counter_uniform]),
252                                    usage: wgpu::BufferUsages::UNIFORM
253                                        | wgpu::BufferUsages::COPY_DST,
254                                });
255                            let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
256                                layout: &self.bind_group_layout,
257                                entries: &[wgpu::BindGroupEntry {
258                                    binding: 0,
259                                    resource: counter_buffer.as_entire_binding(),
260                                }],
261                                label: Some("camera_light_bind_group"),
262                            });
263                            bind_group
264                        })
265                        .collect();
266
267                    for cloud in clouds.values() {
268                        let counter_uniform = CounterUniform {
269                            count: counter,
270                            _padding_1: 0,
271                            _padding_2: 0,
272                            _padding_3: 0,
273                        };
274                        counter += cloud.inner.get_total_elements();
275
276                        let counter_buffer =
277                            device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
278                                label: Some("counter buffer"),
279                                contents: bytemuck::cast_slice(&[counter_uniform]),
280                                usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
281                            });
282                        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
283                            layout: &self.bind_group_layout,
284                            entries: &[wgpu::BindGroupEntry {
285                                binding: 0,
286                                resource: counter_buffer.as_entire_binding(),
287                            }],
288                            label: Some("camera_light_bind_group"),
289                        });
290                        self.bind_groups.push(bind_group);
291                    }
292
293                    for curve in curves.values() {
294                        let counter_uniform = CounterUniform {
295                            count: counter,
296                            _padding_1: 0,
297                            _padding_2: 0,
298                            _padding_3: 0,
299                        };
300                        counter += curve.inner.get_total_elements();
301
302                        let counter_buffer =
303                            device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
304                                label: Some("counter buffer"),
305                                contents: bytemuck::cast_slice(&[counter_uniform]),
306                                usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
307                            });
308                        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
309                            layout: &self.bind_group_layout,
310                            entries: &[wgpu::BindGroupEntry {
311                                binding: 0,
312                                resource: counter_buffer.as_entire_binding(),
313                            }],
314                            label: Some("camera_light_bind_group"),
315                        });
316                        self.bind_groups.push(bind_group);
317                    }
318
319                    self.counters_dirty = false;
320                }
321
322                let mut index = 0;
323                for surface in surfaces.values() {
324                    let counter_bind_group = &self.bind_groups[index];
325                    index += 1;
326                    render_pass.set_bind_group(1, counter_bind_group, &[]);
327                    surface.inner.render_picker(&mut render_pass);
328                }
329                for cloud in clouds.values() {
330                    let counter_bind_group = &self.bind_groups[index];
331                    index += 1;
332                    render_pass.set_bind_group(1, counter_bind_group, &[]);
333                    cloud.inner.render_picker(&mut render_pass);
334                }
335                for curve in curves.values() {
336                    let counter_bind_group = &self.bind_groups[index];
337                    index += 1;
338                    render_pass.set_bind_group(1, counter_bind_group, &[]);
339                    curve.inner.render_picker(&mut render_pass);
340                }
341            }
342            {
343                encoder.copy_texture_to_buffer(
344                    wgpu::ImageCopyTexture {
345                        aspect: wgpu::TextureAspect::All,
346                        texture: &self.texture,
347                        mip_level: 0,
348                        origin: wgpu::Origin3d::ZERO,
349                    },
350                    wgpu::ImageCopyBuffer {
351                        buffer: &self.buffer,
352                        layout: wgpu::ImageDataLayout {
353                            offset: 0,
354                            bytes_per_row: Some(self.buffer_dimensions.padded_bytes_per_row as u32),
355                            //rows_per_image: std::num::NonZeroU32::new(self.size.height),
356                            rows_per_image: None,
357                        },
358                    },
359                    wgpu::Extent3d {
360                        width: self.buffer_dimensions.width as u32,
361                        height: self.buffer_dimensions.height as u32,
362                        depth_or_array_layers: 1,
363                    },
364                );
365            }
366            true
367        } else {
368            false
369        }
370    }

372    pub fn post_render(&mut self, event_loop_proxy: &winit::event_loop::EventLoopProxy<UserEvent>) {
373        if !self.pick_locked && self.item_to_pick.is_some() {
374            let buffer_slice = self.buffer.slice(..);
375            let event_loop_proxy = event_loop_proxy.clone();
376
377            #[cfg(not(target_arch = "wasm32"))]
378            {
379                buffer_slice.map_async(wgpu::MapMode::Read, move |_v| {
380                    event_loop_proxy.send_event(UserEvent::Pick).ok();
381                });
382            }
383            #[cfg(target_arch = "wasm32")]
384            {
385                // `event_loop_proxy` isn't `Send` so we have to find another way
386                let (sender, receiver) = oneshot::channel();
387                buffer_slice.map_async(wgpu::MapMode::Read, move |_v| {
388                    sender.send(()).unwrap();
389                });
390                wasm_bindgen_futures::spawn_local(async move {
391                    if let Ok(()) = receiver.await {
392                        event_loop_proxy.send_event(UserEvent::Pick).ok();
393                    }
394                });
395            }
396            //buffer_slice = Some(buffer_slice_t);
397            self.pick_locked = true;
398        }
399    }

401    pub fn pick(
402        &mut self,
403        surfaces: &IndexMap<String, Surface>,
404        clouds: &IndexMap<String, PointCloud>,
405        curves: &IndexMap<String, Segment>,
406        camera: &Camera,
407    ) {
408        {
409            let buffer_slice = self.buffer.slice(..);
410            let data = buffer_slice.get_mapped_range();
411            if let Some((i, j)) = self.item_to_pick {
412                let index = j * self.buffer_dimensions.padded_bytes_per_row + 4 * i;
413                let mut value = (data[index + 3] as u32) << 24
414                    | (data[index + 2] as u32) << 16
415                    | (data[index + 1] as u32) << 8
416                    | (data[index] as u32);
417                let mut c = 1;
418                if let Some(name) = surfaces
419                    .iter()
420                    .find(|(_key, surface)| {
421                        let found = c <= value && value < c + surface.inner.get_total_elements();
422                        if !found {
423                            c += surface.inner.get_total_elements();
424                        }
425                        found
426                    })
427                    .map(|(n, s)| {
428                        let transform = &s.inner.updater.transform;
429                        let pos_x = (i as f32 / self.width as f32) * 2. - 1.;
430                        let pos_y = -((j as f32 / self.height as f32) * 2. - 1.);
431                        let new_value = s.inner.picker.get_element(
432                            &s.inner.geometry,
433                            transform,
434                            camera,
435                            value - c,
436                            pos_x,
437                            pos_y,
438                        );
439                        value = new_value;
440                        c = 0;
441                        n
442                    })
443                    .or_else(|| {
444                        clouds
445                            .iter()
446                            .find(|(_key, cloud)| {
447                                let found =
448                                    c <= value && value < c + cloud.inner.get_total_elements();
449                                if !found {
450                                    c += cloud.inner.get_total_elements();
451                                }
452                                found
453                            })
454                            .map(|(n, _pc)| n)
455                    })
456                    .or_else(|| {
457                        curves
458                            .iter()
459                            .find(|(_key, curve)| {
460                                let found =
461                                    c <= value && value < c + curve.inner.get_total_elements();
462                                if !found {
463                                    c += curve.inner.get_total_elements();
464                                }
465                                found
466                            })
467                            .map(|(n, _pc)| n)
468                    })
469                {
470                    self.picked_item = Some((name.clone(), (value - c) as usize));
471                } else {
472                    self.picked_item = None;
473                }
474                self.item_to_pick = None;
475            }
476        }
477        self.buffer.unmap();
478        self.pick_locked = false;
479    }
