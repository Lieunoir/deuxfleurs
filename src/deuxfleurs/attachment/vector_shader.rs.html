104fn iCappedCone(ro: vec3<f32>, rd: vec3<f32>,
105                  pa: vec3<f32>, pb: vec3<f32>,
106                  ra: f32, rb: f32 ) -> vec4<f32>
107{
108    let  ba = pb - pa;
109    let  oa = ro - pa;
110    let  ob = ro - pb;
111
112    let m0 = dot(ba,ba);
113    let m1 = dot(oa,ba);
114    let m2 = dot(ob,ba);
115    let m3 = dot(rd,ba);
116
117    //caps
118         if( m1<0.0 ) { if( dot2(oa*m3-rd*m1)<(ra*ra*m3*m3) ) { return vec4<f32>(-m1/m3,-ba*inverseSqrt(m0)); } }
119    else if( m2>0.0 ) { if( dot2(ob*m3-rd*m2)<(rb*rb*m3*m3) ) { return vec4<f32>(-m2/m3, ba*inverseSqrt(m0)); } }
120
121    // body
122    let m4 = dot(rd,oa);
123    let m5 = dot(oa,oa);
124    let rr = ra - rb;
125    let hy = m0 + rr*rr;
126
127    let k2 = m0*m0    - m3*m3*hy;
128    let k1 = m0*m0*m4 - m1*m3*hy + m0*ra*(rr*m3*1.0        );
129    let k0 = m0*m0*m5 - m1*m1*hy + m0*ra*(rr*m1*2.0 - m0*ra);
130
131    let h = k1*k1 - k2*k0;
132    if( h<0.0 ) { return vec4(-1.0); }
133
134    let t = (-k1-sqrt(h))/k2;
135
136    let y = m1 + t*m3;
137    if( y>0.0 && y<m0 )
138    {
139        return vec4<f32>(t, normalize(m0*(m0*(oa+t*rd)+rr*ba*ra)-ba*hy*y));
140    }
141
142    return vec4<f32>(-1.0);
143}

145fn cylIntersect( ro: vec3<f32>, rd: vec3<f32>, pa: vec3<f32>, pb: vec3<f32>, ra: f32 ) -> vec4<f32>
146{
147    let ba = pb-pa;
148    let oc = ro - pa;
149    let baba = dot(ba,ba);
150    let bard = dot(ba,rd);
151    let baoc = dot(ba,oc);
152    let k2 = baba            - bard*bard;
153    let k1 = baba*dot(oc,rd) - baoc*bard;
154    let k0 = baba*dot(oc,oc) - baoc*baoc - ra*ra*baba;
155    var h = k1*k1 - k2*k0;
156    if( h<0.0 ) { return vec4<f32>(-1.0); }//no intersection
157    h = sqrt(h);
158    var t = (-k1-h)/k2;
159    // body
160    let y = baoc + t*bard;
161    if( y>0.0 && y<baba ) { return vec4<f32>( t, (oc+t*rd - ba*y/baba)/ra ); }
162    // caps
163	t = ( select(baba, 0., y< 0.) - baoc) / bard;
164    if( abs(k1+k2*t)<h )
165    {
166        return vec4<f32>( t, ba*sign(y)/sqrt(baba) );
167    }
168    return vec4<f32>(-1.0);//no intersection
169}
