39/// xcb connection errors because of socket, pipe and other stream errors.
40pub const XCB_CONN_ERROR: c_int = 1;
41/// xcb connection shutdown because of extension not supported
42pub const XCB_CONN_CLOSED_EXT_NOTSUPPORTED: c_int = 2;
43/// malloc(), calloc() and realloc() error upon failure, for eg ENOMEM
44pub const XCB_CONN_CLOSED_MEM_INSUFFICIENT: c_int = 3;
45/// Connection closed, exceeding request length that server accepts.
46pub const XCB_CONN_CLOSED_REQ_LEN_EXCEED: c_int = 4;
47/// Connection closed, error during parsing display string.
48pub const XCB_CONN_CLOSED_PARSE_ERR: c_int = 5;
49/// Connection closed because the server does not have a screen
50/// matching the display.
51pub const XCB_CONN_CLOSED_INVALID_SCREEN: c_int = 6;
52/// Connection closed because some FD passing operation failed
53pub const XCB_CONN_CLOSED_FDPASSING_FAILED: c_int = 7;

56/// XCB connection structure
57/// An opaque structure that contain all data that XCB needs to communicate
58/// with an X server.
59pub enum xcb_connection_t {}

76/// Generic reply
77#[derive(Copy, Clone)]
78#[repr(C)]
79pub struct xcb_generic_reply_t {
80    pub response_type: u8,
81    pad0:              u8,
82    pub sequence:      u16,
83    pub length:        u32
84}

86/// Generic event
87#[derive(Copy)]
88#[repr(C)]
89pub struct xcb_generic_event_t {
90    pub response_type: u8,
91    pub pad0:          u8,
92    pub sequence:      u16,
93    pub pad:           [u32; 7],
94    pub full_sequence: u32
95}
96impl Clone for xcb_generic_event_t {
97    fn clone(&self) -> xcb_generic_event_t { *self }
98}

100/// GE event
101///
102/// An event as sent by the XGE extension. The length field specifies the

105/// Deprecated Since some fields in this struct have unfortunate names, it is
106/// recommended to use xcb_ge_generic_event_t instead.

162/// Container for authorization information.
163/// A container for authorization information to be sent to the X server
164#[repr(C)]
165pub struct xcb_auth_info_t {
166    /// length of the string name (as returned by strlen)
167    pub namelen:    c_int,
168    /// String containing the authentication protocol name,
169    /// such as "MIT-MAGIC-COOKIE-1" or "XDM-AUTHORIZATION-1".
170    pub name:       *mut c_char,
171    /// length of the data member
172    pub datalen:    c_int,
173    /// data interpreted in a protocol specific manner
174    pub data:       *mut c_char
175}

180    /// Forces any buffered output to be written to the server. Blocks
181    /// until the write is complete.
182    ///
183    /// Return > 0 on success, <= 0 otherwise.
184    pub fn xcb_flush(c: *mut xcb_connection_t)
185            -> c_int;
186
187    /// Returns the maximum request length that this server accepts.
188    ///
189    /// In the absence of the BIG-REQUESTS extension, returns the
190    /// maximum request length field from the connection setup data, which
191    /// may be as much as 65535. If the server supports BIG-REQUESTS, then
192    /// the maximum request length field from the reply to the
193    /// BigRequestsEnable request will be returned instead.
194    ///
195    /// Note that this length is measured in four-byte units, making the
196    /// theoretical maximum lengths roughly 256kB without BIG-REQUESTS and
197    /// 16GB with.
198    ///
199    /// Returns The maximum request length field.
200    pub fn xcb_get_maximum_request_length(c: *mut xcb_connection_t)
201            -> u32;
202
203    /// Prefetch the maximum request length without blocking.
204    ///
205    /// Without blocking, does as much work as possible toward computing
206    /// the maximum request length accepted by the X server.
207    ///
208    /// Invoking this function may cause a call to xcb_big_requests_enable,
209    /// but will not block waiting for the reply.
210    /// xcb_get_maximum_request_length will return the prefetched data
211    /// after possibly blocking while the reply is retrieved.
212    ///
213    /// Note that in order for this function to be fully non-blocking, the
214    /// application must previously have called
215    /// xcb_prefetch_extension_data(c, &xcb_big_requests_id) and the reply
216    /// must have already arrived.
217    pub fn xcb_prefetch_maximum_request_length(c: *mut xcb_connection_t);
218
219    /// Returns the next event or error from the server.
220    ///
221    /// Returns the next event or error from the server, or returns null in
222    /// the event of an I/O error. Blocks until either an event or error
223    /// arrive, or an I/O error occurs.
224    pub fn xcb_wait_for_event(c: *mut xcb_connection_t)
225            -> *mut xcb_generic_event_t;
226
227    /// Returns the next event or error from the server.
228    ///
229    /// Returns the next event or error from the server, if one is
230    /// available, or returns @c NULL otherwise. If no event is available, that
231    /// might be because an I/O error like connection close occurred while
232    /// attempting to read the next event, in which case the connection is
233    /// shut down when this function returns.
234    pub fn xcb_poll_for_event(c: *mut xcb_connection_t)
235            -> *mut xcb_generic_event_t;
236
237    /// Returns the next event without reading from the connection.
238    ///
239    /// This is a version of xcb_poll_for_event that only examines the
240    /// event queue for new events. The function doesn't try to read new
241    /// events from the connection if no queued events are found.
242    ///
243    /// This function is useful for callers that know in advance that all
244    /// interesting events have already been read from the connection. For
245    /// example, callers might use xcb_wait_for_reply and be interested
246    /// only of events that preceded a specific reply.
247    pub fn xcb_poll_for_queued_event(c: *mut xcb_connection_t)
248            -> *mut xcb_generic_event_t;
249
250    /// Returns the next event from a special queue
251    pub fn xcb_poll_for_special_event(c: *mut xcb_connection_t,
252                                      se: *mut xcb_special_event_t)
253            -> *mut xcb_generic_event_t;
254
255    /// Returns the next event from a special queue, blocking until one arrives
256    pub fn xcb_wait_for_special_event(c: *mut xcb_connection_t,
257                                      se: *mut xcb_special_event_t)
258            -> *mut xcb_generic_event_t;

263                                        eid: u32,
264                                        stamp: *mut u32)

267    /// Stop listening for a special event
268    pub fn xcb_unregister_for_special_event(c: *mut xcb_connection_t,

271    /// Return the error for a request, or NULL if none can ever arrive.
272    ///
273    /// The xcb_void_cookie_t cookie supplied to this function must have resulted
274    /// from a call to xcb_[request_name]_checked().  This function will block
275    /// until one of two conditions happens.  If an error is received, it will be
276    /// returned.  If a reply to a subsequent request has already arrived, no error
277    /// can arrive for this request, so this function will return NULL.
278    ///
279    /// Note that this function will perform a sync if needed to ensure that the
280    /// sequence number will advance beyond that provided in cookie; this is a
281    /// convenience to avoid races in determining whether the sync is needed.
282    pub fn xcb_request_check(c: *mut xcb_connection_t,
283                             cookie: xcb_void_cookie_t)
284            -> *mut xcb_generic_error_t;
285
286    /// Discards the reply for a request.
287    ///
288    /// sequence is the request sequence number from a cookie.
289    ///
290    /// Discards the reply for a request. Additionally, any error generated
291    /// by the request is also discarded (unless it was an _unchecked request
292    /// and the error has already arrived).
293    ///
294    /// This function will not block even if the reply is not yet available.
295    ///
296    /// Note that the sequence really does have to come from an xcb cookie;
297    /// this function is not designed to operate on socket-handoff replies.
298    pub fn xcb_discard_reply(c: *mut xcb_connection_t,
299                             sequence: c_uint);
300
301    /// Discards the reply for a request, given by a 64bit sequence number
302    ///
303    /// sequence is the 64-bit sequence number as returned by xcb_send_request64().
304    ///
305    /// Discards the reply for a request. Additionally, any error generated
306    /// by the request is also discarded (unless it was an _unchecked request
307    /// and the error has already arrived).
308    ///
309    /// This function will not block even if the reply is not yet available.
310    ///
311    /// Note that the sequence really does have to come from xcb_send_request64();
312    /// the cookie sequence number is defined as "unsigned" int and therefore
313    /// not 64-bit on all platforms.
314    /// This function is not designed to operate on socket-handoff replies.
315    ///
316    /// Unlike its xcb_discard_reply() counterpart, the given sequence number is not
317    /// automatically "widened" to 64-bit.
318    ///
319    pub fn xcb_discard_reply64(c: *mut xcb_connection_t,
320                               sequence: u64);
321
322    /// Caches reply information from QueryExtension requests.
323    ///
324    /// This function is the primary interface to the "extension cache",
325    /// which caches reply information from QueryExtension
326    /// requests. Invoking this function may cause a call to
327    /// xcb_query_extension to retrieve extension information from the
328    /// server, and may block until extension data is received from the
329    /// server.
330    ///
331    /// The result must not be freed. This storage is managed by the cache
332    /// itself.
333    ///
334    /// Returns A pointer to the xcb_query_extension_reply_t for the extension.
335    pub fn xcb_get_extension_data(c: *mut xcb_connection_t,
336                                  ext: *mut xcb_extension_t)
337            -> *const xcb_query_extension_reply_t;
338
339    /// Prefetch of extension data into the extension cache
340    ///
341    /// This function allows a "prefetch" of extension data into the
342    /// extension cache. Invoking the function may cause a call to
343    /// xcb_query_extension, but will not block waiting for the
344    /// reply. xcb_get_extension_data will return the prefetched data after
345    /// possibly blocking while it is retrieved.
346    pub fn xcb_prefetch_extension_data(c: *mut xcb_connection_t,
347                                       ext: *mut xcb_extension_t);
348
349    /// Access the data returned by the server.
350    ///
351    /// Accessor for the data returned by the server when the xcb_connection_t
352    /// was initialized. This data includes
353    /// - the server's required format for images,
354    /// - a list of available visuals,
355    /// - a list of available screens,
356    /// - the server's maximum request length (in the absence of the
357    /// BIG-REQUESTS extension),
358    /// - and other assorted information.
359    ///
360    /// See the X protocol specification for more details.
361    ///
362    /// Returns A pointer to an xcb_setup_t structure.
363    /// The result must not be freed.
364    pub fn xcb_get_setup(c: *mut xcb_connection_t)
365            -> *const xcb_setup_t;
366
367    /// Access the file descriptor of the connection.
368    ///
369    /// Accessor for the file descriptor that was passed to the
370    /// xcb_connect_to_fd call that returned @p c.
371    ///
372    /// Returns The file descriptor.
373    pub fn xcb_get_file_descriptor(c: *mut xcb_connection_t)
374            -> c_int;
375
376    /// Test whether the connection has shut down due to a fatal error.
377    ///
378    /// Some errors that occur in the context of an xcb_connection_t
379    /// are unrecoverable. When such an error occurs, the
380    /// connection is shut down and further operations on the
381    /// xcb_connection_t have no effect, but memory will not be freed until
382    /// xcb_disconnect() is called on the xcb_connection_t.
383    ///
384    /// Returns XCB_CONN_ERROR, because of socket errors, pipe errors or other stream errors.
385    /// Returns XCB_CONN_CLOSED_EXT_NOTSUPPORTED, when extension not supported.
386    /// Returns XCB_CONN_CLOSED_MEM_INSUFFICIENT, when memory not available.
387    /// Returns XCB_CONN_CLOSED_REQ_LEN_EXCEED, exceeding request length that server accepts.
388    /// Returns XCB_CONN_CLOSED_PARSE_ERR, error during parsing display string.
389    /// Returns XCB_CONN_CLOSED_INVALID_SCREEN, because the server does not have a screen matching the display.
390    ///
391    /// Returns > 0 if the connection is in an error state; 0 otherwise.
392    pub fn xcb_connection_has_error(c: *mut xcb_connection_t)
393            -> c_int;
394
395    /// Connects to the X server.
396    ///
397    /// Connects to an X server, given the open socket @p fd and the
398    /// xcb_auth_info_t @p auth_info. The file descriptor @p fd is
399    /// bidirectionally connected to an X server. If the connection
400    /// should be unauthenticated, @p auth_info must be @c
401    /// NULL.
402    ///
403    /// Always returns a non-NULL pointer to a xcb_connection_t, even on failure.
404    /// Callers need to use xcb_connection_has_error() to check for failure.
405    /// When finished, use xcb_disconnect() to close the connection and free
406    /// the structure.
407    pub fn xcb_connect_to_fd(fd: c_int,
408                             auth_info: *mut xcb_auth_info_t)
409            -> *mut xcb_connection_t;
410
411    /// Closes the connection.
412    ///
413    /// Closes the file descriptor and frees all memory associated with the
414    /// connection @c c. If @p c is @c NULL, nothing is done.
415    pub fn xcb_disconnect(c: *mut xcb_connection_t);

417    /// Parses a display string name in the form documented by X(7x).
418    /// name: The name of the display.
419    /// host: A pointer to a malloc'd copy of the hostname.
420    /// display: A pointer to the display number.
421    /// screen: A pointer to the screen number.
422    ///
423    /// Parses the display string name display_name in the form
424    /// documented by X(7x). Has no side effects on failure. If
425    /// displayname is NULL or empty, it uses the environment
426    /// variable DISPLAY. hostp is a pointer to a newly allocated string
427    /// that contain the host name. displayp is set to the display
428    /// number and screenp to the preferred screen number. screenp
429    /// can be NULL. If displayname does not contain a screen number,
430    /// it is set to 0.
431    ///
432    /// Returns 0 on failure, non 0 otherwise.
433    pub fn xcb_parse_display(name: *const c_char,
434                             host: *mut *mut c_char,
435                             display: *mut c_int,
436                             screen: *mut c_int)
437            -> c_int;

439    /// Connects to the X server.
440    /// displayname: The name of the display.
441    /// screenp: A pointer to a preferred screen number.
442    /// Returns A newly allocated xcb_connection_t structure.
443    ///
444    /// Connects to the X server specified by displayname. If
445    /// displayname is NULL, uses the value of the DISPLAY environment
446    /// variable. If a particular screen on that server is preferred, the
447    /// int pointed to by screenp (if not NULL) will be set to that
448    /// screen; otherwise the screen will be set to 0.
449    ///
450    /// Always returns a non-NULL pointer to a xcb_connection_t, even on failure.
451    /// Callers need to use xcb_connection_has_error() to check for failure.
452    /// When finished, use xcb_disconnect() to close the connection and free
453    /// the structure.
454    pub fn xcb_connect(displayname: *const c_char,
455                       screenp: *mut c_int)
456            -> *mut xcb_connection_t;
