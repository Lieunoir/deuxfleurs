18/// A texture and a buffer for reading the rendered frame back to the cpu.
19/// The texture is required since [`wgpu::TextureUsages::COPY_DST`] is not an allowed
20/// flag for the surface texture on all platforms. This means that anytime we want to
21/// capture the frame, we first render it to this texture, and then we can copy it to
22/// both the surface texture and the buffer, from where we can pull it back to the cpu.
23struct CaptureState {
24    texture: wgpu::Texture,
25    buffer: wgpu::Buffer,
26    padding: BufferPadding,
27}
28
29impl CaptureState {
30    fn new(device: &Arc<wgpu::Device>, surface_texture: &wgpu::Texture) -> Self {
31        let texture = device.create_texture(&wgpu::TextureDescriptor {
32            label: Some("egui_screen_capture_texture"),
33            size: surface_texture.size(),
34            mip_level_count: surface_texture.mip_level_count(),
35            sample_count: surface_texture.sample_count(),
36            dimension: surface_texture.dimension(),
37            format: surface_texture.format(),
38            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::COPY_SRC,
39            view_formats: &[],
40        });
41
42        let padding = BufferPadding::new(surface_texture.width());
43
44        let buffer = device.create_buffer(&wgpu::BufferDescriptor {
45            label: Some("egui_screen_capture_buffer"),
46            size: (padding.padded_bytes_per_row * texture.height()) as u64,
47            usage: wgpu::BufferUsages::COPY_DST | wgpu::BufferUsages::MAP_READ,
48            mapped_at_creation: false,
49        });
50
51        Self {
52            texture,
53            buffer,
54            padding,
55        }
56    }
57}
58
59struct BufferPadding {
60    unpadded_bytes_per_row: u32,
61    padded_bytes_per_row: u32,
62}
63
64impl BufferPadding {
65    fn new(width: u32) -> Self {
66        let bytes_per_pixel = std::mem::size_of::<u32>() as u32;
67        let unpadded_bytes_per_row = width * bytes_per_pixel;
68        let padded_bytes_per_row =
69            wgpu::util::align_to(unpadded_bytes_per_row, wgpu::COPY_BYTES_PER_ROW_ALIGNMENT);
70        Self {
71            unpadded_bytes_per_row,
72            padded_bytes_per_row,
73        }
74    }
75}

148    fn configure_surface(
149        surface_state: &SurfaceState,
150        render_state: &RenderState,
151        config: &WgpuConfiguration,
152    ) {
153        crate::profile_function!();
154
155        let usage = if surface_state.supports_screenshot {
156            wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::COPY_DST
157        } else {
158            wgpu::TextureUsages::RENDER_ATTACHMENT
159        };
160
161        let width = surface_state.width;
162        let height = surface_state.height;
163
164        let mut surf_config = wgpu::SurfaceConfiguration {
165            usage,
166            format: render_state.target_format,
167            present_mode: config.present_mode,
168            alpha_mode: surface_state.alpha_mode,
169            view_formats: vec![render_state.target_format],
170            ..surface_state
171                .surface
172                .get_default_config(&render_state.adapter, width, height)
173                .expect("The surface isn't supported by this adapter")
174        };
175
176        if let Some(desired_maximum_frame_latency) = config.desired_maximum_frame_latency {
177            surf_config.desired_maximum_frame_latency = desired_maximum_frame_latency;
178        }
179
180        surface_state
181            .surface
182            .configure(&render_state.device, &surf_config);
183    }

185    /// Updates (or clears) the [`winit::window::Window`] associated with the [`Painter`]
186    ///
187    /// This creates a [`wgpu::Surface`] for the given Window (as well as initializing render
188    /// state if needed) that is used for egui rendering.
189    ///
190    /// This must be called before trying to render via
191    /// [`paint_and_update_textures`](Self::paint_and_update_textures)
192    ///
193    /// # Portability
194    ///
195    /// _In particular it's important to note that on Android a it's only possible to create
196    /// a window surface between `Resumed` and `Paused` lifecycle events, and Winit will panic on
197    /// attempts to query the raw window handle while paused._
198    ///
199    /// On Android [`set_window`](Self::set_window) should be called with `Some(window)` for each
200    /// `Resumed` event and `None` for each `Paused` event. Currently, on all other platforms
201    /// [`set_window`](Self::set_window) may be called with `Some(window)` as soon as you have a
202    /// valid [`winit::window::Window`].
203    ///
204    /// # Errors
205    /// If the provided wgpu configuration does not match an available device.
206    pub async fn set_window(
207        &mut self,
208        viewport_id: ViewportId,
209        window: Option<Arc<winit::window::Window>>,
210    ) -> Result<(), crate::WgpuError> {
211        crate::profile_scope!("Painter::set_window"); // profile_function gives bad names for async functions
212
213        if let Some(window) = window {
214            let size = window.inner_size();
215            if !self.surfaces.contains_key(&viewport_id) {
216                let surface = self.instance.create_surface(window)?;
217                self.add_surface(surface, viewport_id, size).await?;
218            }
219        } else {
220            log::warn!("No window - clearing all surfaces");
221            self.surfaces.clear();
222        }
223        Ok(())
224    }

226    /// Updates (or clears) the [`winit::window::Window`] associated with the [`Painter`] without taking ownership of the window.
227    ///
228    /// Like [`set_window`](Self::set_window) except:
229    ///
230    /// # Safety
231    /// The user is responsible for ensuring that the window is alive for as long as it is set.
232    pub async unsafe fn set_window_unsafe(
233        &mut self,
234        viewport_id: ViewportId,
235        window: Option<&winit::window::Window>,
236    ) -> Result<(), crate::WgpuError> {
237        crate::profile_scope!("Painter::set_window_unsafe"); // profile_function gives bad names for async functions
238
239        if let Some(window) = window {
240            let size = window.inner_size();
241            if !self.surfaces.contains_key(&viewport_id) {
242                let surface = unsafe {
243                    self.instance
244                        .create_surface_unsafe(wgpu::SurfaceTargetUnsafe::from_window(&window)?)?
245                };
246                self.add_surface(surface, viewport_id, size).await?;
247            }
248        } else {
249            log::warn!("No window - clearing all surfaces");
250            self.surfaces.clear();
251        }
252        Ok(())
253    }

255    async fn add_surface(
256        &mut self,
257        surface: wgpu::Surface<'static>,
258        viewport_id: ViewportId,
259        size: winit::dpi::PhysicalSize<u32>,
260    ) -> Result<(), crate::WgpuError> {
261        let render_state = if let Some(render_state) = &self.render_state {
262            render_state
263        } else {
264            let render_state = RenderState::create(
265                &self.configuration,
266                &self.instance,
267                &surface,
268                self.depth_format,
269                self.msaa_samples,
270                self.dithering,
271            )
272            .await?;
273            self.render_state.get_or_insert(render_state)
274        };
275        let alpha_mode = if self.support_transparent_backbuffer {
276            let supported_alpha_modes = surface.get_capabilities(&render_state.adapter).alpha_modes;
277
278            // Prefer pre multiplied over post multiplied!
279            if supported_alpha_modes.contains(&wgpu::CompositeAlphaMode::PreMultiplied) {
280                wgpu::CompositeAlphaMode::PreMultiplied
281            } else if supported_alpha_modes.contains(&wgpu::CompositeAlphaMode::PostMultiplied) {
282                wgpu::CompositeAlphaMode::PostMultiplied
283            } else {
284                log::warn!("Transparent window was requested, but the active wgpu surface does not support a `CompositeAlphaMode` with transparency.");
285                wgpu::CompositeAlphaMode::Auto
286            }
287        } else {
288            wgpu::CompositeAlphaMode::Auto
289        };
290        let supports_screenshot =
291            !matches!(render_state.adapter.get_info().backend, wgpu::Backend::Gl);
292        self.surfaces.insert(
293            viewport_id,
294            SurfaceState {
295                surface,
296                width: size.width,
297                height: size.height,
298                alpha_mode,
299                supports_screenshot,
300            },
301        );
302        let Some(width) = NonZeroU32::new(size.width) else {
303            log::debug!("The window width was zero; skipping generate textures");
304            return Ok(());
305        };
306        let Some(height) = NonZeroU32::new(size.height) else {
307            log::debug!("The window height was zero; skipping generate textures");
308            return Ok(());
309        };
310        self.resize_and_generate_depth_texture_view_and_msaa_view(viewport_id, width, height);
311        Ok(())
312    }

325    fn resize_and_generate_depth_texture_view_and_msaa_view(
326        &mut self,
327        viewport_id: ViewportId,
328        width_in_pixels: NonZeroU32,
329        height_in_pixels: NonZeroU32,
330    ) {
331        crate::profile_function!();
332
333        let width = width_in_pixels.get();
334        let height = height_in_pixels.get();
335
336        let render_state = self.render_state.as_ref().unwrap();
337        let surface_state = self.surfaces.get_mut(&viewport_id).unwrap();
338
339        surface_state.width = width;
340        surface_state.height = height;
341
342        Self::configure_surface(surface_state, render_state, &self.configuration);
343
344        if let Some(depth_format) = self.depth_format {
345            self.depth_texture_view.insert(
346                viewport_id,
347                render_state
348                    .device
349                    .create_texture(&wgpu::TextureDescriptor {
350                        label: Some("egui_depth_texture"),
351                        size: wgpu::Extent3d {
352                            width,
353                            height,
354                            depth_or_array_layers: 1,
355                        },
356                        mip_level_count: 1,
357                        sample_count: self.msaa_samples,
358                        dimension: wgpu::TextureDimension::D2,
359                        format: depth_format,
360                        usage: wgpu::TextureUsages::RENDER_ATTACHMENT
361                            | wgpu::TextureUsages::TEXTURE_BINDING,
362                        view_formats: &[depth_format],
363                    })
364                    .create_view(&wgpu::TextureViewDescriptor::default()),
365            );
366        }
367
368        if let Some(render_state) = (self.msaa_samples > 1)
369            .then_some(self.render_state.as_ref())
370            .flatten()
371        {
372            let texture_format = render_state.target_format;
373            self.msaa_texture_view.insert(
374                viewport_id,
375                render_state
376                    .device
377                    .create_texture(&wgpu::TextureDescriptor {
378                        label: Some("egui_msaa_texture"),
379                        size: wgpu::Extent3d {
380                            width,
381                            height,
382                            depth_or_array_layers: 1,
383                        },
384                        mip_level_count: 1,
385                        sample_count: self.msaa_samples,
386                        dimension: wgpu::TextureDimension::D2,
387                        format: texture_format,
388                        usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
389                        view_formats: &[texture_format],
390                    })
391                    .create_view(&wgpu::TextureViewDescriptor::default()),
392            );
393        };
394    }

415    // CaptureState only needs to be updated when the size of the two textures don't match and we want to
416    // capture a frame
417    fn update_capture_state(
418        screen_capture_state: &mut Option<CaptureState>,
419        surface_texture: &wgpu::SurfaceTexture,
420        render_state: &RenderState,
421    ) {
422        let surface_texture = &surface_texture.texture;
423        match screen_capture_state {
424            Some(capture_state) => {
425                if capture_state.texture.size() != surface_texture.size() {
426                    *capture_state = CaptureState::new(&render_state.device, surface_texture);
427                }
428            }
429            None => {
430                *screen_capture_state =
431                    Some(CaptureState::new(&render_state.device, surface_texture));
432            }
433        }
434    }

436    // Handles copying from the CaptureState texture to the surface texture and the cpu
437    fn read_screen_rgba(
438        screen_capture_state: &CaptureState,
439        render_state: &RenderState,
440        output_frame: &wgpu::SurfaceTexture,
441    ) -> Option<epaint::ColorImage> {
442        let CaptureState {
443            texture: tex,
444            buffer,
445            padding,
446        } = screen_capture_state;
447
448        let device = &render_state.device;
449        let queue = &render_state.queue;
450
451        let tex_extent = tex.size();
452
453        let mut encoder = device.create_command_encoder(&Default::default());
454        encoder.copy_texture_to_buffer(
455            tex.as_image_copy(),
456            wgpu::ImageCopyBuffer {
457                buffer,
458                layout: wgpu::ImageDataLayout {
459                    offset: 0,
460                    bytes_per_row: Some(padding.padded_bytes_per_row),
461                    rows_per_image: None,
462                },
463            },
464            tex_extent,
465        );
466
467        encoder.copy_texture_to_texture(
468            tex.as_image_copy(),
469            output_frame.texture.as_image_copy(),
470            tex.size(),
471        );
472
473        let id = queue.submit(Some(encoder.finish()));
474        let buffer_slice = buffer.slice(..);
475        let (sender, receiver) = std::sync::mpsc::channel();
476        buffer_slice.map_async(wgpu::MapMode::Read, move |v| {
477            drop(sender.send(v));
478        });
479        device.poll(wgpu::Maintain::WaitForSubmissionIndex(id));
480        receiver.recv().ok()?.ok()?;
481
482        let to_rgba = match tex.format() {
483            wgpu::TextureFormat::Rgba8Unorm => [0, 1, 2, 3],
484            wgpu::TextureFormat::Bgra8Unorm => [2, 1, 0, 3],
485            _ => {
486                log::error!("Screen can't be captured unless the surface format is Rgba8Unorm or Bgra8Unorm. Current surface format is {:?}", tex.format());
487                return None;
488            }
489        };
490
491        let mut pixels = Vec::with_capacity((tex.width() * tex.height()) as usize);
492        for padded_row in buffer_slice
493            .get_mapped_range()
494            .chunks(padding.padded_bytes_per_row as usize)
495        {
496            let row = &padded_row[..padding.unpadded_bytes_per_row as usize];
497            for color in row.chunks(4) {
498                pixels.push(epaint::Color32::from_rgba_premultiplied(
499                    color[to_rgba[0]],
500                    color[to_rgba[1]],
501                    color[to_rgba[2]],
502                    color[to_rgba[3]],
503                ));
504            }
505        }
506        buffer.unmap();
507
508        Some(epaint::ColorImage {
509            size: [tex.width() as usize, tex.height() as usize],
510            pixels,
511        })
512    }

514    /// Returns two things:
515    ///
516    /// The approximate number of seconds spent on vsync-waiting (if any),
517    /// and the captures captured screenshot if it was requested.
518    pub fn paint_and_update_textures(
519        &mut self,
520        viewport_id: ViewportId,
521        pixels_per_point: f32,
522        clear_color: [f32; 4],
523        clipped_primitives: &[epaint::ClippedPrimitive],
524        textures_delta: &epaint::textures::TexturesDelta,
525        capture: bool,
526    ) -> (f32, Option<epaint::ColorImage>) {
527        crate::profile_function!();
528
529        let mut vsync_sec = 0.0;
530
531        let Some(render_state) = self.render_state.as_mut() else {
532            return (vsync_sec, None);
533        };
534        let Some(surface_state) = self.surfaces.get(&viewport_id) else {
535            return (vsync_sec, None);
536        };
537
538        let mut encoder =
539            render_state
540                .device
541                .create_command_encoder(&wgpu::CommandEncoderDescriptor {
542                    label: Some("encoder"),
543                });
544
545        // Upload all resources for the GPU.
546        let screen_descriptor = renderer::ScreenDescriptor {
547            size_in_pixels: [surface_state.width, surface_state.height],
548            pixels_per_point,
549        };
550
551        let user_cmd_bufs = {
552            let mut renderer = render_state.renderer.write();
553            for (id, image_delta) in &textures_delta.set {
554                renderer.update_texture(
555                    &render_state.device,
556                    &render_state.queue,
557                    *id,
558                    image_delta,
559                );
560            }
561
562            renderer.update_buffers(
563                &render_state.device,
564                &render_state.queue,
565                &mut encoder,
566                clipped_primitives,
567                &screen_descriptor,
568            )
569        };
570
571        let capture = match (capture, surface_state.supports_screenshot) {
572            (false, _) => false,
573            (true, true) => true,
574            (true, false) => {
575                log::error!("The active render surface doesn't support taking screenshots.");
576                false
577            }
578        };
579
580        let output_frame = {
581            crate::profile_scope!("get_current_texture");
582            // This is what vsync-waiting happens on my Mac.
583            let start = web_time::Instant::now();
584            let output_frame = surface_state.surface.get_current_texture();
585            vsync_sec += start.elapsed().as_secs_f32();
586            output_frame
587        };
588
589        let output_frame = match output_frame {
590            Ok(frame) => frame,
591            Err(err) => match (*self.configuration.on_surface_error)(err) {
592                SurfaceErrorAction::RecreateSurface => {
593                    Self::configure_surface(surface_state, render_state, &self.configuration);
594                    return (vsync_sec, None);
595                }
596                SurfaceErrorAction::SkipFrame => {
597                    return (vsync_sec, None);
598                }
599            },
600        };
601
602        {
603            let renderer = render_state.renderer.read();
604            let frame_view = if capture {
605                Self::update_capture_state(
606                    &mut self.screen_capture_state,
607                    &output_frame,
608                    render_state,
609                );
610                self.screen_capture_state
611                    .as_ref()
612                    .map_or_else(
613                        || &output_frame.texture,
614                        |capture_state| &capture_state.texture,
615                    )
616                    .create_view(&wgpu::TextureViewDescriptor::default())
617            } else {
618                output_frame
619                    .texture
620                    .create_view(&wgpu::TextureViewDescriptor::default())
621            };
622
623            let (view, resolve_target) = (self.msaa_samples > 1)
624                .then_some(self.msaa_texture_view.get(&viewport_id))
625                .flatten()
626                .map_or((&frame_view, None), |texture_view| {
627                    (texture_view, Some(&frame_view))
628                });
629
630            let render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
631                label: Some("egui_render"),
632                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
633                    view,
634                    resolve_target,
635                    ops: wgpu::Operations {
636                        load: wgpu::LoadOp::Clear(wgpu::Color {
637                            r: clear_color[0] as f64,
638                            g: clear_color[1] as f64,
639                            b: clear_color[2] as f64,
640                            a: clear_color[3] as f64,
641                        }),
642                        store: wgpu::StoreOp::Store,
643                    },
644                })],
645                depth_stencil_attachment: self.depth_texture_view.get(&viewport_id).map(|view| {
646                    wgpu::RenderPassDepthStencilAttachment {
647                        view,
648                        depth_ops: Some(wgpu::Operations {
649                            load: wgpu::LoadOp::Clear(1.0),
650                            // It is very unlikely that the depth buffer is needed after egui finished rendering
651                            // so no need to store it. (this can improve performance on tiling GPUs like mobile chips or Apple Silicon)
652                            store: wgpu::StoreOp::Discard,
653                        }),
654                        stencil_ops: None,
655                    }
656                }),
657                timestamp_writes: None,
658                occlusion_query_set: None,
659            });
660
661            // Forgetting the pass' lifetime means that we are no longer compile-time protected from
662            // runtime errors caused by accessing the parent encoder before the render pass is dropped.
663            // Since we don't pass it on to the renderer, we should be perfectly safe against this mistake here!
664            renderer.render(
665                &mut render_pass.forget_lifetime(),
666                clipped_primitives,
667                &screen_descriptor,
668            );
669        }
670
671        {
672            let mut renderer = render_state.renderer.write();
673            for id in &textures_delta.free {
674                renderer.free_texture(id);
675            }
676        }
677
678        let encoded = {
679            crate::profile_scope!("CommandEncoder::finish");
680            encoder.finish()
681        };
682
683        // Submit the commands: both the main buffer and user-defined ones.
684        {
685            crate::profile_scope!("Queue::submit");
686            // wgpu doesn't document where vsync can happen. Maybe here?
687            let start = web_time::Instant::now();
688            render_state
689                .queue
690                .submit(user_cmd_bufs.into_iter().chain([encoded]));
691            vsync_sec += start.elapsed().as_secs_f32();
692        };
693
694        let screenshot = if capture {
695            self.screen_capture_state
696                .as_ref()
697                .and_then(|screen_capture_state| {
698                    Self::read_screen_rgba(screen_capture_state, render_state, &output_frame)
699                })
700        } else {
701            None
702        };
703
704        {
705            crate::profile_scope!("present");
706            // wgpu doesn't document where vsync can happen. Maybe here?
707            let start = web_time::Instant::now();
708            output_frame.present();
709            vsync_sec += start.elapsed().as_secs_f32();
710        }
711
712        (vsync_sec, screenshot)
713    }
