18/// The data that egui persists between frames.
19///
20/// This includes window positions and sizes,
21/// how far the user has scrolled in a [`ScrollArea`](crate::ScrollArea) etc.
22///
23/// If you want this to persist when closing your app, you should serialize [`Memory`] and store it.
24/// For this you need to enable the `persistence`.
25///
26/// If you want to store data for your widgets, you should look at [`Memory::data`]
27#[derive(Clone, Debug)]
28#[cfg_attr(feature = "persistence", derive(serde::Deserialize, serde::Serialize))]
29#[cfg_attr(feature = "persistence", serde(default))]
30pub struct Memory {
31    /// Global egui options.
32    pub options: Options,
33
34    /// This map stores some superficial state for all widgets with custom [`Id`]s.
35    ///
36    /// This includes storing whether a [`crate::CollapsingHeader`] is open, how far scrolled a
37    /// [`crate::ScrollArea`] is, where the cursor in a [`crate::TextEdit`] is, etc.
38    ///
39    /// This is NOT meant to store any important data. Store that in your own structures!
40    ///
41    /// Each read clones the data, so keep your values cheap to clone.
42    /// If you want to store a lot of data, you should wrap it in `Arc<Mutex<â€¦>>` so it is cheap to clone.
43    ///
44    /// This will be saved between different program runs if you use the `persistence` feature.
45    ///
46    /// To store a state common for all your widgets (a singleton), use [`Id::NULL`] as the key.
47    pub data: crate::util::IdTypeMap,
48
49    // ------------------------------------------
50    /// Can be used to cache computations from one frame to another.
51    ///
52    /// This is for saving CPU time when you have something that may take 1-100ms to compute.
53    /// Very slow operations (>100ms) should instead be done async (i.e. in another thread)
54    /// so as not to lock the UI thread.
55    ///
56    /// ```
57    /// use egui::util::cache::{ComputerMut, FrameCache};
58    ///
59    /// #[derive(Default)]
60    /// struct CharCounter {}
61    /// impl ComputerMut<&str, usize> for CharCounter {
62    ///     fn compute(&mut self, s: &str) -> usize {
63    ///         s.chars().count() // you probably want to cache something more expensive than this
64    ///     }
65    /// }
66    /// type CharCountCache<'a> = FrameCache<usize, CharCounter>;
67    ///
68    /// # let mut ctx = egui::Context::default();
69    /// ctx.memory_mut(|mem| {
70    ///     let cache = mem.caches.cache::<CharCountCache<'_>>();
71    ///     assert_eq!(cache.get("hello"), 5);
72    /// });
73    /// ```
74    #[cfg_attr(feature = "persistence", serde(skip))]
75    pub caches: crate::util::cache::CacheStorage,
76
77    // ------------------------------------------
78    /// new fonts that will be applied at the start of the next frame
79    #[cfg_attr(feature = "persistence", serde(skip))]
80    pub(crate) new_font_definitions: Option<epaint::text::FontDefinitions>,
81
82    // Current active viewport
83    #[cfg_attr(feature = "persistence", serde(skip))]
84    pub(crate) viewport_id: ViewportId,
85
86    /// Which popup-window is open (if any)?
87    /// Could be a combo box, color picker, menu, etc.
88    #[cfg_attr(feature = "persistence", serde(skip))]
89    popup: Option<Id>,
90
91    #[cfg_attr(feature = "persistence", serde(skip))]
92    everything_is_visible: bool,
93
94    /// Transforms per layer
95    pub layer_transforms: HashMap<LayerId, TSTransform>,
96
97    // -------------------------------------------------
98    // Per-viewport:
99    areas: ViewportIdMap<Areas>,
100
101    #[cfg_attr(feature = "persistence", serde(skip))]
102    pub(crate) interactions: ViewportIdMap<InteractionState>,
103
104    #[cfg_attr(feature = "persistence", serde(skip))]
105    pub(crate) focus: ViewportIdMap<Focus>,
106}

129#[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
130enum FocusDirection {
131    /// Select the widget closest above the current focused widget.
132    Up,
133
134    /// Select the widget to the right of the current focused widget.
135    Right,
136
137    /// Select the widget below the current focused widget.
138    Down,
139
140    /// Select the widget to the left of the current focused widget.
141    Left,
142
143    /// Select the previous widget that had focus.
144    Previous,
145
146    /// Select the next widget that wants focus.
147    Next,
148
149    /// Don't change focus.
150    #[default]
151    None,
152}
153
154impl FocusDirection {
155    fn is_cardinal(&self) -> bool {
156        match self {
157            Self::Up | Self::Right | Self::Down | Self::Left => true,
158
159            Self::Previous | Self::Next | Self::None => false,
160        }
161    }
162}

462/// The state of the interaction in egui,
463/// i.e. what is being dragged.
464///
465/// Say there is a button in a scroll area.
466/// If the user clicks the button, the button should click.
467/// If the user drags the button we should scroll the scroll area.
468/// Therefore, when the mouse is pressed, we register both the button
469/// and the scroll area (as `click_id`/`drag_id`).
470/// If the user releases the button without moving the mouse, we register it as a click on `click_id`.
471/// If the cursor moves too much, we clear the `click_id` and start passing move events to `drag_id`.
472#[derive(Clone, Debug, Default)]
473pub(crate) struct InteractionState {
474    /// A widget interested in clicks that has a mouse press on it.
475    pub potential_click_id: Option<Id>,
476
477    /// A widget interested in drags that has a mouse press on it.
478    ///
479    /// Note that this is set as soon as the mouse is pressed,
480    /// so the widget may not yet be marked as "dragged"
481    /// as that can only happen after the mouse has moved a bit
482    /// (at least if the widget is interesated in both clicks and drags).
483    pub potential_drag_id: Option<Id>,
484}

486/// Keeps tracks of what widget has keyboard focus
487#[derive(Clone, Debug, Default)]
488pub(crate) struct Focus {
489    /// The widget with keyboard focus (i.e. a text input field).
490    focused_widget: Option<FocusWidget>,
491
492    /// The ID of a widget that had keyboard focus during the previous frame.
493    id_previous_frame: Option<Id>,
494
495    /// The ID of a widget to give the focus to in the next frame.
496    id_next_frame: Option<Id>,
497
498    #[cfg(feature = "accesskit")]
499    id_requested_by_accesskit: Option<accesskit::NodeId>,
500
501    /// If set, the next widget that is interested in focus will automatically get it.
502    /// Probably because the user pressed Tab.
503    give_to_next: bool,
504

506    last_interested: Option<Id>,
507
508    /// Set when looking for widget with navigational keys like arrows, tab, shift+tab.
509    focus_direction: FocusDirection,
510
511    /// A cache of widget IDs that are interested in focus with their corresponding rectangles.
512    focus_widgets_cache: IdMap<Rect>,
513}

531impl InteractionState {

538impl Focus {
539    /// Which widget currently has keyboard focus?
540    pub fn focused(&self) -> Option<Id> {
541        self.focused_widget.as_ref().map(|w| w.id)
542    }
543
544    fn begin_pass(&mut self, new_input: &crate::data::input::RawInput) {
545        self.id_previous_frame = self.focused();
546        if let Some(id) = self.id_next_frame.take() {
547            self.focused_widget = Some(FocusWidget::new(id));
548        }
549        let event_filter = self.focused_widget.map(|w| w.filter).unwrap_or_default();
550
551        #[cfg(feature = "accesskit")]
552        {
553            self.id_requested_by_accesskit = None;
554        }
555
556        self.focus_direction = FocusDirection::None;
557
558        for event in &new_input.events {
559            if !event_filter.matches(event) {
560                if let crate::Event::Key {
561                    key,
562                    pressed: true,
563                    modifiers,
564                    ..
565                } = event
566                {
567                    if let Some(cardinality) = match key {
568                        crate::Key::ArrowUp => Some(FocusDirection::Up),
569                        crate::Key::ArrowRight => Some(FocusDirection::Right),
570                        crate::Key::ArrowDown => Some(FocusDirection::Down),
571                        crate::Key::ArrowLeft => Some(FocusDirection::Left),
572
573                        crate::Key::Tab => {
574                            if modifiers.shift {
575                                Some(FocusDirection::Previous)
576                            } else {
577                                Some(FocusDirection::Next)
578                            }
579                        }
580                        crate::Key::Escape => {
581                            self.focused_widget = None;
582                            Some(FocusDirection::None)
583                        }
584                        _ => None,
585                    } {
586                        self.focus_direction = cardinality;
587                    }
588                }
589            }
590
591            #[cfg(feature = "accesskit")]
592            {
593                if let crate::Event::AccessKitActionRequest(accesskit::ActionRequest {
594                    action: accesskit::Action::Focus,
595                    target,
596                    data: None,
597                }) = event
598                {
599                    self.id_requested_by_accesskit = Some(*target);
600                }
601            }
602        }
603    }
604
605    pub(crate) fn end_pass(&mut self, used_ids: &IdMap<Rect>) {
606        if self.focus_direction.is_cardinal() {
607            if let Some(found_widget) = self.find_widget_in_direction(used_ids) {
608                self.focused_widget = Some(FocusWidget::new(found_widget));
609            }
610        }
611
612        if let Some(focused_widget) = self.focused_widget {
613            // Allow calling `request_focus` one frame and not using it until next frame
614            let recently_gained_focus = self.id_previous_frame != Some(focused_widget.id);
615
616            if !recently_gained_focus && !used_ids.contains_key(&focused_widget.id) {
617                // Dead-mans-switch: the widget with focus has disappeared!
618                self.focused_widget = None;
619            }
620        }
621    }
622
623    pub(crate) fn had_focus_last_frame(&self, id: Id) -> bool {
624        self.id_previous_frame == Some(id)
625    }
626
627    fn interested_in_focus(&mut self, id: Id) {
628        #[cfg(feature = "accesskit")]
629        {
630            if self.id_requested_by_accesskit == Some(id.accesskit_id()) {
631                self.focused_widget = Some(FocusWidget::new(id));
632                self.id_requested_by_accesskit = None;
633                self.give_to_next = false;
634                self.reset_focus();
635            }
636        }
637
638        // The rect is updated at the end of the frame.
639        self.focus_widgets_cache
640            .entry(id)
641            .or_insert(Rect::EVERYTHING);
642
643        if self.give_to_next && !self.had_focus_last_frame(id) {
644            self.focused_widget = Some(FocusWidget::new(id));
645            self.give_to_next = false;
646        } else if self.focused() == Some(id) {
647            if self.focus_direction == FocusDirection::Next {
648                self.focused_widget = None;
649                self.give_to_next = true;
650                self.reset_focus();
651            } else if self.focus_direction == FocusDirection::Previous {
652                self.id_next_frame = self.last_interested; // frame-delay so gained_focus works
653                self.reset_focus();
654            }
655        } else if self.focus_direction == FocusDirection::Next
656            && self.focused_widget.is_none()
657            && !self.give_to_next
658        {
659            // nothing has focus and the user pressed tab - give focus to the first widgets that wants it:
660            self.focused_widget = Some(FocusWidget::new(id));
661            self.reset_focus();
662        } else if self.focus_direction == FocusDirection::Previous
663            && self.focused_widget.is_none()
664            && !self.give_to_next
665        {
666            // nothing has focus and the user pressed Shift+Tab - give focus to the last widgets that wants it:
667            self.focused_widget = self.last_interested.map(FocusWidget::new);
668            self.reset_focus();
669        }
670
671        self.last_interested = Some(id);
672    }
673
674    fn reset_focus(&mut self) {
675        self.focus_direction = FocusDirection::None;
676    }
677
678    fn find_widget_in_direction(&mut self, new_rects: &IdMap<Rect>) -> Option<Id> {
679        // NOTE: `new_rects` here include some widgets _not_ interested in focus.
680
681        /// * negative if `a` is left of `b`
682        /// * positive if `a` is right of `b`
683        /// * zero if the ranges overlap significantly
684        fn range_diff(a: Rangef, b: Rangef) -> f32 {
685            let has_significant_overlap = a.intersection(b).span() >= 0.5 * b.span().min(a.span());
686            if has_significant_overlap {
687                0.0
688            } else {
689                a.center() - b.center()
690            }
691        }
692
693        let current_focused = self.focused_widget?;
694
695        // In what direction we are looking for the next widget.
696        let search_direction = match self.focus_direction {
697            FocusDirection::Up => Vec2::UP,
698            FocusDirection::Right => Vec2::RIGHT,
699            FocusDirection::Down => Vec2::DOWN,
700            FocusDirection::Left => Vec2::LEFT,
701            _ => {
702                return None;
703            }
704        };
705
706        // Update cache with new rects
707        self.focus_widgets_cache.retain(|id, old_rect| {
708            if let Some(new_rect) = new_rects.get(id) {
709                *old_rect = *new_rect;
710                true // Keep the item
711            } else {
712                false // Remove the item
713            }
714        });
715
716        let current_rect = self.focus_widgets_cache.get(&current_focused.id)?;
717
718        let mut best_score = std::f32::INFINITY;
719        let mut best_id = None;
720
721        for (candidate_id, candidate_rect) in &self.focus_widgets_cache {
722            if *candidate_id == current_focused.id {
723                continue;
724            }
725
726            // There is a lot of room for improvement here.
727            let to_candidate = vec2(
728                range_diff(candidate_rect.x_range(), current_rect.x_range()),
729                range_diff(candidate_rect.y_range(), current_rect.y_range()),
730            );
731
732            let acos_angle = to_candidate.normalized().dot(search_direction);
733
734            // Only interested in widgets that fall in a 90Â° cone (Â±45Â°)
735            // of the search direction.
736            let is_in_search_cone = 0.5_f32.sqrt() <= acos_angle;
737            if is_in_search_cone {
738                let distance = to_candidate.length();
739
740                // There is a lot of room for improvement here.
741                let score = distance / (acos_angle * acos_angle);
742
743                if score < best_score {
744                    best_score = score;
745                    best_id = Some(*candidate_id);
746                }
747            }
748        }
749
750        best_id
751    }
752}

1035/// Keeps track of [`Area`](crate::containers::area::Area)s, which are free-floating [`Ui`](crate::Ui)s.
1036/// These [`Area`](crate::containers::area::Area)s can be in any [`Order`].
1037#[derive(Clone, Debug, Default)]
1038#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
1039#[cfg_attr(feature = "serde", serde(default))]
1040pub struct Areas {
1041    areas: IdMap<area::AreaState>,
1042
1043    /// Back-to-front,  top is last.
1044    order: Vec<LayerId>,
1045
1046    visible_last_frame: ahash::HashSet<LayerId>,
1047    visible_current_frame: ahash::HashSet<LayerId>,
1048
1049    /// When an area wants to be on top, it is assigned here.
1050    /// This is used to reorder the layers at the end of the frame.
1051    /// If several layers want to be on top, they will keep their relative order.
1052    /// This means closing three windows and then reopening them all in one frame
1053    /// results in them being sent to the top and keeping their previous internal order.
1054    wants_to_be_on_top: ahash::HashSet<LayerId>,
1055
1056    /// List of sublayers for each layer.
1057    ///
1058    /// When a layer has sublayers, they are moved directly above it in the ordering.
1059    sublayers: ahash::HashMap<LayerId, HashSet<LayerId>>,
1060}
1061
1062impl Areas {
1063    pub(crate) fn count(&self) -> usize {
1064        self.areas.len()
1065    }
1066
1067    pub(crate) fn get(&self, id: Id) -> Option<&area::AreaState> {
1068        self.areas.get(&id)
1069    }
1070
1071    /// Back-to-front, top is last.
1072    pub(crate) fn order(&self) -> &[LayerId] {
1073        &self.order
1074    }
1075
1076    /// For each layer, which [`Self::order`] is it in?
1077    pub(crate) fn order_map(&self) -> HashMap<LayerId, usize> {
1078        self.order
1079            .iter()
1080            .enumerate()
1081            .map(|(i, id)| (*id, i))
1082            .collect()
1083    }
1084
1085    pub(crate) fn set_state(&mut self, layer_id: LayerId, state: area::AreaState) {
1086        self.visible_current_frame.insert(layer_id);
1087        self.areas.insert(layer_id.id, state);
1088        if !self.order.iter().any(|x| *x == layer_id) {
1089            self.order.push(layer_id);
1090        }
1091    }
1092
1093    /// Top-most layer at the given position.
1094    pub fn layer_id_at(
1095        &self,
1096        pos: Pos2,
1097        layer_transforms: &HashMap<LayerId, TSTransform>,
1098    ) -> Option<LayerId> {
1099        for layer in self.order.iter().rev() {
1100            if self.is_visible(layer) {
1101                if let Some(state) = self.areas.get(&layer.id) {
1102                    let mut rect = state.rect();
1103                    if state.interactable {
1104                        if let Some(transform) = layer_transforms.get(layer) {
1105                            rect = *transform * rect;
1106                        }
1107
1108                        if rect.contains(pos) {
1109                            return Some(*layer);
1110                        }
1111                    }
1112                }
1113            }
1114        }
1115        None
1116    }
1117
1118    pub fn visible_last_frame(&self, layer_id: &LayerId) -> bool {
1119        self.visible_last_frame.contains(layer_id)
1120    }
1121
1122    pub fn is_visible(&self, layer_id: &LayerId) -> bool {
1123        self.visible_last_frame.contains(layer_id) || self.visible_current_frame.contains(layer_id)
1124    }
1125
1126    pub fn visible_layer_ids(&self) -> ahash::HashSet<LayerId> {
1127        self.visible_last_frame
1128            .iter()
1129            .copied()
1130            .chain(self.visible_current_frame.iter().copied())
1131            .collect()
1132    }
1133
1134    pub(crate) fn visible_windows(&self) -> impl Iterator<Item = (LayerId, &area::AreaState)> {
1135        self.visible_layer_ids()
1136            .into_iter()
1137            .filter(|layer| layer.order == crate::Order::Middle)
1138            .filter(|&layer| !self.is_sublayer(&layer))
1139            .filter_map(|layer| Some((layer, self.get(layer.id)?)))
1140    }
1141
1142    pub fn move_to_top(&mut self, layer_id: LayerId) {
1143        self.visible_current_frame.insert(layer_id);
1144        self.wants_to_be_on_top.insert(layer_id);
1145
1146        if !self.order.iter().any(|x| *x == layer_id) {
1147            self.order.push(layer_id);
1148        }
1149    }
1150
1151    /// Mark the `child` layer as a sublayer of `parent`.
1152    ///
1153    /// Sublayers are moved directly above the parent layer at the end of the frame. This is mainly
1154    /// intended for adding a new [Area](crate::Area) inside a [Window](crate::Window).
1155    ///
1156    /// This currently only supports one level of nesting. If `parent` is a sublayer of another
1157    /// layer, the behavior is unspecified.
1158    pub fn set_sublayer(&mut self, parent: LayerId, child: LayerId) {
1159        self.sublayers.entry(parent).or_default().insert(child);
1160    }
1161
1162    pub fn top_layer_id(&self, order: Order) -> Option<LayerId> {
1163        self.order
1164            .iter()
1165            .filter(|layer| layer.order == order && !self.is_sublayer(layer))
1166            .last()
1167            .copied()
1168    }
1169
1170    pub(crate) fn is_sublayer(&self, layer: &LayerId) -> bool {
1171        self.sublayers
1172            .iter()
1173            .any(|(_, children)| children.contains(layer))
1174    }
1175
1176    pub(crate) fn end_pass(&mut self) {
1177        let Self {
1178            visible_last_frame,
1179            visible_current_frame,
1180            order,
1181            wants_to_be_on_top,
1182            sublayers,
1183            ..
1184        } = self;
1185
1186        std::mem::swap(visible_last_frame, visible_current_frame);
1187        visible_current_frame.clear();
1188        order.sort_by_key(|layer| (layer.order, wants_to_be_on_top.contains(layer)));
1189        wants_to_be_on_top.clear();
1190        // For all layers with sublayers, put the sublayers directly after the parent layer:
1191        let sublayers = std::mem::take(sublayers);
1192        for (parent, children) in sublayers {
1193            let mut moved_layers = vec![parent];
1194            order.retain(|l| {
1195                if children.contains(l) {
1196                    moved_layers.push(*l);
1197                    false
1198                } else {
1199                    true
1200                }
1201            });
1202            let Some(parent_pos) = order.iter().position(|l| l == &parent) else {
1203                continue;
1204            };
1205            order.splice(parent_pos..=parent_pos, moved_layers);
1206        }
1207    }
1208}
